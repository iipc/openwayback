3   import javax.servlet.http.HttpServletResponse;
4   
5   import org.archive.wayback.archivalurl.ArchivalUrl;
6   import org.archive.wayback.core.CaptureSearchResult;
7   import org.archive.wayback.core.CaptureSearchResults;
8   import org.archive.wayback.core.WaybackRequest;

12  /**
13   * Sub-class of {@link BetterRequestException} for instructing a redirect to
14   * ArchivalUrl for the URL on different timestamp.
15   * <p>
16   * This class generates additional HTTP headers for Memento support.
17   * </p>
18   */
19  public class BetterReplayRequestException extends BetterRequestException {
20  
21  	private static final long serialVersionUID = -873414298713087775L;
22  
23  	private String targetURI;
24  	private String timestamp;
25  	private CaptureSearchResults captures;
26  
27  	/**
28  	 * Initializes with capture information and additional information for
29  	 * generating navigation headers (Memento headers).
30  	 * @param targetURI target URI, typically capture's {@code originalUrl}
31  	 * @param timestamp timestamp, typically capture's  {@code cpatureTimestamp}
32  	 * @param results Other captures for targetURI.
33  	 */
34  	public BetterReplayRequestException(String targetURI, String timestamp,
35  			CaptureSearchResults captures) {
36  		// hacky - this is okay, because #generateResponse() does not use
37  		// betterURI member.
38  		// TODO: better to define a common superclass of BetterRequestException
39  		// and BetterReplayRequestException,
40  		// that defines abstract generateResponse() method.
41  		super(null);
42  		this.targetURI = targetURI;
43  		this.timestamp = timestamp;
44  		this.captures = captures;
45  	}
46  
47  	/**
48  	 * Initializes with {@code originalUrl} and {@code captureTimestamp} from {@link CaptureSearchResult}.
49  	 * @param capture Capture to redirect to.
50  	 * @param captures Other captures for targetURI.
51  	 */
52  	public BetterReplayRequestException(CaptureSearchResult capture, CaptureSearchResults captures) {
53  		this(capture.getOriginalUrl(), capture.getCaptureTimestamp(), captures);
54  	}
55  
56  	@Override
57  	public void generateResponse(HttpServletResponse response,
58  			WaybackRequest wbRequest) {
59  		AccessPoint accessPoint = wbRequest.getAccessPoint();
60  		String replayURL = accessPoint.getUriConverter().makeReplayURI(
61  			ArchivalUrl.getDateSpec(wbRequest, timestamp), targetURI);
62  		response.setStatus(HttpServletResponse.SC_FOUND);
63  		response.setHeader("Location", replayURL);
64  
65  		if (wbRequest.isMementoEnabled()) {
66  			// both URI-G redirect response and intermediate resource response
67  			// MUST NOT have Memento-Datetime header.
68  			if (wbRequest.isMementoTimegate()) {
69  				// URI-G redirect response
70  				if (accessPoint.getMementoHandler() != null) {
71  					accessPoint.getMementoHandler().addTimegateHeaders(response, captures, wbRequest, true);
72  				} else {
73  					MementoUtils.addTimegateHeaders(response, captures, wbRequest, true);
74  				}
75  			} else {
76  				// intermediate resource that redirects to URL-M
77  				// should the second argument wbRequest.getRequestUrl() or this.targetURI?
78  				// there are different approaches in the code.
79  				MementoUtils.addOrigHeader(response, this.targetURI);
80  			}
81  		}
82  	}
83  
84  }

617 	 * @deprecated 1.8.1 use {@link UIResults#getFormatter()}.

619 	public StringFormatter getFormatter() {
620 		if (formatter == null) {
621 			Locale l = locale;
622 			if (l == null) {
623 				l = Locale.getAvailableLocales()[0];
624 			}
625 			ResourceBundle b = ResourceBundle.getBundle(UIResults.UI_RESOURCE_BUNDLE_NAME);
626 			formatter = new StringFormatter(b, l);
627 		}
628 		return formatter;
629 	}

666 	private int getInt(String key) {
667 		String value = get(key);
668 		if(value == null) {
669 			return -1;
670 		}
671 		return Integer.parseInt(value);
672 	}	
673 	private void setInt(String key, int value) {
674 		put(key,String.valueOf(value));
675 	}	
676 	private boolean getBoolean(String key) {
677 		String value = get(key);
678 		return(value != null && value.equals(REQUEST_YES));
679 	}

695 	private boolean isRequestType(String requestType) {
696 		String type = get(REQUEST_TYPE);
697 		if(type != null && type.equals(requestType)) {
698 			return true;
699 		}
700 		return false;
701 	}

747 	public void setRequestUrl(String urlStr) {
748 
749 		// This looks a little confusing: We're trying to fixup an incoming
750 		// request URL that starts with:

752 		// so it becomes:
753 		//       "http://www.archive.org"
754 		// (note the missing second "/" in the first)
755 		//

757 		// is known, adding an implied "http://" scheme if there doesn't appear
758 		// to be a scheme..

760 		if (!urlStr.startsWith(UrlOperations.HTTP_SCHEME)) {

763 	    	} else {
764 	    		if(UrlOperations.urlToScheme(urlStr) == null) {
765 	    			urlStr = UrlOperations.HTTP_SCHEME + urlStr;
766 	    		}
767 	    	}
768 	    }
769         put(REQUEST_URL, urlStr);
770 	}

1081 	/**
1082 	 * extract REFERER, remote IP and authorization information from the
1083 	 * HttpServletRequest
1084 	 * 
1085 	 * @param httpRequest
1086 	 * @throws BadQueryException 
1087 	 */
1088 	public void extractHttpRequestInfo(HttpServletRequest httpRequest) {
1089 		
1090 		putUnlessNull(REQUEST_REFERER_URL, httpRequest.getHeader("REFERER"));
1091 		
1092 		String remoteAddr = httpRequest.getHeader("X-Forwarded-For");
1093 		if (remoteAddr == null) {
1094 			remoteAddr = httpRequest.getRemoteAddr();
1095 		}
1096 		putUnlessNull(REQUEST_REMOTE_ADDRESS, remoteAddr);
1097 		
1098 		// Check for AJAX
1099 		String x_req_with = httpRequest.getHeader("X-Requested-With");
1100 		if (x_req_with != null) {
1101 			if (x_req_with.equals("XMLHttpRequest")) {
1102 				this.setAjaxRequest(true);
1103 			}
1104 		} else if (this.getRefererUrl() != null && httpRequest.getParameter("ajaxpipe") != null) {
1105 			this.setAjaxRequest(true);
1106 		}
1107 		
1108 		if (isMementoEnabled()) {
1109 			// Check for Memento Accept-Datetime
1110 			String acceptDateTime = httpRequest.getHeader(MementoUtils.ACCEPT_DATETIME);
1111 			if (acceptDateTime != null) {
1112 				this.setMementoAcceptDatetime(true);
1113 			}
1114 		}
1115 		
1116 		putUnlessNull(REQUEST_WAYBACK_HOSTNAME, httpRequest.getLocalName());
1117 		putUnlessNull(REQUEST_AUTH_TYPE, httpRequest.getAuthType());
1118 		putUnlessNull(REQUEST_REMOTE_USER, httpRequest.getRemoteUser());
1119 		
1120 		putUnlessNull(REQUEST_AUTHORIZATION,
1121 				httpRequest.getHeader(REQUEST_AUTHORIZATION));
1122 		putUnlessNull(REQUEST_WAYBACK_PORT, 
1123 				String.valueOf(httpRequest.getLocalPort()));
1124 		putUnlessNull(REQUEST_WAYBACK_CONTEXT, httpRequest.getContextPath());
1125 
1126 		Locale l = null;
1127 		if (accessPoint != null) {
1128 			l = accessPoint.getLocale();
1129 		}
1130 		if (l == null) {
1131 			l = httpRequest.getLocale();
1132 		}
1133 		//setLocale(l);
1134 		this.locale = l;
1135 
1136 		putUnlessNull(REQUEST_LOCALE_LANG,l.getDisplayLanguage());
1137 
1138 		Cookie[] cookies = httpRequest.getCookies();
1139 		if(cookies != null) {
1140 			for(Cookie cookie : cookies) {
1141 				String name = cookie.getName();
1142 				String value = cookie.getValue();
1143 				String oldVal = get(name);
1144 				if(oldVal == null || oldVal.length() == 0) {
1145 					put(name,value);
1146 				}
1147 			}
1148 		}
1149 	}

1159 	/**
1160 	 * @param pageNum
1161 	 * @return String hex-encoded GET CGI arguments which will duplicate the
1162 	 *         same request, but for page 'pageNum' of the results
1163 	 */
1164 	public String getQueryArguments(int pageNum) {
1165 		int numPerPage = resultsPerPage;
1166 
1167 		try {
1168 			StringBuffer queryString = new StringBuffer("");
1169 			Iterator<String> itr = filters.keySet().iterator();
1170 			while(itr.hasNext()) {
1171 				String key = itr.next();
1172 				boolean isStandard = false;
1173 				for(int i=0; i<standardHeaders.length; i++) {
1174 					if(standardHeaders[i].equals(key)) {
1175 						isStandard = true;
1176 						break;
1177 					}
1178 				}
1179 				if(isStandard) continue;
1180 				String val = filters.get(key);
1181 				if(val == null) continue;
1182 				if (queryString.length() > 0) {
1183 					queryString.append(" ");
1184 				}
1185 				key = URLEncoder.encode(key, "UTF-8");
1186 				val = URLEncoder.encode(val, "UTF-8");
1187 				queryString.append(key + ":" + val);
1188 			}
1189 			String escapedQuery = queryString.toString();
1190 
1191 			escapedQuery = URLEncoder.encode(escapedQuery, "UTF-8");
1192 
1193 			return OpenSearchRequestParser.SEARCH_QUERY + "=" + escapedQuery + "&"
1194 				+ OpenSearchRequestParser.SEARCH_RESULTS + "=" + numPerPage + "&"
1195 				+ OpenSearchRequestParser.START_PAGE + "=" + pageNum;
1196 
1197 		} catch (UnsupportedEncodingException e) {
1198 			// oops.. what to do?
1199 			e.printStackTrace();
1200                         throw new RuntimeException(e);
1201 		}
1202 	}

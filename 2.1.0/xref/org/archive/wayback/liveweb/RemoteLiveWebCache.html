52  public class RemoteLiveWebCache implements LiveWebCache {
53  	private static final Logger LOGGER = Logger.getLogger(
54  			RemoteLiveWebCache.class.getName());
55  
56      protected MultiThreadedHttpConnectionManager connectionManager = null;
57      protected HostConfiguration hostConfiguration = null;
58      protected HttpClient http = null; 
59      protected String requestPrefix = null;
60  
61      /**
62       * 
63       */
64      public RemoteLiveWebCache() {
65      	connectionManager = new MultiThreadedHttpConnectionManager();
66      	hostConfiguration = new HostConfiguration();
67  		HttpClientParams params = new HttpClientParams();
68          params.setParameter(HttpClientParams.RETRY_HANDLER, new NoRetryHandler());
69      	http = new HttpClient(params,connectionManager);
70      	http.setHostConfiguration(hostConfiguration);
71      }

73      /* (non-Javadoc)
74  	 * @see org.archive.wayback.liveweb.LiveWebCache#getCachedResource(java.net.URL, long, boolean)
75  	 */
76  	public Resource getCachedResource(URL url, long maxCacheMS,
77  			boolean bUseOlder) throws LiveDocumentNotAvailableException,
78  			LiveWebCacheUnavailableException, LiveWebTimeoutException, IOException {
79  		String urlString = url.toExternalForm();
80  		
81  		if (requestPrefix != null) {
82  			urlString = requestPrefix + urlString;
83  		}
84  		
85  		HttpMethod method = null;
86  		try {
87  			method = new GetMethod(urlString);
88  		} catch(IllegalArgumentException e) {
89  			LOGGER.warning("Bad URL for live web fetch:" + urlString);
90  			throw new LiveDocumentNotAvailableException("Url:" + urlString +
91  					"does not look like an URL?");
92  		}
93  		boolean success = false;
94  	    try {
95  	    	int status = http.executeMethod(method);
96  	    	if(status == 200) {
97  
98  	    		ByteArrayInputStream bais = new ByteArrayInputStream(method.getResponseBody());
99  	    		ARCRecord r = new ARCRecord(
100 	    				new GZIPInputStream(bais),
101 	    				"id",0L,false,false,true);
102 	    		ArcResource ar = (ArcResource) 
103 	    			ResourceFactory.ARCArchiveRecordToResource(r, null);
104 	    		if(ar.getStatusCode() == 502) {
105 	    			throw new LiveDocumentNotAvailableException(urlString);
106 	    		} else if(ar.getStatusCode() == 504) {
107 	    			throw new LiveWebTimeoutException("Timeout:" + urlString);
108 	    		}
109 	    		success = true;
110 	    		return ar;
111 	    		
112 	    	} else {
113 	    		throw new LiveWebCacheUnavailableException(urlString);
114 	    	}
115 
116 	    } catch (ResourceNotAvailableException e) {
117     		throw new LiveDocumentNotAvailableException(urlString);
118 
119 	    } catch (NoHttpResponseException e) {
120 
121 	    	throw new LiveWebCacheUnavailableException("No Http Response for "
122 	    			+ urlString);
123 
124 	    } catch (ConnectException e) {
125     		throw new LiveWebCacheUnavailableException(e.getLocalizedMessage() 
126     				+ " : " + urlString);
127 	    } catch (SocketException e) {
128     		throw new LiveWebCacheUnavailableException(e.getLocalizedMessage() 
129     				+ " : " + urlString);
130 	    } catch (SocketTimeoutException e) {
131     		throw new LiveWebTimeoutException(e.getLocalizedMessage() 
132     				+ " : " + urlString);
133 	    } catch(ConnectTimeoutException e) {
134     		throw new LiveWebTimeoutException(e.getLocalizedMessage() 
135     				+ " : " + urlString);	    	
136 		} finally {
137 			if (!success) {
138 				method.abort();
139 			}
140 	    	method.releaseConnection();
141 	    }
142 	}

152     /**
153      * @param hostPort to proxy requests through - ex. "localhost:3128"
154      */
155     public void setProxyHostPort(String hostPort) {
156     	int colonIdx = hostPort.indexOf(':');
157     	if(colonIdx > 0) {
158     		String host = hostPort.substring(0,colonIdx);
159     		int port = Integer.valueOf(hostPort.substring(colonIdx+1));
160     		
161 //            http.getHostConfiguration().setProxy(host, port);
162     		hostConfiguration.setProxy(host, port);
163     	}
164     }

260 				+ "</html>";

1003     public String doEndToEnd(String input) throws Exception {
1004 		ByteArrayInputStream bais = new ByteArrayInputStream(input.getBytes(charSet));
1005 		
1006 		// To make sure we get the length, we have to buffer it all up...
1007 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
1008 		context.setOutputStream(baos);
1009 		context.setJspExec(jspExec);
1010 
1011 		// and finally, parse, using the special lexer that knows how to
1012 		// handle javascript blocks containing unescaped HTML entities:
1013 		Page lexPage = new Page(bais,charSet);
1014 		Lexer lexer = new Lexer(lexPage);
1015 		Lexer.STRICT_REMARKS = false;
1016 		ContextAwareLexer lex = new ContextAwareLexer(lexer, context);
1017 
1018 		Node node;
1019 		while ((node = lex.nextNode()) != null) {
1020 			delegator.handleNode(context, node);
1021 		}
1022 		delegator.handleParseComplete(context);
1023 
1024 		// At this point, baos contains the utf-8 encoded bytes of our result:
1025 		return new String(baos.toByteArray(), outputCharset);
1026 	}

1066      * {@code TagNode}.  Quirk is that {@code tagName} can also include

1068      * to update the content.
1069      * @throws Exception

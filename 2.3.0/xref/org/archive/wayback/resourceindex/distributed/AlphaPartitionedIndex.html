43  /**
44   * ResourceIndex which delegates queries to multiple partitioned remote indexes.
45   * <p>
46   * The remote index can be duplicated for failover and load balancing.
47   * Configuration is done with a map file which contains the range for each
48   * remote index partition and the urls to all instances of the partition.
49   * <p>
50   * <em>Format of mapfile:</em>
51   * <pre>
52   * &lt;partition name&gt; &lt;range start&gt; &lt;range end&gt; &lt;url to index partition&gt; [&lt;url to alternative index partition&gt; ...]
53   * </pre>
54   * <em>Example:</em>
55   * <pre>
56   * part_a  apple.com/ http://part_a1.foo.com http://part_a2.foo.com
57   * part_b apple.com/ banana.com/ http://part_b1.foo.com http://part_b2.foo.com
58   * part_c banana.com/ cups.com/ http://part_c1.foo.com http://part_c2.foo.com
59   * part_d cups.com/ zorro.com/ http://part_d1.foo.com http://part_d2.foo.com
60   * </pre>
61   * 
62   * @author brad
63   * @see org.archive.wayback.resourceindex.RemoteResourceIndex
64   */
65  public class AlphaPartitionedIndex implements ResourceIndex {
66  	private static final Logger LOGGER =
67          Logger.getLogger(AlphaPartitionedIndex.class.getName());
68  
69  
70  	/**
71  	 * config name for path where map file is found
72  	 */
73  	public static String RANGE_MAP_PATH = "resourceindex.distributed.mappath";
74  	/**
75  	 * config name for interval, in seconds, to check if the map file changed
76  	 */
77  	public static String RANGE_CHECK_INTERVAL =
78  		"resourceindex.distributed.checkinterval";
79  	private static long MS_PER_SEC = 1000;
80  	private static long DEFAULT_CHECK_INTERVAL = 100;
81  	
82  	private long lastLoadStat = 0;
83  	private long nextCheck = 0;
84  	private long checkInterval = DEFAULT_CHECK_INTERVAL;
85  	private RangeGroup groups[] = null;
86  	private String mapPath;
87  	private static Comparator<RangeGroup> comparator = 
88  		RangeGroup.getComparator();
89  	private UrlCanonicalizer canonicalizer = null;
90  
91  	public AlphaPartitionedIndex() {
92  		canonicalizer = new AggressiveUrlCanonicalizer();
93  	}

95  	@SuppressWarnings("unchecked")
96  	private void reloadMapFile() throws IOException {
97  		FlatFile ff = new FlatFile(mapPath);
98  		Iterator itr = ff.getSequentialIterator();
99  		HashMap<String,RangeGroup> newGroupsMap = 
100 			new HashMap<String,RangeGroup>();
101 		HashMap<String,RangeGroup> oldGroupsMap = 
102 			new HashMap<String,RangeGroup>();
103 		
104 		if(groups != null) {
105 			for(int i = 0; i < groups.length; i++) {
106 				oldGroupsMap.put(groups[i].getName(),groups[i]);
107 			}
108 		}
109 		while(itr.hasNext()) {
110 			String line = (String) itr.next();
111 			String[] parts = line.split(" ");
112 			if(parts.length < 3) {
113 				throw new IOException("Unparseable map line(" + line + ")");
114 			}
115 			String name = parts[0];
116 			String start = parts[1];
117 			String end = parts[2];
118 			int numMembers = parts.length - 3;
119 			String[] members = new String[numMembers];
120 			for(int i = 0; i < numMembers; i++) {
121 				members[i] = parts[3 + i];
122 			}
123 			RangeGroup group = null;
124 			if(oldGroupsMap.containsKey(name)) {
125 				group = oldGroupsMap.get(name);
126 				if(start.compareTo(group.getStart()) != 0) {
127 					throw new IOException("Change of start range in " + 
128 							mapPath + " for range " + name);
129 				}
130 				if(end.compareTo(group.getEnd()) != 0) {
131 					throw new IOException("Change of end range in " + 
132 							mapPath + " for range " + name);
133 				}
134 			} else {
135 				group = new RangeGroup(name,start,end);
136 			}
137 			group.setMembers(members);
138 			newGroupsMap.put(name,group);
139 		}
140 		Collection<RangeGroup> c = newGroupsMap.values();
141 		RangeGroup[] newGroups = new RangeGroup[c.size()];
142 		Iterator itrg = c.iterator();
143 		for (int i=0; itrg.hasNext(); i++)
144 		    newGroups[i] = (RangeGroup) itrg.next();
145 
146 //		RangeGroup[] newGroups = (RangeGroup[]) c.toArray();
147 		Arrays.sort(newGroups,comparator);
148 		groups = newGroups;
149 		LOGGER.info("Reloaded assignments from " + mapPath);
150 	}
151 	
152 	private void checkMapFile() throws IOException {
153 		long now = System.currentTimeMillis();
154 		if(nextCheck < now) {
155 			nextCheck = now + (checkInterval * MS_PER_SEC);
156 			File f = new File(mapPath);
157 			long curStat = f.lastModified();
158 			if(curStat > lastLoadStat) {
159 
160 				reloadMapFile();
161 				lastLoadStat = curStat;
162 			}
163 		}
164 	}
165 	
166 	protected RangeGroup getRangeGroupForRequest(WaybackRequest wbRequest)
167 		throws BadQueryException, ResourceIndexNotAvailableException {
168 		
169 		String keyUrl;
170 		try {
171 			checkMapFile();
172 		} catch (IOException e) {
173 			// TODO: this is too much error info if we're repeatedly failing..
174 			e.printStackTrace();
175 			throw new ResourceIndexNotAvailableException(e.getMessage());
176 		}
177 
178 		if(groups == null || groups.length == 0) {
179 			throw new ResourceIndexNotAvailableException("empty map file");			
180 		}
181 
182 	
183 		String searchUrl = wbRequest.getRequestUrl();
184 		if (searchUrl == null) {
185 			throw new BadQueryException("No " + WaybackRequest.REQUEST_URL 
186 					+ " specified");
187 		}
188 
189 		try {
190 			keyUrl = canonicalizer.urlStringToKey(searchUrl);
191 		} catch (URIException e) {
192 			throw new BadQueryException("invalid "
193 					+ WaybackRequest.REQUEST_URL + " " + searchUrl);
194 		}
195 		RangeGroup dummy = new RangeGroup("",keyUrl,"");
196 		int loc = Arrays.binarySearch(groups,dummy,comparator);
197 		if(loc < 0) {
198 			loc = (loc * -1) - 2;
199 		}
200 		LOGGER.info("Using group(" + groups[loc].getName() + ") for url (" +
201 				keyUrl + ")");
202 		return groups[loc];
203 	}
204 	
205 	/* (non-Javadoc)
206 	 * @see org.archive.wayback.ResourceIndex#query(org.archive.wayback.core.WaybackRequest)
207 	 */
208 	public SearchResults query(WaybackRequest wbRequest)
209 		throws ResourceIndexNotAvailableException,
210 		ResourceNotInArchiveException, BadQueryException,
211 		AccessControlException {
212 		
213 		RangeGroup group = getRangeGroupForRequest(wbRequest);
214 		return group.query(wbRequest);
215 	}

223 	}

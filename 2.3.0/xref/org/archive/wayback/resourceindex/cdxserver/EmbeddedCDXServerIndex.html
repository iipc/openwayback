132 	protected AuthToken createAuthToken(WaybackRequest wbRequest, String urlkey) {
133 		AuthToken waybackAuthToken = new APContextAuthToken(
134 				wbRequest.getAccessPoint());
135 		waybackAuthToken.setAllCdxFieldsAllow();

139 
140 		if (ignoreRobots) {
141 			waybackAuthToken.setIgnoreRobots(true);

144 		if (ignoreRobotPaths != null) {
145 			for (String path : ignoreRobotPaths) {
146 				if (urlkey.startsWith(path)) {
147 					waybackAuthToken.setIgnoreRobots(true);

156     public SearchResults doQuery(WaybackRequest wbRequest)
157             throws ResourceIndexNotAvailableException,
158             ResourceNotInArchiveException, BadQueryException,
159             AccessControlException {
160 
161     	//Compute url key (surt)
162 		String urlkey = null;
163 
164 		// If no canonicalizer is set, use selfRedirFilter's canonicalizer
165 		// Either selfRedirFilter or a canonicalizer must be set
166 
167 		UrlCanonicalizer canon = getCanonicalizer();
168 
169 		if (canon == null && selfRedirFilter != null) {
170 			canon = selfRedirFilter.getCanonicalizer();
171 		}
172 
173 		if (canon == null) {
174 			throw new IllegalArgumentException(
175 					"Unable to find canonicalizer, canonicalizer property or selfRedirFilter property must be set");
176 		}
177 
178 		try {
179 			urlkey = canon.urlStringToKey(wbRequest.getRequestUrl());
180 		} catch (URIException ue) {
181 			throw new BadQueryException(ue.toString());
182 		}
183 
184 		//Do local access/url validation check		
185         //AuthToken waybackAuthToken = new AuthToken(wbRequest.get(CDXServer.CDX_AUTH_TOKEN));
186 		AuthToken waybackAuthToken = createAuthToken(wbRequest, urlkey);
187 
188 		CDXToSearchResultWriter resultWriter = null;
189 		SearchResults searchResults = null;
190 
191 		if (wbRequest.isReplayRequest() || wbRequest.isCaptureQueryRequest()) {
192 			resultWriter = this.getCaptureSearchWriter(wbRequest,
193 					waybackAuthToken, false);
194 		} else if (wbRequest.isUrlQueryRequest()) {
195 			resultWriter = this.getUrlSearchWriter(wbRequest);
196 		} else {
197 			throw new BadQueryException("Unknown Query Type");
198 		}
199 
200 		try {
201 			loadWaybackCdx(urlkey, wbRequest, resultWriter.getQuery(),
202 					waybackAuthToken, resultWriter, false);
203 
204 			if (resultWriter.getErrorMsg() != null) {
205 				throw new BadQueryException(resultWriter.getErrorMsg());
206 			}
207 
208 			searchResults = resultWriter.getSearchResults();
209 
210 			if ((searchResults.getReturnedCount() == 0) &&
211 					(wbRequest.isReplayRequest() || wbRequest
212 							.isCaptureQueryRequest()) && tryFuzzyMatch) {
213 				resultWriter = this.getCaptureSearchWriter(wbRequest,
214 						waybackAuthToken, true);
215 
216 				if (resultWriter != null) {
217 					loadWaybackCdx(urlkey, wbRequest, resultWriter.getQuery(),
218 							waybackAuthToken, resultWriter, true);
219 
220 					searchResults = resultWriter.getSearchResults();
221 				}
222 			}
223 
224 			if (searchResults.getReturnedCount() == 0) {
225 				throw new ResourceNotInArchiveException(
226 						wbRequest.getRequestUrl() + " was not found");
227 			}
228 
229 		} catch (IOException e) {
230 			throw new ResourceIndexNotAvailableException(e.toString());
231 		} catch (RuntimeException rte) {
232 			Throwable cause = rte.getCause();
233 
234 			if (cause instanceof AccessControlException) {
235 				throw (AccessControlException)cause;
236 			}
237 
238 			if (cause instanceof IOException) {
239 				throw new ResourceIndexNotAvailableException(cause.toString());
240 			}
241 
242 			rte.printStackTrace(); // for now, for better debugging
243 			throw new ResourceIndexNotAvailableException(rte.toString());
244 		}
245 
246 		return searchResults;
247 	}

249 	protected void loadWaybackCdx(String urlkey, WaybackRequest wbRequest,
250 			CDXQuery query, AuthToken waybackAuthToken,
251 			CDXToSearchResultWriter resultWriter, boolean fuzzy)
252 			throws IOException, AccessControlException {
253 
254 		if ((remoteCdxPath != null) && !wbRequest.isUrlQueryRequest()) {
255 			try {
256 				// Not supported for remote requests, caching the entire cdx
257 				wbRequest.setTimestampSearchKey(false);
258 				this.remoteCdxServerQuery(urlkey, resultWriter.getQuery(),
259 						waybackAuthToken, resultWriter);
260 				return;
261 
262 			} catch (IOException io) {
263 				// Try again below
264 			} catch (RuntimeIOException rte) {
265 				Throwable cause = rte.getCause();
266 
267 				if (cause instanceof AccessControlException) {
268 					throw (AccessControlException)cause;
269 				} else {
270 					LOGGER.warning(rte.toString());
271 				}
272 			}
273 		}
274 
275 		cdxServer.getCdx(resultWriter.getQuery(), waybackAuthToken,
276 				resultWriter);
277 	}

279 	/**
280 	 * Create {@link CDXQuery} that is sent to {@link CDXServer}.
281 	 *
282 	 * The query specifies standard CDX server params described at:
283 	 * https://github.com/internetarchive/wayback/tree/master/wayback-cdx-server
284 	 *
285 	 * Note: this method adds extra filters meant for interactive (Wayback UI)
286 	 * use. CDXServer web API should not use this method.  this method is used
287 	 * for replay and capture-search requests only.
288 	 *
289 	 * TODO: move this to {@link CDXQuery} as static method.
290 	 *
291 	 * @param wbRequest {@link WaybackRequest} either replay or capture-query
292 	 * @param isFuzzy unused (?)
293 	 * @return A {@link CDXQuery} object for the supplied request.
294 	 */
295 	protected CDXQuery createQuery(WaybackRequest wbRequest, boolean isFuzzy) {
296 		CDXQuery query = new CDXQuery(wbRequest.getRequestUrl());
297 
298 		query.setLimit(limit);
299 		// query.setSort(CDXQuery.SortType.reverse);
300 
301 		String statusFilter = baseStatusFilter;
302 
303 		if (wbRequest.isReplayRequest()) {
304 			if (wbRequest.isBestLatestReplayRequest()) {
305 				statusFilter = "statuscode:[23]..";
306 			}
307 
308 			if (wbRequest.isTimestampSearchKey()) {
309 				query.setClosest(wbRequest.getReplayTimestamp());
310 			}
311 		} else if (wbRequest.isCaptureQueryRequest()) {
312 			// Add support for range calendar queries:
313 			// eg: /2005-2007*/
314 			// by mapping request start and end timestamp
315 			// to cdx server from= and to= params
316 			String start = wbRequest.getStartTimestamp();
317 			if (start != null) {
318 				query.setFrom(start);
319 			}
320 			String end = wbRequest.getEndTimestamp();
321 			if (end != null) {
322 				query.setTo(end);
323 			}
324 		}
325 
326 		// CDXServer#writeCdxResponse sets up timestamp collapsing filter
327 		// if collapseTime > 0
328 		int collapseTime = wbRequest.getCollapseTime();
329 		if (collapseTime < 0) {
330 			// unspecified - default to timestampDedupLength
331 			// (note: even zero is considered "specified")
332 			collapseTime = timestampDedupLength;
333 		}
334 		query.setCollapseTime(collapseTime);
335 
336 		// CDXServer#writeCdxResponse translates this into FieldRegexFilter
337 		if (statusFilter != null && !statusFilter.isEmpty())
338 			query.setFilter(new String[] { statusFilter });
339 
340 		return query;
341 	}

344 	protected void remoteCdxServerQuery(String urlkey, CDXQuery query,
345 			AuthToken authToken, CDXToSearchResultWriter resultWriter)
346 			throws IOException, AccessControlException {
347 		HTTPSeekableLineReader reader = null;
348 
349 		// This will throw AccessControlException if blocked
350 		// (in fact, it throws RuntimeIOException wrapping
351 		// AccessControlException)
352 		cdxServer.getAuthChecker().createAccessFilter(authToken)
353 				.includeUrl(urlkey, query.getUrl());
354 
355 		CloseableIterator<String> iter = null;
356 
357 		try {
358 			StringBuilder sb = new StringBuilder(remoteCdxPath);
359 
360 			sb.append("?url=");
361 			sb.append(URLEncoder.encode(query.getUrl(), "UTF-8"));
362 			//sb.append(query.getUrl());
363 //			sb.append("&limit=");
364 //			sb.append(query.getLimit());
365 			sb.append("&filter=");
366 			sb.append(URLEncoder.encode(query.getFilter()[0], "UTF-8"));
367 
368 //			if (!query.getClosest().isEmpty()) {
369 //				sb.append("&closest=");
370 //				sb.append(query.getClosest().substring(0, 4));
371 //			}
372 
373 			if (query.getCollapseTime() > 0) {
374 				sb.append("&collapseTime=");
375 				sb.append(query.getCollapseTime());
376 			}
377 
378 			sb.append("&gzip=true");
379 
380 			String finalUrl = sb.toString();
381 
382 			reader = this.remoteCdxHttp.get(finalUrl);
383 
384 			if (remoteAuthCookie != null) {
385 
386 				String cookie;
387 
388 				if (authToken.isIgnoreRobots() &&
389 						(remoteAuthCookieIgnoreRobots != null)) {
390 					cookie = remoteAuthCookieIgnoreRobots;
391 				} else {
392 					cookie = remoteAuthCookie;
393 				}
394 
395 				reader.setCookie(CDXServer.CDX_AUTH_TOKEN + "=" + cookie);
396 			}
397 
398 			reader.setSaveErrHeader(HttpCDXWriter.RUNTIME_ERROR_HEADER);
399 
400 			reader.seekWithMaxRead(0, true, -1);
401 
402 			iter = createRemoteIter(urlkey, reader);
403 
404 			resultWriter.begin();
405 
406 			while (iter.hasNext() && !resultWriter.isAborted()) {
407 				String rawLine = iter.next();
408 				CDXLine line = cdxLineFactory.createStandardCDXLine(rawLine,
409 						StandardCDXLineFactory.cdx11);
410 				resultWriter.writeLine(line);
411 			}
412 
413 			resultWriter.end();
414 			iter.close();
415 
416 		} finally {
417 			if (reader != null) {
418 				reader.close();
419 			}
420 		}
421 	}

423 	protected CloseableIterator<String> createRemoteIter(String urlkey,
424 			HTTPSeekableLineReader reader) throws IOException {
425 
426 		CloseableIterator<String> iter = new SeekableLineReaderIterator(reader);
427 
428 		String cacheInfo = reader.getHeaderValue("X-Page-Cache");
429 
430 		if ((cacheInfo != null) && cacheInfo.equals("HIT")) {
431 			if (LOGGER.isLoggable(Level.FINE)) {
432 				LOGGER.fine("CACHED");
433 			}
434 		}
435 
436 		if (extraSource != null) {
437 			ZipNumParams params = new ZipNumParams();
438 			CloseableIterator<String> extraIter = extraSource.getCDXIterator(
439 					urlkey, urlkey, urlkey, params);
440 
441 			if (extraIter.hasNext()) {
442 				SortedCompositeIterator<String> sortedIter = new SortedCompositeIterator<String>(
443 						MultiCDXInputSource.defaultComparator);
444 				sortedIter.addIterator(iter);
445 				sortedIter.addIterator(extraIter);
446 				return sortedIter;
447 			}
448 		}
449 
450 		return iter;
451 	}

455 	 * <p>possible future changes:
456 	 * <ul>

466 	protected CDXToCaptureSearchResultsWriter getCaptureSearchWriter(
467 			WaybackRequest wbRequest, AuthToken waybackAuthToken,
468 			boolean isFuzzy) {
469 		final CDXQuery query = createQuery(wbRequest, isFuzzy);
470 
471 		if (isFuzzy && query == null) {
472 			return null;
473 		}
474 
475 		boolean resolveRevisits = wbRequest.isReplayRequest();
476 
477 		// For now, not using seek single capture to allow for run time checking of additional records  
478 		//boolean seekSingleCapture = resolveRevisits && wbRequest.isTimestampSearchKey();
479 		boolean seekSingleCapture = false;
480 		//boolean seekSingleCapture = resolveRevisits && (wbRequest.isTimestampSearchKey() || (wbRequest.isBestLatestReplayRequest() && !wbRequest.hasMementoAcceptDatetime()));
481 
482 		CDXToCaptureSearchResultsWriter captureWriter = new CDXToCaptureSearchResultsWriter(
483 				query, resolveRevisits, seekSingleCapture, preferContains);
484 
485 		captureWriter.setTargetTimestamp(wbRequest.getReplayTimestamp());
486 
487 		captureWriter.setSelfRedirFilter(selfRedirFilter);
488 
489 		if ("true".equals(wbRequest.get(REQUEST_REVISIT_LOOKUP))) {
490 			captureWriter.setIncludeBlockedCaptures(true);

492

525 						(CaptureSearchResults)cResults, wbRequest, response);
526 				return true;
527 			}
528 
529 			CDXQuery query = new CDXQuery(wbRequest.getRequestUrl());
530 
531 			query.setOutput(wbRequest.getMementoTimemapFormat());
532 
533 			String from = wbRequest.get(MementoConstants.PAGE_STARTS);
534 
535 			if (from != null) {
536 				query.setFrom(from);
537 			}
538 
539 			try {
540 				query.fill(request);
541 			} catch (ServletRequestBindingException e1) {
542 				// Ignore
543 			}
544 
545 			cdxServer.getCdx(request, response, query);
546 
547 		} finally {
548 			PerfStats.timeEnd(PerfStat.IndexLoad);
549 		}
550 
551 		return true;
552 	}

581 	/**
582 	 * The number of digits of timestamp used for culling (<em>deduplicating</em>)
583 	 * captures in CDX query result.
584 	 * <p>For example, with this property set to 11, {#query} will
585 	 * return at most only one captures within each 10 minutes span.</p>
586 	 * <p>Non-positive value or 14 disables deduplication.</p>
587 	 * <p>Note: deduplication is done by {@link CDXServer}.
588 	 * {@code ZipNumIndex} also implements timestamp-deduplication, which
589 	 * can be turned on by setting positive value to {@code defaultParams.timestampDedupLength}.
590 	 * It is recommended to leave this off and use this parameter only,
591 	 * for several reasons:
592 	 * <ul>
593 	 * <li>ZipNumIndex's collapsing works at ZipNum block level. It works
594 	 * only when captures of single URL spans multiple ZipNum blocks, and
595 	 * can produce unexpected result if ever worked.</li>
596 	 * <li>ZipNumIndex's collapsing is closed to single index cluster.
597 	 * it can produce confusing result when multiple index clusters are
598 	 * combined.</li>
599 	 * <li>There's no point running timestamp-collapsing more than once
600 	 * (although the intent of ZipNumIndex's collapsing may be to quickly
601 	 * cut down on the number of results from URL with many many captures.)</li>
602 	 * </ul>
603 	 * <p>Note now it is possible to pass {@code collapseTime} parameter to
604 	 * {@code EmbeddedCDXServerIndex#query}, and this {@code timestampDedupLength}
605 	 * parameter serves as a default, used only when {@code collapseTime}
606 	 * is unspecified.
607 	 * See {@link WaybackRequest#setCollapseTime(int)}.</p>
608 	 * @param timestampDedupLength the number of digits of timestamp
609 	 * used for deduplication.
610 	 * @see ZipNumParams#setTimestampDedupLength(int)
611 	 * @see CDXServer#writeCdxResponse
612 	 * @see WaybackRequest#setCollapseTime(int)
613 	 */
614 	public void setTimestampDedupLength(int timestampDedupLength) {
615 		this.timestampDedupLength = timestampDedupLength;
616 	}

645             CaptureSearchResults results,
646             WaybackRequest wbRequest,
647             boolean includeOriginal) {
648 
649 		MementoUtils.addTimegateHeaders(response, results, wbRequest, includeOriginal);
650 
651 		// Add custom JSON header
652 		CaptureSearchResult result = results.getClosest();
653 
654 		JSONObject obj = new JSONObject();
655 
656 		JSONObject closestSnapshot = new JSONObject();
657 
658 		try {
659 			obj.put("wb_url", MementoUtils.getMementoPrefix(wbRequest.getAccessPoint()) + wbRequest.getAccessPoint().getUriConverter().makeReplayURI(result.getCaptureTimestamp(), wbRequest.getRequestUrl()));
660 			obj.put("timestamp", result.getCaptureTimestamp());
661 			obj.put("status", result.getHttpCode());
662 			closestSnapshot.put("closest", obj);
663 		} catch (JSONException je) {
664 
665 		}
666 		String json = closestSnapshot.toString();
667 		json = json.replace("\\/", "/");
668 		response.setHeader("X-Link-JSON", json);
669     }

744 	/**
745 	 * filter on {@code statuscode} field applied by default for <em>interactive</em>
746 	 * CDX lookup (i.e. from Wayback UI, not via CDX Server API).
747 	 * <p>Value is a regular expression for status code field. Only those CDXes
748 	 * with matching statuscode field will be returned. Leading/Trailing spaces are stripped off.
749 	 * If value starts with "{@code !}",
750 	 * only CDXes with <em>unmatching</em> statuscode field will be returned
751 	 * (<em>exception</em>: value "!" is treated as empty string, i.e. no filtering).</p>
752 	 * <p>Value will be ignored if WybackRequest.isBestLatestReplayRequest is set, for which
753 	 * hard-coded value "{@code [23]..}" is used.</p>
754 	 * <p>Default value is "{@code !(500|502|504)}".</p>
755 	 * <p><strong>NOTE</strong>: this is a quick hack to allow for customizing replay/listing of 5xx captures.
756 	 * it may be replaced by different customization method, or moved to other class in the
757 	 * future.</p>
758 	 * @param baseStatusRegexp regular expression for status code.
759 	 */
760 	public void setBaseStatusRegexp(String baseStatusRegexp) {
761 		this.baseStatusRegexp = baseStatusRegexp;
762 		this.baseStatusFilter = buildStatusFilter(baseStatusRegexp);
763 	}
764 
765 	protected static String buildStatusFilter(String regexp) {
766 		if (regexp == null)
767 			return "";
768 		String re = regexp.trim();
769 		if (re.isEmpty())
770 			return "";
771 		else {
772 			if (re.charAt(0) == '!') {
773 				re = re.substring(1).trim();
774 				if (re.isEmpty())
775 					return "";
776 				else
777 					return "!statuscode:" + re;
778 			} else {
779 				return "statuscode:" + re;
780 			}
781 		}
782 	}

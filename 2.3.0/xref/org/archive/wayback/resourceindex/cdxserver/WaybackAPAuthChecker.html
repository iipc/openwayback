3   import org.archive.cdxserver.auth.AuthToken;
4   import org.archive.cdxserver.filter.CDXAccessFilter;
5   import org.archive.cdxserver.filter.FilenamePrefixFilter;
6   import org.archive.wayback.accesspoint.CompositeAccessPoint;
7   import org.archive.wayback.resourceindex.filters.ExclusionFilter;
8   import org.archive.wayback.webapp.AccessPoint;
9   
10  /**
11   * {@link WaybackAuthChecker} modified to set up {@link FilenamePrefixFilter}
12   * from {@code fileIncludePrefixes} and {@code fileExcludePrefixes} of
13   * {@link AccessPoint}. This is meant to be used with {@link CompositeAccessPoint}
14   * to use different prefix filters for different collections.
15   *
16   * Needs {@link APContextAuthToken} to be passed to {@link #createAccessFilter(AuthToken)}.
17   * @see EmbeddedCDXServerIndex#createAuthToken(org.archive.wayback.core.WaybackRequest, String)
18   */
19  public class WaybackAPAuthChecker extends WaybackAuthChecker {
20  
21  	@Override
22  	public CDXAccessFilter createAccessFilter(AuthToken token) {
23  		APContextAuthToken apToken = null;
24  
25  		if (!(token instanceof APContextAuthToken)) {
26  			return super.createAccessFilter(apToken);
27  		}
28  
29  		apToken = (APContextAuthToken)token;
30  
31  		AccessPoint ap = apToken.ap;
32  
33  		FilenamePrefixFilter include = null, exclude = null;
34  
35  		if (ap.getFileIncludePrefixes() != null) {
36  			include = new FilenamePrefixFilter();
37  			include.setExclusion(false);
38  			include.setPrefixList(ap.getFileIncludePrefixes());
39  		}
40  
41  		if (ap.getFileExcludePrefixes() != null) {
42  			exclude = new FilenamePrefixFilter();
43  			exclude.setExclusion(true);
44  			exclude.setPrefixList(ap.getFileExcludePrefixes());
45  		}
46  
47  		ExclusionFilter adminFilter = null;
48  		if (adminExclusions != null) {
49  			adminFilter = adminExclusions.get();
50  		}
51  
52  		ExclusionFilter robotsFilter = null;
53  		if (robotsExclusions != null &&
54  				!isAllowed(token, ignoreRobotsAccessTokens) &&
55  				!token.isIgnoreRobots()) {
56  			robotsFilter = robotsExclusions.get();
57  		}
58  
59  		return new AccessCheckFilter(token, adminFilter, robotsFilter, include,
60  			exclude);
61  	}
62  }

25  public class HttpRequestMessage extends HttpMessage {
26  	private static byte[] METHOD_HEAD = {'H', 'E', 'A', 'D'};
27  	private static byte[] METHOD_GET = {'G', 'E', 'T'};
28  	
29  	private byte[] method = null;
30  	private byte[] path = null;
31  	private byte[] version = null;
32  	public HttpRequestMessage(byte[] method, byte[] path, byte[] version) {
33  		this.method = method;
34  		this.path = path;
35  		this.version = version;
36  	}
37  	public String getMethod() {
38  		return new String(method);
39  	}
40  	public String getPath() {
41  		return new String(path);
42  	}
43  	public String getVersion() {
44  		return new String(version);
45  	}
46  	public HttpRequestMessage _clone() {
47  		return new HttpRequestMessage(method,path,version);
48  	}
49  	public void setPath(byte[] path) {
50  		this.path = path;
51  	}
52  	public byte[] getBytes(boolean addCrLf) {
53  		byte[][] fields = {method,path,version};
54  		return concatBytes(fields,addCrLf);
55  	}

83  		return ByteOp.cmp(method,METHOD_HEAD);
84  	}
85  	public boolean isGet() {
86  		return ByteOp.cmp(method,METHOD_GET);

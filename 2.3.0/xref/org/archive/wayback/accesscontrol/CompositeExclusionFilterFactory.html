20  package org.archive.wayback.accesscontrol;
21  
22  import java.util.ArrayList;
23  import java.util.Iterator;
24  import java.util.logging.Logger;
25  
26  import org.archive.wayback.resourceindex.filters.CompositeExclusionFilter;
27  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
28  
29  /**
30   * Class that provides SearchResult Filtering based on multiple
31   * ExclusionFilterFactory instances by returning a single composite
32   * SearchResultFilter based on the results of each ExclusionFilter.
33   *
34   * @author brad
35   */
36  public class CompositeExclusionFilterFactory implements ContextExclusionFilterFactory {
37  
38  	private static final Logger LOGGER = Logger
39  		.getLogger(CompositeExclusionFilterFactory.class.getName());
40  
41  	private ArrayList<ExclusionFilterFactory> factories = new ArrayList<ExclusionFilterFactory>();
42  
43  	/**
44  	 * @param factory to be added to the composite
45  	 */
46  	public void addFactory(ExclusionFilterFactory factory) {
47  		factories.add(factory);
48  	}
49  
50  	protected static ExclusionFilter getInstance(ExclusionFilterFactory factory, CollectionContext context) {
51  		if (factory instanceof ContextExclusionFilterFactory) {
52  			return ((ContextExclusionFilterFactory)factory).getExclusionFilter(context);
53  		} else {
54  			return factory.get();
55  		}
56  	}
57  
58  	@Override
59  	public ExclusionFilter getExclusionFilter(CollectionContext context) {
60  		CompositeExclusionFilter filter = new CompositeExclusionFilter();
61  		for (ExclusionFilterFactory factory : factories) {
62  			ExclusionFilter memberFilter = getInstance(factory, context);
63  			if (memberFilter != null) {
64  				filter.addComponent(memberFilter);
65  			} else {
66  				LOGGER.warning("Skipping null filter returned from factory: " + factory);
67  			}
68  		}
69  		return filter;
70  	}
71  
72  	/*
73  	 * (non-Javadoc)
74  	 *
75  	 * @see org.archive.wayback.resourceindex.ExclusionFilterFactory#get()
76  	 */
77  	public ExclusionFilter get() {
78  		return getExclusionFilter(null);
79  	}
80  
81  	/**
82  	 * @return the factories
83  	 */
84  	public ArrayList<ExclusionFilterFactory> getFactories() {
85  		return factories;
86  	}
87  
88  	/**
89  	 * @param factories the factories to set
90  	 */
91  	public void setFactories(ArrayList<ExclusionFilterFactory> factories) {
92  		this.factories = factories;
93  	}
94  
95  	/*
96  	 * (non-Javadoc)
97  	 *
98  	 * @see org.archive.wayback.accesscontrol.ExclusionFilterFactory#shutdown()
99  	 */
100 	public void shutdown() {
101 		Iterator<ExclusionFilterFactory> itr = factories.iterator();
102 		while (itr.hasNext()) {
103 			ExclusionFilterFactory i = itr.next();
104 			i.shutdown();
105 		}
106 	}
107 }

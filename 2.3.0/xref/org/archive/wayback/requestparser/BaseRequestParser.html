20  package org.archive.wayback.requestparser;
21  
22  import java.util.Map;
23  
24  import javax.servlet.http.HttpServletRequest;
25  
26  import org.archive.wayback.RequestParser;
27  import org.archive.wayback.core.WaybackRequest;
28  import org.archive.wayback.exception.BadQueryException;
29  import org.archive.wayback.exception.BetterRequestException;
30  import org.archive.wayback.util.Timestamp;
31  import org.archive.wayback.webapp.AccessPoint;
32  
33  /**
34   * Abstract implementation of the RequestParser interface, which provides some
35   * convenience methods for accessing data in Map<String,String>'s, and also
36   * allows for configuring maxRecords, and earliest and latest timestamp strings.
37   * 
38   * Subclasses must still implement parse().
39   *
40   * @author brad
41   * @version $Date$, $Revision$
42   */
43  public abstract class BaseRequestParser implements RequestParser {
44  
45  	/**
46  	 * String path matching a query request, by form or OpenSearch
47  	 */
48  	public final static String QUERY_BASE = "query";
49  
50  	/**
51  	 * String path matching a query request, by form or OpenSearch, indicating
52  	 * user requests XML data in response
53  	 */
54  	public final static String XQUERY_BASE = "xmlquery";
55  
56  	/**
57  	 * String path matching a replay request, by form or OpenSearch
58  	 */
59  	public final static String REPLAY_BASE = "replay";
60  	
61  	/**
62  	 * Default maximum number of records to assume, overridden by configuration,
63  	 * when not specified in the client request
64  	 */
65  	public final static int DEFAULT_MAX_RECORDS = 10;
66  
67  	
68  	private int maxRecords = DEFAULT_MAX_RECORDS;
69  	private String earliestTimestamp = null;
70  	private String latestTimestamp = null;
71  
72  	public abstract WaybackRequest parse(HttpServletRequest httpRequest, 
73  			AccessPoint wbContext) throws BadQueryException, 
74  			BetterRequestException;
75  
76  	/**
77  	 * @return the maxRecords to use with this RequestParser, when not specified
78  	 * by the client request
79  	 */
80  	public int getMaxRecords() {
81  		return maxRecords;
82  	}
83  
84  	/**
85  	 * @param maxRecords the maxRecords to use with this RequestParser, when not
86  	 * specified by the client request
87  	 */
88  	public void setMaxRecords(int maxRecords) {
89  		this.maxRecords = maxRecords;
90  	}
91  
92  	/**
93  	 * @param timestamp the earliest timestamp to use with this RequestParser 
94  	 * when none is supplied by the user request
95  	 */
96  	public void setEarliestTimestamp(String timestamp) {
97  		earliestTimestamp = Timestamp.parseBefore(timestamp).getDateStr();
98  	}
99  
100 	/**
101 	 * @return the default earliest timestamp to use with this RequestParser
102 	 */
103 	public String getEarliestTimestamp() {
104 		return earliestTimestamp;
105 	}
106 
107 	/**
108 	 * @return default latest timestamp for this RequestParser
109 	 */
110 	public String getLatestTimestamp() {
111 		return latestTimestamp;
112 	}
113 
114 	/**
115 	 * @param timestamp the default latest timestamp to use with this 
116 	 * RequestParser
117 	 */
118 	public void setLatestTimestamp(String timestamp) {
119 		this.latestTimestamp = Timestamp.parseAfter(timestamp).getDateStr();
120 	}
121 }

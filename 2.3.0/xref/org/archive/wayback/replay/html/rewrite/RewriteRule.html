3   import org.archive.util.iterator.StringTransformer;
4   import org.archive.wayback.replay.html.ReplayParseContext;
5   import org.springframework.beans.factory.BeanNameAware;

7   /**
8    * {@code RewriteRule} is similar to {@link StringTransformer}, but it receives
9    * policy name (rewrite rule name), from which it can extract parameters.
10   * <p>
11   * StringTransformer can be seen as super-interface of this class (although
12   * {@code RewriteRule} is not an interface), used where transformation is
13   * static. As {@link RewriteRule} does not provide consistent parameter parsing
14   * service, it'd be good idea to define {@code ParameterizedStringTransformer}
15   * with better service to replace {@code RewriteRule}.
16   * </p>
17   */
18  public abstract class RewriteRule implements BeanNameAware {
19  
20  	String name;
21  
22  	@Override
23  	public void setBeanName(String beanName) {
24  		// This method will be called after properties are set.
25  		// Name set explicitly shall take precedence over bean name.
26  		if (name == null)
27  			name = beanName;
28  	}
29  
30  	public String getName() {
31  		return name;
32  	}
33  
34  	/**
35  	 * Set name explicitly. Takes precedence over bean name.
36  	 * @param name rule name
37  	 */
38  	public void setName(String name) {
39  		this.name = name;
40  	}
41  
42  	public abstract String rewrite(ReplayParseContext context, String policy,
43  			String input);
44  }

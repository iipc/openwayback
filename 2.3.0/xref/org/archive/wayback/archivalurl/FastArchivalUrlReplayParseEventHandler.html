46  /**
47   * Lean and mean ParseEventHandler implementing current best-known server-side
48   * HTML rewrite rules, and should be much faster than the fully configurable
49   * version.
50   * 
51   * <p>This class has kludgy support for disabling JavaScript inclusion
52   * with <code>&lt;SCRIPT SRC=...&gt;</code> element.  <code>jsBlockTrans</code>,
53   * whose primary use is translating JavaScript code block (inline &lt;SCRIPT&gt;,
54   * <code>javascript:</code> URI in <code>HREF</code> attribute, and event
55   * handler attributes), is also called with a value of <code>SRC</code> attribute
56   * of &lt;SCRIPT&gt; tag.  If <code>jsBlockTrans</code> returns either <code>null</code>
57   * or an empty String, &lt;SCRIPT&gt; element is disabled by changing <code>SRC</code> to
58   * an empty value.  Note that in this case <code>jsBlockTrans</code> is used only for
59   * a test, and its return value is simply discarded.
60   * URL translation is done by a subsequent call to {@link URLStringTransformer} for
61   * <code>js_</code> context.  This feature is very likely a subject of refactoring
62   * in the future.</p>
63   *
64   * @author brad
65   * 
66   */
67  public class FastArchivalUrlReplayParseEventHandler implements
68  		ParseEventHandler {

124         /**
125          *
126          * @param pContext
127          * @param node
128          * @throws IOException
129          */
130         @Override
131 	public void handleNode(ParseContext pContext, Node node) throws IOException {
132 		ReplayParseContext context = (ReplayParseContext)pContext;
133 		if (NodeUtils.isRemarkNode(node)) {
134 			// HTML Comment - often used in JAVASCRIPT/STYLE element
135 			// to hide its content on old browsers.
136 			RemarkNode remarkNode = (RemarkNode)node;
137 			if (context.isInCSS()) {
138 				remarkNode.setText(cssBlockTrans.transform(context,
139 					remarkNode.getText()));
140 			} else {
141 				remarkNode.setText(jsBlockTrans.transform(context,
142 					remarkNode.getText()));
143 			}
144 			emit(context, null, node, null);
145 
146 		} else if (NodeUtils.isTextNode(node)) {
147 			TextNode textNode = (TextNode)node;
148 			if (context.isInCSS()) {
149 				handleCSSTextNode(context, textNode);
150 			} else if (context.isInScriptText()) {
151 				handleJSTextNode(context, textNode);
152 			}
153 			emit(context, null, textNode, null);
154 //				handleContentTextNode(context,textNode);
155 		} else if (NodeUtils.isTagNode(node)) {
156 			TagNode tagNode = (TagNode)node;
157 
158 			if (tagNode.isEndTag()) {
159 				if (tagNode.getTagName().equals("HEAD")) {
160 					context.putData(FERRET_IN_HEAD, null);
161 				}
162 
163 				if (checkAllowTag(pContext, tagNode)) {
164 					emit(context, null, tagNode, null);
165 				}
166 
167 //				handleCloseTagNode(context,tagNode);
168 			} else if (tagNode.getTagName().startsWith("![CDATA[")) {
169 				// CDATA section is delivered as TagNode, and it
170 				// appears there's no ordinary way of replacing its
171 				// body content. Also CSS/JS handling method wants
172 				// TextNode. Create a temporary TextNode for them,
173 				// and write "<![CDATA["..."]]>" around it.
174 				String text = tagNode.getText();
175 				int s = "![CDATA[".length();
176 				// text is supposed to end with "]]", but just in case.
177 				int e = text.endsWith("]]") ? text.length() - 2 : text.length();
178 				if (context.isInCSS()) {
179 					TextNode textNode = new TextNode(text.substring(s, e));
180 					handleCSSTextNode(context, textNode);
181 					emit(context, "<![CDATA[", textNode, "]]>");
182 				} else if (context.isInScriptText()) {
183 					TextNode textNode = new TextNode(text.substring(s, e));
184 					handleJSTextNode(context, textNode);
185 					emit(context, "<![CDATA[", textNode, "]]>");
186 				} else {
187 					emit(context, null, tagNode, null);
188 				}
189 			} else {
190 				context.setInHTML(true);
191 				// assume start, possibly empty:
192 				handleOpenTagNode(context, tagNode);
193 			}
194 		} else {
195 			throw new IllegalArgumentException("Unknown node type..");
196 		}
197 	}

214 	private void handleJSTextNode(ReplayParseContext context, TextNode textNode) throws IOException {
215 		boolean alreadyInsertedHead = (context.getData(FERRET_HEAD_INSERTED) != null);
216 		context.incJSBlockCount();
217 		if (alreadyInsertedHead) {

223 	/**
224 	 * kludgy support for selectively disabling JavaScript that messes up
225 	 * replay.
226 	 * <p>
227 	 * If {@code jsBlockTrans.transform} returns {@code null} or empty for
228 	 * {@code SCRIPT/@SRC}, {@code SCRIPT} element is disabled by replacing
229 	 * {@code SRC} attribute with empty value. {@link DisableJSIncludeRewriteRule}
230 	 * provides one implementation to make use of this behavior.
231 	 * </p>
232 	 * @param context {@link ReplayParseContext}
233 	 * @param tagNode {@code SCRIPT} tag.
234 	 * @see DisableJSIncludeRewriteRule
235 	 */
236 	private void handleJSIncludeNode(ReplayParseContext context, TagNode tagNode) {
237 		String file = tagNode.getAttribute("SRC");
238 		if (file != null) {
239 			String result = jsBlockTrans.transform(context, file);
240 			// URL rewrite is done by AttributeRewriter, which should ignore
241 			// empty value.
242 			if (result == null || result.isEmpty()) {
243 				tagNode.setAttribute("SRC", "");
244 			}
245 		}
246 	}

248 	private void handleOpenTagNode(ReplayParseContext context, TagNode tagNode) 
249 	throws IOException {
250 		
251 		String preEmit = null;
252 		String postEmit = null;
253 
254 		String tagName = tagNode.getTagName();
255 		
256 		boolean alreadyInsertedHead = (context.getData(FERRET_HEAD_INSERTED) != null);
257 		boolean insertedJsp = context.getData(FERRET_DONE_KEY) != null;
258 		boolean inHead = (context.getData(FERRET_IN_HEAD) != null);
259 
260 		if (!alreadyInsertedHead) {
261 			// If we're at the beginning of a <head> tag, and haven't inserted yet, 
262 			// insert right AFTER head tag
263 			if (tagName.equals("HEAD")) {
264 				emitHeadInsert(context, tagNode, true);
265 				context.putData(FERRET_IN_HEAD, FERRET_IN_HEAD);
266 				// this means HEAD tag does not get its attribute
267 				// rewritten. probably that's ok...
268 				return;
269 			}
270 			// If we're at the beginning of any tag, other than <html>,
271 			// <!DOCTYPE ...>, <?xml ... ?> (tagName is "?" for it),
272 			// (including <body>) and haven't inserted yet,
273 			// insert right BEFORE the next tag, also continue other default processing
274 			// of the tag
275 			if (!tagName.equals("HTML") && !tagName.equals("!DOCTYPE") &&
276 					!tagName.equals("?")) {
277 				emitHeadInsert(context, null, false);
278 				// Don't return continue to further processing
279 			}
280 		} else if (tagName.equals(BODY_TAG) && inHead) {
281 			context.putData(FERRET_IN_HEAD, null);
282 			inHead = false;
283 
284 			OutputStream out = context.getOutputStream();
285 			out.write("</head>".getBytes(context.getOutputCharset()));
286 		}
287 				
288 		// Time to insert the JSP header?
289 		//IK added check to avoid inserting inside css or script
290 		if (!insertedJsp && !context.isInCSS() && !context.isInScriptText()) {
291 			if (tagName.equals(FRAMESET_TAG)) {
292 				// don't put the insert in FRAMESET
293 				context.putData(FERRET_DONE_KEY, "");
294 			} else if (tagName.equals(BODY_TAG)) {
295 				postEmit = bodyInsertContent(context);
296 				context.putData(FERRET_DONE_KEY, "");
297 			} else if (!okHeadTagMap.containsKey(tagName)) {
298 				// hrm... we are seeing a node that should be in
299 				// the body.. lets emit the jsp now, *before*
300 				// the current Tag:
301 				preEmit = bodyInsertContent(context);
302 				context.putData(FERRET_DONE_KEY, "");
303 			}
304 		}
305 		
306 		if (tagName.equals("BASE")) {
307 			String baseURL = tagNode.getAttribute("HREF");
308 			if (baseURL != null) {
309 				context.setBaseUrl(baseURL);
310 			}
311 		} else if (tagName.equals("SCRIPT")) {
312 			// hacky disable-SCRIPT feature.
313 			handleJSIncludeNode(context, tagNode);
314 		}
315 
316 		// now do all the usual attribute rewriting
317 		attributeRewriter.rewrite(context, tagNode);
318 
319 		// drop tags named by rewrite policy as such.
320 		if (!checkAllowTag(context, tagNode)) return;
321 
322 		emit(context, preEmit, tagNode, postEmit);
323 
324 	}

326 	protected boolean checkAllowTag(ParseContext context, TagNode tagNode) {
327 		String tagName = tagNode.getTagName();
328 		
329 		// Check the NOSCRIPT tag, if force-noscript is set,
330 		// then  skip the NOSCRIPT tags and include contents explicitly
331 		if (tagName.equals("NOSCRIPT")) {
332 			String allPolicies = context.getOraclePolicy();
333 			
334 			if ((allPolicies != null) && allPolicies.contains("force-noscript")) {
335 				return false;
336 			}
337 		}
338 		
339 		return true;
340 	}

342 	protected void emit(ReplayParseContext context, String pre, Node node, 
343 			String post) throws IOException {
344 		
345 		OutputStream out = context.getOutputStream();
346 		if (out != null) {
347 //			Charset charset = Charset.forName(context.getOutputCharset());
348 			String charset = context.getOutputCharset();
349 
350 			if (pre != null) {
351 				out.write(pre.getBytes(charset));
352 			}
353 			
354 			if (node != null) {
355 				out.write(node.toHtml(true).getBytes(charset));
356 			}
357 
358 			if (post != null) {
359 				out.write(post.getBytes(charset));
360 			}
361 		}

526 	protected void emitHeadInsert(ReplayParseContext context, Node node,
527 			boolean postInsert) throws IOException {
528 		String headInsert = null;
529 
530 		if (headInsertJsp == null) {
531 			this.emit(context, null, node, null);
532 			return;
533 		}
534 
535 		try {
536 			headInsert = context.getJspExec().jspToString(headInsertJsp);
537 			context.putData(FERRET_HEAD_INSERTED, FERRET_HEAD_INSERTED);
538 		} catch (ServletException e) {
539 			e.printStackTrace();
540 		}
541 
542 		if (postInsert) {
543 			this.emit(context, null, node, headInsert);
544 		} else {
545 			this.emit(context, headInsert, node, null);
546 		}
547 	}

549 	/**
550 	 * return body-insert text.
551 	 * <p>Run {@code jspInsertPath} and return its output as String.
552 	 * if {@code jspInsertPath} is {@code null}, or body-insert should not be
553 	 * inserted into the resource being processed, returns {@code null}.</p>
554 	 * @param context context for the resource being processed
555 	 * @return insert text as String, or {@code null} if no insertion shall be
556 	 *         made.
557 	 */
558 	protected String bodyInsertContent(ReplayParseContext context) {
559 		if (jspInsertPath == null)
560 			return null;
561 		JSPExecutor jspExec = context.getJspExec();
562 		// FIXME bad chain of references. add method to ReplayParseContext?
563 		WaybackRequest wbRequest = jspExec.getUiResults().getWbRequest();
564 		// isAnyEmbeddedContext() used as shorthand for (isFrameWrapperContext()
565 		// && isIFrameWrapperContext()).
566 		if (wbRequest.isAnyEmbeddedContext())
567 			return null;
568 		try {
569 			return jspExec.jspToString(jspInsertPath);
570 		} catch (ServletException ex) {
571 			LOGGER.log(Level.WARNING, "execution of " + jspInsertPath +
572 					" failed", ex);
573 			return null;
574 		} catch (IOException ex) {
575 			LOGGER.log(Level.WARNING, "erorr executing " + jspInsertPath, ex);
576 			return null;
577 		}
578 	}

590 	/**
591 	 * set this property false if you want to disable unescaping
592 	 * (and corresponding re-escaping) of attribute values.
593 	 * <p>By default, HTML entities (such as <code>&amp;amp;</code>)
594 	 * in attribute values are unescaped before translation attempt,
595 	 * and then escaped back before writing out.  Although this is
596 	 * supposedly the right thing to do, it has a side-effect: all
597 	 * bare "<code>&amp;</code>" (not escaped as "<code>&amp;amp;</code>")
598 	 * will be replaced by "<code>&amp;amp;</code>".  Setting this property
599 	 * to <code>false</code> disables it.</p>
600 	 * <p>As URL rewrite does neither parse nor modify query part, it
601 	 * should mostly work without unescaping.  But there may be some
602 	 * corner cases where escaping is crucial.  Don't set this to {@code false}
603 	 * unless it's absolutely necessary.</p>
604 	 * @param unescapeAttributeValues <code>false</code> to disable unescaping
605 	 * @deprecated 1.8.1/05-23-2014 property moved to {@link StandardAttributeRewriter}
606 	 *   This property still works, but only with {@code StandardAttributeRewriter}.
607 	 */
608 	public void setUnescapeAttributeValues(boolean unescapeAttributeValues) {
609 		this.unescapeAttributeValues = unescapeAttributeValues;
610 		if (attributeRewriter instanceof StandardAttributeRewriter) {
611 			((StandardAttributeRewriter)attributeRewriter).setUnescapeAttributeValues(unescapeAttributeValues);
612 		}

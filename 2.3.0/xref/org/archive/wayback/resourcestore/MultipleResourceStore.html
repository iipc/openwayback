20  package org.archive.wayback.resourcestore;
21  
22  import java.io.IOException;
23  import java.util.List;
24  import java.util.logging.Logger;
25  
26  import javax.servlet.http.HttpServletResponse;
27  
28  import org.archive.wayback.ResourceStore;
29  import org.archive.wayback.core.CaptureSearchResult;
30  import org.archive.wayback.core.Resource;
31  import org.archive.wayback.exception.ResourceNotAvailableException;
32  import org.archive.wayback.resourcestore.resourcefile.ArcWarcFilenameFilter;
33  
34  
35  /**
36   * A ResourceStore which contains one or more other resource stores.
37   * This container simply searches each one in turn for a requested
38   * resource, returning the first one found.
39   */
40  public class MultipleResourceStore implements ResourceStore {
41  
42  	private final static Logger LOGGER = Logger.getLogger(MultipleResourceStore.class.getName());
43  
44  	private List<ResourceStore> stores;
45  	
46  	// If true, subsequent resource stores are skipped if the first
47  	private boolean failOnFirstUnavailable = false;
48  
49  	public boolean isFailOnFirstUnavailable() {
50  		return failOnFirstUnavailable;
51  	}
52  
53  	public void setFailOnFirstUnavailable(boolean failOnFirstUnavailable) {
54  		this.failOnFirstUnavailable = failOnFirstUnavailable;
55  	}
56  
57  	public Resource retrieveResource(CaptureSearchResult result)
58  		throws ResourceNotAvailableException {
59      
60  		String fileName = result.getFile();
61  		if(fileName == null || fileName.length() < 1) {
62  			throw new ResourceNotAvailableException("No ARC/WARC name in search result...", fileName);
63  		}
64  
65  		if(!fileName.endsWith(ArcWarcFilenameFilter.ARC_SUFFIX)
66  				&& !fileName.endsWith(ArcWarcFilenameFilter.ARC_GZ_SUFFIX)
67  				&& !fileName.endsWith(ArcWarcFilenameFilter.WARC_SUFFIX)
68  				&& !fileName.endsWith(ArcWarcFilenameFilter.WARC_GZ_SUFFIX)) {
69  			fileName = fileName + ArcWarcFilenameFilter.ARC_GZ_SUFFIX;
70  		}
71  		
72  		String errMsg = "";
73  		Exception origException = null;
74  
75  		for ( ResourceStore store : stores ) {
76  			try {
77  				Resource r = store.retrieveResource(result);
78  
79  				if ( r != null ) {
80  					return r;
81  				}
82  				
83  			} catch (ResourceNotAvailableException e) {				
84  				LOGGER.info(e.toString());
85  				if (!errMsg.isEmpty()) {
86  					errMsg += " ";
87  				}
88  				origException = e;
89  				errMsg += e.getMessage();
90  				
91  				// if skipOnUnavailable, then don't try any more resource stores if the current one is unavailable
92  				if (failOnFirstUnavailable && (e.getStatus() == HttpServletResponse.SC_SERVICE_UNAVAILABLE)) {
93  					break;
94  				}
95  			}
96  		}
97  
98  		throw new ResourceNotAvailableException(errMsg.isEmpty() ? "Unable to retrieve: "+fileName : errMsg, fileName, origException);
99  	}
100 
101 	public void setStores(List<ResourceStore> stores) {
102 		this.stores = stores;
103 	}
104   
105 	public List<ResourceStore> getStores( ) {
106 		return this.stores;
107 	}
108 
109 	public void shutdown() throws IOException {
110 		// no-op
111 	}
112 }

20  package org.archive.wayback.domainprefix;
21  
22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  
25  import javax.servlet.http.HttpServletRequest;
26  
27  import org.archive.wayback.core.WaybackRequest;
28  import org.archive.wayback.exception.BadQueryException;
29  import org.archive.wayback.requestparser.BaseRequestParser;
30  import org.archive.wayback.requestparser.WrappedRequestParser;
31  import org.archive.wayback.util.Timestamp;
32  import org.archive.wayback.webapp.AccessPoint;
33  
34  /**
35   *
36   *
37   * @author brad
38   * @version $Date$, $Revision$
39   */
40  public class DomainPrefixRequestParser extends WrappedRequestParser {
41  
42  	/**
43  	 * @param wrapped
44  	 */
45  	public DomainPrefixRequestParser(BaseRequestParser wrapped) {
46  		super(wrapped);
47  	}
48  
49  	String hostPort = "localhost:8081";
50  
51  	private final Pattern REPLAY_REGEX = 
52  		Pattern.compile("^(\\d{1,14})\\.(.*)$");
53  	private final Pattern QUERY_REGEX = 
54  		Pattern.compile("^(\\d{0,13})\\*\\.(.*)$");
55  	
56  	private String getRequestString(final String host, 
57  			HttpServletRequest httpRequest) {
58  		String path = httpRequest.getRequestURI();
59  		String query = httpRequest.getQueryString();
60  
61  		String r = "";
62  		if(path == null) {
63  			path = "/";
64  		}
65  		if(query != null && query.length() > 0) {
66  			r = "http://" + host + path + "?" + query;
67  		} else {
68  			r = "http://" + host + path;						
69  		}
70  		return r;
71  	}
72  	
73  	/* (non-Javadoc)
74  	 * @see org.archive.wayback.RequestParser#parse(javax.servlet.http.HttpServletRequest, org.archive.wayback.webapp.WaybackContext)
75  	 */
76  	public WaybackRequest parse(HttpServletRequest httpRequest,
77  			AccessPoint wbContext) throws BadQueryException {
78  		
79  		WaybackRequest wbRequest = null;
80  		String server = httpRequest.getServerName() + 
81  			":" + httpRequest.getServerPort();
82  		if(server.toLowerCase().endsWith(hostPort.toLowerCase())) {
83  			int length = server.length() - hostPort.length();
84  			if(server.length() > hostPort.length()) {
85  				String prefix = server.substring(0,length - 1);
86  				Matcher replayMatcher = REPLAY_REGEX.matcher(prefix);
87  				if (replayMatcher != null && replayMatcher.matches()) {
88  					wbRequest = new WaybackRequest();
89  					String dateStr = replayMatcher.group(1);
90  					String host = replayMatcher.group(2);
91  
92  					String requestUrl = getRequestString(host,httpRequest);
93  
94  					wbRequest.setReplayRequest();
95  					wbRequest.setReplayTimestamp(dateStr);
96  					wbRequest.setRequestUrl(requestUrl);
97  
98  				} else {
99  					Matcher queryMatcher = QUERY_REGEX.matcher(prefix);
100 					if(queryMatcher != null && queryMatcher.matches()) {
101 						wbRequest = new WaybackRequest();
102 						String dateStr = queryMatcher.group(1);
103 						String host = queryMatcher.group(2);
104 
105 						String requestUrl = getRequestString(host,httpRequest);
106 
107 						String startDate;
108 						String endDate;
109 						if(dateStr.length() == 0) {
110 							startDate = getEarliestTimestamp();
111 							endDate = getLatestTimestamp();
112 						} else {
113 							startDate = Timestamp.parseBefore(dateStr).getDateStr();
114 							endDate = Timestamp.parseAfter(dateStr).getDateStr();
115 						}
116 						wbRequest.setCaptureQueryRequest();
117 						wbRequest.setStartTimestamp(startDate);
118 						wbRequest.setEndTimestamp(endDate);
119 						wbRequest.setRequestUrl(requestUrl);
120 					}
121 					// TODO: what if it doesn't match the QUERY_REGEX?
122 					//       throw a BadQueryException?
123 				}
124 			}
125 		}
126 		return wbRequest;
127 	}
128 
129 	/**
130 	 * @return the hostPort
131 	 */
132 	public String getHostPort() {
133 		return hostPort;
134 	}
135 
136 	/**
137 	 * @param hostPort the hostPort to set
138 	 */
139 	public void setHostPort(String hostPort) {
140 		this.hostPort = hostPort;
141 	}
142 }

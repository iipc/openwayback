1   package org.archive.cdxserver.processor;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import org.apache.commons.lang.math.NumberUtils;
7   import org.archive.format.cdx.CDXLine;
8   
9   /**
10   * A variant of {@link DupeTimestampBestStatusFilter} that returns
11   * the last best capture instead of the first one.
12   * Support of {@code noCollapsePrefix} complicates processing, so this
13   * may be slightly slower than {@link DupeTimestampBestStatusFilter}.
14   * <p>Note that the semantics of {@link #writeLine(CDXLine)} is slightly
15   * different from other processors. It returns 1 if some (non-pass-through)
16   * CDX line, which is not necessarily the same as the CDX line passed as argument,
17   * is written out. Count of ones would be one less than others.</p>
18   * <p>Tests in {@link DupeTimestampBestStatusFilterTest}.</p>
19   */
20  public class DupeTimestampLastBestStatusFilter extends DupeTimestampBestStatusFilter {
21  
22  	/**
23  	 * Keeps the best CDX line so far within a group.
24  	 */
25  	protected CDXLine bestLine;
26  	/**
27  	 * Keeps a list of CDXLines that matches {@code noCollapsePreifx}, but
28  	 * cannot be written yet because their {@code timestamp}s are larger than
29  	 * {@code bestLine.timestamp}.
30  	 */
31  	protected List<CDXLine> pendingPassThroughs;
32  
33  	public DupeTimestampLastBestStatusFilter(BaseProcessor output,
34  			int timestampDedupLength, String[] noCollapsePrefix) {
35  		super(output, timestampDedupLength, noCollapsePrefix);
36  		this.pendingPassThroughs = new ArrayList<CDXLine>();
37  	}
38  
39  	/**
40  	 * Write out all pending pass-throughs, and
41  	 * clear pass-through buffer.
42  	 */
43  	protected final void flushPassThrough() {
44  		for (CDXLine line : pendingPassThroughs) {
45  			// NB: don't call super.writeLine()
46  			inner.writeLine(line);
47  		}
48  		pendingPassThroughs.clear();
49  	}
50  
51  	/**
52  	 * return group key of {@code line}.
53  	 * @param line CDX line
54  	 * @return first {@code timestampDedupLength} digits
55  	 * of {@code timestamp}
56  	 */
57  	protected final String groupKey(CDXLine line) {
58  		String timestamp = line.getTimestamp();
59  		return timestamp.substring(0,
60  			Math.min(timestampDedupLength, timestamp.length()));
61  	}
62  
63  	@Override
64  	public int writeLine(CDXLine line) {
65  		if (timestampDedupLength <= 0) {
66  			// NB: do not call super.writeLine()
67  			// same for all writeLine() calls below.
68  			return inner.writeLine(line);
69  		}
70  		if (passThrough(line)) {
71  			if (bestLine != null) {
72  				pendingPassThroughs.add(line);
73  				return 1;
74  			} else {
75  				return inner.writeLine(line);
76  			}
77  		}
78  		String key = groupKey(line);
79  		int httpCode = NumberUtils.toInt(line.getStatusCode(), WORST_HTTP_CODE);
80  		if (lastTimestamp != null && key.equals(lastTimestamp)) {
81  			// within a collapse group
82  			if (httpCode <= bestHttpCode) {
83  				flushPassThrough();
84  				bestLine = line;
85  				bestHttpCode = httpCode;
86  			}
87  			return 0;
88  		} else {
89  			// new collapse group
90  			int r = 0;
91  			if (bestLine != null) {
92  				// for the first line
93  				r = inner.writeLine(bestLine);
94  				flushPassThrough();
95  			}
96  			bestLine = line;
97  			bestHttpCode = httpCode;
98  			lastTimestamp = key;
99  			return r;
100 		}
101 	}
102 
103 	@Override
104 	public void end() {
105 		// last collapse group. bestLine == null happens
106 		// only when writeLine() was never called, or
107 		// timestampDedupLength <= 0
108 		if (bestLine != null) {
109 			inner.writeLine(bestLine);
110 			flushPassThrough();
111 		}
112 		super.end();
113 	}
114 }

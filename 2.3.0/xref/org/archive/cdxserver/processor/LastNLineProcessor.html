1   package org.archive.cdxserver.processor;
2   
3   import java.util.LinkedList;
4   
5   import org.archive.format.cdx.CDXLine;
6   
7   public class LastNLineProcessor extends WrappedProcessor {
8       
9       protected LinkedList<CDXLine> lines;
10      protected int limit = 1;
11      
12      public LastNLineProcessor(BaseProcessor output, int limit)
13      {
14          super(output);
15          this.lines = new LinkedList<CDXLine>();
16          this.limit = limit;
17      }
18  
19      @Override
20      public int writeLine(CDXLine line) {
21          lines.add(line);
22          
23          if (lines.size() > limit) {
24              lines.removeFirst();
25          }
26          
27          return 0;
28      }
29      
30      protected void flush()
31      {
32          for (CDXLine line : lines) {
33              inner.writeLine(line);
34          }
35          lines.clear();
36      }
37  
38      @Override
39      public void writeResumeKey(String resumeKey) {
40          if (!lines.isEmpty()) {
41              flush();
42          }
43          
44          inner.writeResumeKey(resumeKey);
45      }
46  
47      @Override
48      public void end() {
49          if (!lines.isEmpty()) {
50              flush();
51          }
52          
53          inner.end();
54      }
55  }

34  /**
35   * {@code ReplayParseContext} holds context information shared among
36   * replay rewriter components.
37   * <p>2014-05-02 small behavior/interface changes:
38   * <ul>
39   * <li>{@link #setJspExec(JSPExecutor)} no longer copies {@code CaptureSearchResult}
40   * object from its {@code UIResults} object to {@code result} member. Use new constructor
41   * taking {@code CaptureSearchResult} object (recommended), or use
42   * {@link #setCaptureSearchResult(CaptureSearchResult)} method.</li>
43   * <li>
44   * </ul>
45   * TODO: consider replacing {@code CaptureSearchResult} reference with {@code Capture}.

64  	/**
65  	 * Constructs {@code ReplayParseContext} for rewriting a resource
66  	 * represented by {@code result}.
67  	 * <p>Initializes {@code baseUrl} and {@code datespec} from {@code result}'s
68  	 * {@code originalUrl} and {@code captureTimestamp}, respectively.</p>
69  	 * @param uriConverterFactory
70  	 * @param result
71  	 * @throws IOException
72  	 */
73  	public ReplayParseContext(ContextResultURIConverterFactory uriConverterFactory,
74  			CaptureSearchResult result) throws IOException {
75  		this.uriConverterFactory = uriConverterFactory;
76  		this.result = result;
77  		setBaseUrl(result.getOriginalUrl());
78  		this.datespec = result.getCaptureTimestamp();
79  
80  		this.converters = new HashMap<String, ResultURIConverter>();
81  	}

89  	 * @deprecated 2014-05-02 use {@link #ReplayParseContext(ContextResultURIConverterFactory, CaptureSearchResult)}
90  	 */
91  	public ReplayParseContext(ContextResultURIConverterFactory uriConverterFactory,
92  			URL baseUrl, String datespec) {

95  		setBaseUrl(baseUrl.toExternalForm());
96  		this.datespec = datespec;

98  	}

111 	/**
112 	 * return {@code true} if {@code url} needs rewrite in this
113 	 * replay.
114 	 * <p>As {@link #contextualizeUrl(String, String)} runs this test,
115 	 * there's no real point doing this check outside of ReplayParseContext.
116 	 * this method may be changed to {@code protected} in the future.</p>
117 	 * @param url URL to test. it must be free of escaping (i.e. no {@code "https:\/\/"}.)
118 	 * @return {@code true} if {@code url} needs rewrite.
119 	 * @see #setRewriteHttpsOnly(boolean)
120 	 */
121 	public boolean isRewriteSupported(String url) {
122 		if (rewriteHttpsOnly)
123 			return url.startsWith(WaybackConstants.HTTPS_URL_PREFIX);
124 		return true;
125 	}

164 	/**
165 	 * returns {@link ResultURIConverter} for resource context <code>flags</code>.
166 	 * @param flags resource context indicator such as "{@code cs_}", "{@code im_}".
167 	 * @return ResultURIConverter for translating URL
168 	 */
169 	public ResultURIConverter getConverter(String flags) {
170 		// TODO: caching should be a responsibility of ContextResultURIConverterFactory.
171 		// but it's a API-breaking change as converters is exposed through getter.
172 		ResultURIConverter converter = converters.get(flags);
173 		if(converter == null) {
174 			converter = uriConverterFactory.getContextConverter(flags);
175 			converters.put(flags,converter);
176 		}
177 		return converter;
178 	}

180         /**
181 	 * Rewrite URL {@code url} in accordance with current replay mode, not using
182 	 * replay context {@code flags}.
183 	 * @param url URL, candidate for rewrite. may contain escaping. must not be {@code null}.
184 	 * @return rewrittenURL, or {@code url} if no rewrite is necessary. never {@code null}.
185 	 */
186         @Override
187 	public String contextualizeUrl(String url) {
188 		return contextualizeUrl(url,"");
189 	}
190         
191 	/**
192 	 * Rewrite URL {@code url} in accordance with current replay mode, taking
193 	 * replay context {@code flags} into account.
194 	 * <p>It is important to return the same String object {@code url} if no rewrite
195 	 * is necessary, so that caller can short-circuit to avoid expensive String operations.</p>
196 	 * @param url URL, candidate for rewrite. may contain escaping. must not be {@code null}.
197 	 * @param flags <em>context</em> designator, such as {@code "cs_"}. can be {@code null}.
198 	 * @return rewrittenURL, or {@code url} if no rewrite is necessary. never {@code null}.
199 	 */
200 	public String contextualizeUrl(final String url, String flags) {
201 		// if we get an empty string, just return it:
202 		if (url.length() == 0) {
203 			return url;
204 		}
205 		
206 		if (url.startsWith(JAVASCRIPT_PREFIX) || url.startsWith(MAILTO_PREFIX) || url.startsWith(ANCHOR_PREFIX)) {
207 	    	return url;
208 	    }
209 	    // XXX duplicated check for MAILTO_PREFIX??
210 		if (url.startsWith(DATA_PREFIX) || url.startsWith(MAILTO_PREFIX)) {
211 	    	return url;
212 	    }
213 
214 		// don't rewrite path-relative urls. For
215 		// https://webarchive.jira.com/browse/ARI-3985
216 		String trimmedUrl = url.trim();
217 
218 		if (!trimmedUrl.startsWith("http://") &&
219 				!trimmedUrl.startsWith("https://") &&
220 				!trimmedUrl.startsWith("//") &&
221 				!trimmedUrl.startsWith("http:\\\\/\\\\/") &&
222 				!trimmedUrl.startsWith("http\\\\u00253A\\\\u00252F\\\\u00252F") &&
223 				!trimmedUrl.startsWith("https:\\\\/\\\\/") &&
224 				!trimmedUrl
225 					.startsWith("https\\\\u00253A\\\\u00252F\\\\u00252F") &&
226 				!trimmedUrl.startsWith("http:\\/\\/") &&
227 				!trimmedUrl.startsWith("https:\\/\\/") &&
228 				!trimmedUrl.startsWith("/") &&
229 				!trimmedUrl.startsWith(".")) {
230 			return url;
231 		}
232 
233 	    // first make url into absolute, taking BASE into account.
234 		// (this also removes escaping: ex. "https:\/\/" -> "https://")
235 	    String absurl = super.contextualizeUrl(url);
236 	    if (!isRewriteSupported(absurl)) {
237 	    	return url;
238 	    }
239 	    // XXX do this in getConverter
240 		if (flags == null) {
241 	    	flags = "";
242 	    }
243 	    ResultURIConverter converter = getConverter(flags);
244 		return converter.makeReplayURI(datespec, absurl);
245 	}

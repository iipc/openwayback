20  package org.archive.wayback.replay.html;

22  import org.archive.wayback.ResultURIConverter;
23  import org.archive.wayback.accesspoint.AccessPointAdapter;
24  import org.archive.wayback.archivalurl.ArchivalURLJSStringTransformerReplayRenderer;
25  import org.archive.wayback.archivalurl.ArchivalUrlSAXRewriteReplayRenderer;
26  
27  /**
28   * Abstracts creation of specialized ResultURIConverters based on particular
29   * flags.
30   * <p>
31   * Currently {@code ContextResultURIConverterFactory} is used in two different
32   * contexts, and it receives critically different information as {@code flags}
33   * argument for each context.
34   * <ol>
35   * <li>customizing the way parent request handler builds
36   * {@link ResultURIConverter} for each of its children
37   *  - receives {@code replayURIPrefix} as {@code flags}.
38   * ({@link AccessPointAdapter})</li>
39   * <li>passing requested resource type (often called <i>flags</i>) information to
40   * {@link ReplayParseContext}
41   * - receives resource type (such as {@code "cs_"}) as {@code flags}.
42   * ({@link ArchivalUrlSAXRewriteReplayRenderer} and
43   * {@link ArchivalURLJSStringTransformerReplayRenderer}).</li>
44   * </ol>
45   * <p>Using single interface in semantically different contexts leads to
46   * a lot of confusion and awkward code. Redesign is highly desired.
47   * Current plan is to do away with usage 2, and design better interface for
48   * usage 1.</p>
49   *
50   * @author brad
51   * @see ReplayParseContext#makeConverter
52   * @see AccessPointAdapter#getUriConverter
53   */
54  public interface ContextResultURIConverterFactory {
55  	public ResultURIConverter getContextConverter(String flags);
56  }

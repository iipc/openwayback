190     /**
191      * Run a regex against a StringBuilder, removing group 1 if it matches.
192      * 
193      * Assumes the regex has a form that wants to strip elements of the passed
194      * string.  Assumes that if a match, group 1 should be removed
195      * @param url Url to search in.
196      * @param matcher Matcher whose form yields a group to remove
197      * @return true if the StringBuilder was modified
198      */
199     protected boolean doStripRegexMatch(StringBuilder url, Matcher matcher) {
200     	if(matcher != null && matcher.matches()) {
201     		url.delete(matcher.start(1), matcher.end(1));
202     		return true;
203     	}
204     	return false;
205     }

207 	public String urlStringToKey(final String urlString) throws URIException {
208 
209 		if(urlString.startsWith("dns:")) {
210 			return urlString;
211 		}
212 		String searchUrl = canonicalize(urlString);
213 		String scheme = UrlOperations.urlToScheme(searchUrl);
214 		if(scheme != null) {
215 			searchUrl = searchUrl.substring(scheme.length());
216 		} else {
217 			scheme = UrlOperations.HTTP_SCHEME;
218 		}
219 	
220 		if (-1 == searchUrl.indexOf("/")) {
221 			searchUrl = scheme + searchUrl + "/";
222 		} else {
223 			searchUrl = scheme + searchUrl;
224 		}
225 
226         // Custom rules
227 
228         for (CanonicalizationRule rule : getProcessingRules()) {
229             searchUrl = rule.processIfMatches(new CanonicalizationInput(searchUrl));
230         }
231 
232         // Core rules
233 
234 		// TODO: These next few lines look crazy -- need to be reworked.. This
235 		// was the only easy way I could find to get the correct unescaping
236 		// out of UsableURIs, possible a bug. Definitely needs some TLC in any case,
237 		// as building UsableURIs is *not* a cheap operation.
238 		
239 		// unescape anything that can be:
240 		UsableURI tmpURI = null;
241 		try {
242 			tmpURI = UsableURIFactory.getInstance(searchUrl);
243 		} catch (StringIndexOutOfBoundsException e) {
244 			LOGGER.warning(e.getMessage() + ": " + searchUrl);
245 			return searchUrl;
246 //		} catch(URIException e) {
247 //			LOGGER.warning(e.getMessage() + ": " + searchUrl);
248 //			return searchUrl;			
249 		}
250 		tmpURI.setPath(tmpURI.getPath());
251 		
252 		// convert to UsableURI to perform required URI fixup:
253 		UsableURI searchURI = UsableURIFactory.getInstance(tmpURI.getURI());
254 		
255 		// replace ' ' with '+' (this is only to match Alexa's canonicalization)
256 		String newPath = searchURI.getEscapedPath().replace("%20","+");
257 		
258 		// replace multiple consecutive '/'s in the path.
259 		while(newPath.contains("//")) {
260 			newPath = newPath.replace("//","/");
261 		}
262 		
263 		// this would remove trailing a '/' character, unless the path is empty
264 		// but we're not going to do this just yet..
265 //		if((newPath.length() > 1) && newPath.endsWith("/")) {
266 //			newPath = newPath.substring(0,newPath.length()-1);
267 //		}
268 		
269 		StringBuilder sb = new StringBuilder(searchUrl.length());
270 		sb.append(searchURI.getHostBasename());
271 
272 		// omit port if scheme default:
273 		int defaultSchemePort = UrlOperations.schemeToDefaultPort(scheme);
274 		if(searchURI.getPort() != defaultSchemePort 
275 				&& searchURI.getPort() != -1) {
276 
277 			sb.append(":").append(searchURI.getPort());
278 		}
279 
280 		sb.append(newPath);
281 		if(searchURI.getEscapedQuery() != null) {
282 			sb.append("?").append(searchURI.getEscapedQuery());
283 		}
284 
285 		return sb.toString();
286 	}

288 	/**
289 	 * Idempotent operation that will determine the 'fuzziest'
290 	 * form of the url argument. This operation is done prior to adding records
291 	 * to the ResourceIndex, and prior to lookup. Current version is exactly
292 	 * the default found in Heritrix. When the configuration system for
293 	 * Heritrix stabilizes, hopefully this can use the system directly within
294 	 * Heritrix.
295 	 * 
296 	 * @param url to be canonicalized.
297 	 * @return canonicalized version of url argument.
298 	 */
299 	public String canonicalize(String url) {
300 
301         if (url == null || url.length() <= 0) {
302             return url;
303         }
304 
305         // hang on, we're about to get aggressive:
306         url = url.toLowerCase();
307         StringBuilder sb = new StringBuilder(url);
308         boolean changed = false;
309 		for(int i=0; i<choosers.length; i++) {
310 			if(sb.indexOf(choosers[i]) != -1) {
311 				changed |= doStripRegexMatch(sb,strippers[i].matcher(sb));
312 			}
313 		}
314 		if(changed) {
315 			url = sb.toString();
316 		}
317         
318         int index = url.lastIndexOf('?');
319         if (index > 0) {
320             if (index == (url.length() - 1)) {
321                 // '?' is last char in url.  Strip it.
322                 url = url.substring(0, url.length() - 1);
323             } else if (url.charAt(index + 1) == '&') {
324                 // Next char is '&'. Strip it.
325                 if (url.length() == (index + 2)) {
326                     // Then url ends with '?&'.  Strip them.
327                     url = url.substring(0, url.length() - 2);
328                 } else {
329                     // The '&' is redundant.  Strip it.
330                     url = url.substring(0, index + 1) + 
331                     	url.substring(index + 2);
332                 }
333             } else if (url.charAt(url.length() - 1) == '&') {
334                 // If we have a lone '&' on end of query str,
335                 // strip it.
336                 url = url.substring(0, url.length() - 1);
337             }
338         }
339         return url;
340 	}

342 	private static void USAGE() {
343 		System.err.println("Usage: [-f FIELD] [-d DELIM]");
344 		System.exit(3);
345 	}
346 	/**
347 	 * @param args program arguments
348 	 */
349 	public static void main(String[] args) {
350 		AggressiveUrlCanonicalizer canonicalizer = new AggressiveUrlCanonicalizer();
351 		int n = 0;
352 		int i = 0;
353 		ArrayList<Integer> columns = new ArrayList<Integer>();
354 		
355 		long lineNumber = 0;
356 		boolean cdxPassThru = false;
357 		String delimiter = " ";
358 		while(n < args.length) {
359 			String arg = args[n];
360 			if(arg.compareTo("-cdx") == 0) {
361 				cdxPassThru = true;
362 				n++;
363 				continue;
364 			}
365 			if(n == (args.length -1)) {
366 				USAGE();
367 			}
368 			String val = args[n+1];
369 			if(arg.compareTo("-f") == 0) {
370 				columns.add(new Integer(val));
371 			} else if(arg.compareTo("-d") == 0) {
372 				delimiter = val;
373 			} else {
374 				USAGE();
375 			}
376 			n += 2;
377 		}
378 		// place default '0' in case none specified:
379 		if(columns.size() == 0) {
380 			columns.add(new Integer(1));
381 		}
382 		
383 		// convert to int[]:
384 		int[] cols = new int[columns.size()];
385 		for(int idx = 0; idx < columns.size(); idx++) {
386 			cols[idx] = columns.get(idx).intValue() - 1;
387 		}
388 		BufferedReader r = new BufferedReader(new InputStreamReader(System.in,ByteOp.UTF8));
389 		StringBuilder sb = new StringBuilder();
390 		String line = null;
391 		
392 		while(true) {
393 			try {
394 				line = r.readLine();
395 			} catch (IOException e) {
396 				e.printStackTrace();
397 				System.exit(1);
398 			}
399 			if(line == null) {
400 				break;
401 			}
402 			lineNumber++;
403 			if(cdxPassThru && line.startsWith(CDX_PREFIX)) {
404 				System.out.println(line);
405 				continue;
406 			}
407 			String parts[] = line.split(delimiter);
408 			for(int column : cols) {
409 				if(column >= parts.length) {
410 					System.err.println("Invalid line " + lineNumber + " (" +
411 							line + ") skipped");
412 				} else {
413 					try {
414 						parts[column] = canonicalizer.urlStringToKey(parts[column]);
415 					} catch (URIException e) {
416 						System.err.println("Invalid URL in line " + lineNumber + " (" +
417 								line + ") skipped (" + parts[column] + ")");
418 						e.printStackTrace();
419 						continue;
420 					} catch (StringIndexOutOfBoundsException e) {
421 						System.err.println("Invalid URL in line " + lineNumber + " (" +
422 								line + ") skipped (" + parts[column] + ")");
423 						e.printStackTrace();
424 						continue;
425 					}
426 				}
427 			}
428 			sb.setLength(0);
429 			for(i = 0; i < parts.length; i++) {
430 				sb.append(parts[i]);
431 				if(i < (parts.length-1)) {
432 					sb.append(delimiter);
433 				}
434 			}
435 			System.out.println(sb.toString());
436 		}
437 	}

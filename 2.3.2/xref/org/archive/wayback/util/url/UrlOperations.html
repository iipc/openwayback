124         Pattern.compile("(([0-9a-z_.-]+)\\.(" + ALL_TLDS + "))|" +

134     /**
135      * Tests if the String argument looks like it could be a legitimate 
136      * authority fragment of a URL, that is, is it an IP address, or, are the
137      * characters legal in an authority, and does the string end with a legal
138      * TLD.
139      * 
140 	 * @param authString String representation of a fragment of a URL 
141 	 * @return boolean indicating whether urlPart might be an Authority.
142 	 */
143 	public static boolean isAuthority(String authString) {
144 		Matcher m = AUTHORITY_REGEX.matcher(authString);
145 		
146 		return (m != null) && m.matches();
147 	}

149 	/** Resolve URL, but return a minimally escaped version in case of
150 	 *  error
151 	 * @param baseUrl the base URL against which the url should be resolved
152 	 * @param url the URL, possibly relative, to make absolute.
153 	 * @return url resolved against baseUrl, unless it is absolute already, and
154 	 * further transformed by whatever escaping normally takes place with a 
155 	 * UsableURI.
156 	 * In case of error, return URL.
157 	 */
158 	public static String resolveUrl(String baseUrl, String url) {
159 		String resolvedUrl = resolveUrl(baseUrl, url, null);
160 		if (resolvedUrl == null) {
161 			resolvedUrl = url.replace(" ", "%20");
162 			resolvedUrl = resolvedUrl.replace("\r", "%0D");
163 		}
164 		return resolvedUrl;
165 	}

167 	/**
168 	 * Resolve a possibly relative url argument against a base URL.
169 	 * @param baseUrl the base URL against which the url should be resolved
170 	 * @param url the URL, possibly relative, to make absolute.
171          * @param defaultValue The default value to return if the supplied values can't be resolved.
172 	 * @return url resolved against baseUrl, unless it is absolute already, and
173 	 * further transformed by whatever escaping normally takes place with a 
174 	 * UsableURI.
175 	 * In case of error, return the defaultValue
176 	 */
177 	public static String resolveUrl(String baseUrl, String url, String defaultValue) {
178 				
179 		for(final String scheme : ALL_SCHEMES) {
180 			if(url.startsWith(scheme)) {
181 				try {
182 					return UsableURIFactory.getInstance(url).getEscapedURI();
183 				} catch (URIException e) {
184 					LOGGER.warning(e.getLocalizedMessage() + ": " + url);
185 					// can't let a space exist... send back close to whatever came
186 					// in...
187 					return defaultValue;
188 				}
189 			}
190 		}
191 		UsableURI absBaseURI;
192 		UsableURI resolvedURI = null;
193 		try {
194 			absBaseURI = UsableURIFactory.getInstance(baseUrl);
195 			resolvedURI = UsableURIFactory.getInstance(absBaseURI, url);
196 		} catch (URIException e) {
197 			LOGGER.warning(e.getLocalizedMessage() + ": " + url);
198 			return defaultValue;
199 		}
200 		
201 		return resolvedURI.getEscapedURI();
202 	}

204 	/**
205 	 * Attempt to find the scheme (http://, https://, etc) from a given URL.
206 	 * @param url URL String to parse for a scheme.
207 	 * @return the scheme, including trailing "://" if known, null otherwise.
208 	 */
209 	public static String urlToScheme(final String url) {
210 		for(final String scheme : ALL_SCHEMES) {
211 			if(url.startsWith(scheme)) {
212 				return scheme;
213 			}
214 		}
215 		return null;
216 	}

218 	/**
219 	 * Return the default port for the scheme String argument, if known.
220 	 * @param scheme String scheme, including '://', as in, "http://", "ftp://"
221 	 * @return the default port for the scheme, or -1 if the scheme isn't known.
222 	 */
223 	public static int schemeToDefaultPort(final String scheme) {
224 		if(scheme.equals(HTTP_SCHEME)) {
225 			return 80;
226 		}
227 		if(scheme.equals(HTTPS_SCHEME)) {
228 			return 443;
229 		}
230 		if(scheme.equals(FTP_SCHEME)) {
231 			return 21;
232 		}
233 		if(scheme.equals(RTSP_SCHEME)) {
234 			return 554;
235 		}
236 		if(scheme.equals(MMS_SCHEME)) {
237 			return 1755;
238 		}
239 		return -1;
240 	}

243 	 * Attempt to extract the path component of a url String argument.
244 	 * @param url the URL which may contain a path, sans scheme.
245 	 * @return the path component of the URL, or "" if it contains no path.
246 	 */
247 	public static String getURLPath(String url) {
248 		url = stripURLScheme(url);
249 		int pathIdx = url.indexOf(UrlOperations.PATH_START);
250 		if(pathIdx == -1) {
251 			return "/";
252 		}
253 		return url.substring(pathIdx);
254 	}
255 	/**
256 	 * Attempt to extract the path component of a url String argument.
257 	 * @param url the URL which may contain a path, sans scheme.
258 	 * @return the path component of the URL, or "" if it contains no path.
259 	 */
260 	public static String stripURLScheme(String url) {
261 		String lcUrl = url.toLowerCase();
262 		for(String scheme : ALL_SCHEMES) {
263 			if(lcUrl.startsWith(scheme)) {
264 				return url.substring(scheme.length());
265 			}
266 		}
267 		return url;
268 	}	
269 	/**
270 	 * Attempt to strip default ports out of URL strings.
271 	 * @param url the original URL possibly including a port
272 	 * @return the URL sans port, if the scheme was recognized and the default
273 	 * port was supplied, otherwise, the original URL.
274 	 */
275 	public static String stripDefaultPortFromUrl(String url) {
276 		String scheme = urlToScheme(url);
277 		if(scheme == null) {
278 			return url;
279 		}
280 		int defaultPort = schemeToDefaultPort(scheme);
281 		if(defaultPort == -1) {
282 			return url;
283 		}
284 
285 		String portStr = null;
286 		// is there a slash after the scheme?
287 		int slashIdx = url.indexOf('/', scheme.length());
288 		if(slashIdx == -1) {
289 			portStr = String.format(":%d", defaultPort);
290 			if(url.endsWith(portStr)) {
291 				return url.substring(0,url.length() - portStr.length());
292 			}
293 		}
294 		portStr = String.format(":%d/", defaultPort);
295 		int idx = url.indexOf(portStr);
296 		if(idx == -1) {
297 			return url;
298 		}
299 		// if that occurred before the first / (after the scheme) then strip it:
300 		if(slashIdx < idx) {
301 			return url;
302 		}
303 		// we want to strip out the portStr:
304 		StringBuilder sb = new StringBuilder(url.length());
305 		sb.append(url.substring(0,idx));
306 		sb.append(url.substring(idx + (portStr.length()-1)));
307 		return sb.toString();
308 	}

311 	 * @param orig String containing a URL, possibly beginning with "http:/".
312 	 * @return original string if orig begins with "http://", or a new String
313 	 * with the extra slash, if orig only had one slash.
314 	 * @see #fixupScheme
315 	 */
316 	public static String fixupHTTPUrlWithOneSlash(String orig) {
317 		if(orig.startsWith("http:/") && ! orig.startsWith(HTTP_SCHEME)) {
318 			// very likely the IE "you must have meant 1 slash, not 2 bug:
319 			StringBuilder sb = new StringBuilder(orig.length()+1);
320 			sb.append(HTTP_SCHEME);
321 			return sb.append(orig.substring(6)).toString();
322 		}
323 		return orig;
324 	}

326 	/**
327 	 * fixes up malformed scheme part.
328 	 * <p>currently supports fixing missing second slash for protocols
329 	 * {@code http}, {@code https}, {@code ftp}, {@code rtsp} and
330 	 * {@code mms}. For example fixing {@code http:/} to {@code https://}</p>
331 	 * @param url URL to be checked and fixed
332 	 * @return new String, or {@code url} if not fix is required.
333 	 * @since 1.8.1
334 	 */
335 	public static String fixupScheme(String url) {
336 		final String[] SCHEMES = {
337 			"http:/", "https:/", "ftp:/", "rtsp:/", "mms:/"
338 		};
339 		int ul = url.length();
340 		for (String scheme : SCHEMES) {
341 			int sl = scheme.length();
342 			if (url.startsWith(scheme) && (ul == sl || url.charAt(sl) != '/')) {
343 				return scheme + "/" + url.substring(sl);
344 			}
345 		}
346 		return url;
347 	}

349 	/**
350 	 * Attempt to extract the hostname component of an absolute URL argument.
351 	 * @param url the url String from which to extract the hostname
352 	 * @return the hostname within the URL, or the url argument if the host 
353 	 * cannot be found.
354 	 */
355 	public static String urlToHost(String url) {
356 		String lcUrl = url.toLowerCase();
357 		if(lcUrl.startsWith(DNS_SCHEME)) {
358 			return lcUrl.substring(DNS_SCHEME.length());
359 		}
360 		for(String scheme : ALL_SCHEMES) {
361 			if(lcUrl.startsWith(scheme)) {
362 				int authorityIdx = scheme.length();
363 
364 				Matcher m = 
365 					HOST_REGEX_SIMPLE.matcher(lcUrl.substring(authorityIdx));
366 				if(m.find()) {
367 					return m.group(1);
368 				}
369 			}
370 		}
371 		return url;
372 	}

374 	/**
375 	 * Extract userinfo from the absolute URL argument, that is, "username@", or
376 	 * "username:password@" if present.
377 	 * @param url the URL from which to extract the userinfo
378 	 * @return the userinfo found, not including the "@", or null if no userinfo
379 	 * is found
380 	 */
381 	public static String urlToUserInfo(String url) {
382 		String lcUrl = url.toLowerCase();
383 		if(lcUrl.startsWith(DNS_SCHEME)) {
384 			return null;
385 		}
386 		for(String scheme : ALL_SCHEMES) {
387 			if(lcUrl.startsWith(scheme)) {
388 				int authorityIdx = scheme.length();
389 
390 				Matcher m = 
391 					USERINFO_REGEX_SIMPLE.matcher(lcUrl.substring(authorityIdx));
392 				if(m.find()) {
393 					return m.group(1);
394 				}
395 			}
396 		}
397 		return null;
398 	}

400 	/**
401 	 * Find and return the parent directory of the URL argument
402 	 * @param url to find the parent directory of
403 	 * @return parent directory of URL, or null, if either the url argument is
404 	 * invalid, or if the url is the root of the authority.
405 	 */
406 	public static String getUrlParentDir(String url) {
407 		try {
408 			UsableURI uri = UsableURIFactory.getInstance(url);
409 			String path = uri.getPath();
410 			if(path.length() > 1) {
411 				int startIdx = path.length()-1;
412 				if(path.charAt(path.length()-1) == '/') {
413 					startIdx--;
414 				}
415 				int idx = path.lastIndexOf('/',startIdx);
416 				if(idx >= 0) {
417 					uri.setPath(path.substring(0,idx+1));
418 					uri.setQuery(null);
419                     return uri.toUnicodeHostString();
420 				}
421 			}
422 		} catch (URIException e) {
423 			LOGGER.warning(e.getLocalizedMessage() + ": " + url);
424 		}
425 		return null;
426 	}

429 	 * build replay Archival-URL for the same capture as request
430 	 * {@code wbRequest}, with identity-context ({@code id_}) flag on.
431 	 * <p>
432 	 * REFACTOR: move this method to {@link ArchivalUrl}.
433 	 * </p>
434 	 * @param wbRequest requested capture and URL scheme info.
435 	 * @return URL string
436 	 */
437 	public static String computeIdentityUrl(WaybackRequest wbRequest) {
438 		AccessPoint accessPoint = wbRequest.getAccessPoint();
439 
440 		boolean origIdentity = wbRequest.isIdentityContext();
441 		wbRequest.setIdentityContext(true);
442 		
443 		ArchivalUrl aUrl = new ArchivalUrl(wbRequest);
444 		String bestPath = aUrl.toString();
445 		String betterURI = accessPoint.getReplayPrefix() + bestPath;
446 
447 		//reset the isIdentity flag just in case
448 		wbRequest.setIdentityContext(origIdentity);
449 		
450 		return betterURI;
451 	}

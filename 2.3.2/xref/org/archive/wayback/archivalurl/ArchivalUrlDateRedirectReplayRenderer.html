22  import java.io.IOException;
23  
24  import javax.servlet.ServletException;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  
28  import org.archive.wayback.ReplayRenderer;
29  import org.archive.wayback.ResultURIConverter;
30  import org.archive.wayback.core.CaptureSearchResult;
31  import org.archive.wayback.core.CaptureSearchResults;
32  import org.archive.wayback.core.Resource;
33  import org.archive.wayback.core.WaybackRequest;
34  
35  /**
36   *
37   * @deprecated 1.8.1 2014-06-26 with no replacement.
38   * @author brad
39   */
40  
41  public class ArchivalUrlDateRedirectReplayRenderer implements ReplayRenderer {
42  
43  	/* (non-Javadoc)
44  	 * @see org.archive.wayback.ReplayRenderer#renderResource(javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse, org.archive.wayback.core.WaybackRequest, org.archive.wayback.core.SearchResult, org.archive.wayback.core.Resource, org.archive.wayback.ResultURIConverter, org.archive.wayback.core.SearchResults)
45  	 */
46  	public void renderResource(HttpServletRequest httpRequest,
47  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
48  			CaptureSearchResult result, Resource resource,
49  			ResultURIConverter uriConverter, CaptureSearchResults results)
50  					throws ServletException, IOException {
51  
52  		// redirect to the better version:
53  		String url = result.getOriginalUrl();
54  		ArchivalUrl aUrl = new ArchivalUrl(wbRequest);
55  		String dateSpec = aUrl.getDateSpec(result.getCaptureTimestamp());
56  		String betterURI = uriConverter.makeReplayURI(dateSpec,url);
57  		httpResponse.sendRedirect(betterURI);
58  	}
59  
60  	@Override
61  	public void renderResource(HttpServletRequest httpRequest,
62  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
63  			CaptureSearchResult closest, Resource httpHeadersResource,
64  			Resource payloadResource, ResultURIConverter uriConverter,
65  			CaptureSearchResults captureResults) throws ServletException, IOException {
66  		renderResource(httpRequest, httpResponse, wbRequest, closest,
67  				httpHeadersResource, uriConverter, captureResults);
68  	}
69  }

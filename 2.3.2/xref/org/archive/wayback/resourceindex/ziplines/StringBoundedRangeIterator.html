1   package org.archive.wayback.resourceindex.ziplines;
2   
3   import java.util.Iterator;
4   
5   public class StringBoundedRangeIterator extends StringPrefixIterator {
6   	
7   	protected String startMatch;
8   	protected String endMatch;
9   	protected boolean endPrefixInclusive;
10  	
11  	/**
12  	 * 
13  	 * @param inner - underlying cdx line iterator
14  	 * @param startMatch - string match to start with (inclusive)
15  	 * @param endMatch - end match to end with (exclusive or inclusive)
16  	 * @param endPrefixInclusive - specify if end match is exclusive or inclusive (if inclusive, use startsWith to check the end)
17  	 */
18  	public StringBoundedRangeIterator(Iterator<String> inner, String startMatch, String endMatch, boolean endPrefixInclusive) {
19  		super(inner, startMatch);
20  		this.startMatch = startMatch;
21  		this.endMatch = endMatch;
22  		this.endPrefixInclusive = endPrefixInclusive;
23  		
24  		if (startMatch.compareTo(endMatch) > 0) {
25  			throw new RuntimeException("StringBoundRangeIterator: start > end" + startMatch + " > " + endMatch);
26  		}
27  	}
28  
29  	@Override
30  	public boolean hasNext() {
31  		if (done) {
32  			return false;
33  		}
34  		
35  		if (cachedNext != null) {
36  			return true;
37  		}
38  		
39  		while(inner.hasNext()) {
40  			String currLine = inner.next();
41  			
42  			// If past the end line, and doesn't start with
43  			// endMatch, we're done
44  			if ((currLine.compareTo(endMatch) > 0) &&
45  				!currLine.startsWith(endMatch)) {
46  				done = true;
47  				return false;
48  			}
49  			
50  			// If currLine before startMatch, skip over
51  			if (currLine.compareTo(startMatch) < 0) {
52  				continue;
53  			}
54  			
55  			// Otherwise, startMatch <= currLine < endMatch
56  			cachedNext = currLine;
57  			return true;
58  		}
59  		
60  		return false;
61  	}
62  
63  	
64  	
65  }

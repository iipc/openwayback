147 	@Override
148 	public boolean includeUrl(String urlKey, String originalUrl) {
149 //		return include(urlKey, originalUrl, true);
150 		if (UrlOperations.urlToScheme(originalUrl) == null) {
151 			originalUrl = UrlOperations.HTTP_SCHEME + originalUrl;
152 		}
153 
154 		CaptureSearchResult resultTester = new FastCaptureSearchResult();
155 		resultTester.setUrlKey(urlKey);
156 		resultTester.setOriginalUrl(originalUrl);
157 		// null captureTimestamp signifies per-URL access-check.
158 		resultTester.setCaptureTimestamp(null);
159 
160 		return include(resultTester, true);
161 	}

163 	/**
164 	 * Adapts CDXLine to CaptureSearchResult interface. Fetches
165 	 * {@code originalUrl}, {@code captureTimestamp} and {@code robotFlags} from
166 	 * the {@code CDXLine} adopted (minimum required for known existing filter
167 	 * implementations). It also have {@code setRobotFlag} call modify
168 	 * {@code robotflags} field in the underlining {@code CDXLine} (necessary
169 	 * for soft-block feature).
170 	 * <p>
171 	 * TODO: Unfortunately this is not as lightweight as it should have been.
172 	 * Only if CaptureSearchResult was an interface.
173 	 * </p>
174 	 * <p>
175 	 * Caveat: it only overrides those methods used by known existing filter
176 	 * implementations.
177 	 * </p>
178 	 * @see CustomPolicyOracleFilter
179 	 */
180 	protected static class CDXSearchResult extends FastCaptureSearchResult {
181 		final CDXLine cdxLine;
182 
183 		public CDXSearchResult(CDXLine cdxLine) {
184 			this.cdxLine = cdxLine;
185 		}
186 
187 		@Override
188 		public String getUrlKey() {
189 			return cdxLine.getUrlKey();
190 		}
191 
192 		@Override
193 		public final String getOriginalUrl() {
194 			return cdxLine.getOriginalUrl();
195 		}
196 
197 		// CustomPolicyOracleFilter calls getCaptureDate(),
198 		// which is implemented by CaptureSearchResult on top
199 		// of getCaptureTimestamp()
200 		@Override
201 		public final String getCaptureTimestamp() {
202 			return cdxLine.getTimestamp();
203 		}
204 
205 		@Override
206 		public final void setRobotFlag(char flag) {
207 			String robotFlags = cdxLine.getRobotFlags();
208 			if (robotFlags == null || robotFlags.equals("-")) {
209 				setRobotFlags(Character.toString(flag));
210 			} else {
211 				if (robotFlags.indexOf(flag) == -1) {
212 					setRobotFlags(robotFlags + flag);
213 				}
214 			}
215 		}
216 
217 		@Override
218 		public final void setRobotFlags(String robotFlags) {
219 			// CDXLine does not have setter for robotFlags,
220 			// but setField method is available.
221 			cdxLine.setField(CDXLine.robotflags, robotFlags);
222 		}
223 	}

228

234 			return false;

240 		if (cdxFilter2 != null && !cdxFilter2.include(line))
241 			return false;
242 
243 		return true;
244 	}

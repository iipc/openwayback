20  package org.archive.wayback.resourceindex.filterfactory;
21  
22  import java.io.IOException;
23  import java.util.ArrayList;
24  import java.util.HashMap;
25  import java.util.List;
26  import java.util.Map;
27  
28  import org.archive.wayback.UrlCanonicalizer;
29  import org.archive.wayback.core.CaptureSearchResult;
30  import org.archive.wayback.core.SearchResults;
31  import org.archive.wayback.core.WaybackRequest;
32  import org.archive.wayback.exception.BadQueryException;
33  import org.archive.wayback.resourceindex.filters.DateRangeFilter;
34  import org.archive.wayback.resourceindex.filters.EndDateFilter;
35  import org.archive.wayback.resourceindex.filters.HostMatchFilter;
36  import org.archive.wayback.resourceindex.filters.SchemeMatchFilter;
37  import org.archive.wayback.resourceindex.filters.SelfRedirectFilter;
38  import org.archive.wayback.resourceindex.filters.UrlMatchFilter;
39  import org.archive.wayback.resourceindex.filters.UrlPrefixMatchFilter;
40  import org.archive.wayback.util.ObjectFilter;
41  import org.archive.wayback.util.ObjectFilterChain;
42  import org.archive.wayback.util.Timestamp;
43  import org.archive.wayback.util.url.UrlOperations;
44  
45  public class QueryCaptureFilterGroup implements CaptureFilterGroup {
46  
47  	private ObjectFilterChain<CaptureSearchResult> chain = null;
48  	private String requestType = null;
49  	private String keyUrl = null;
50  	private String startDate;
51  	private String endDate;
52  	private String exactDate;
53  	/**
54  	 * List of URL Strings that are "close" to the current request, but not
55  	 * included in the current CaptureSearchResults.
56  	 */
57  	private Map<String,String> closeMatches = new HashMap<String,String>();
58  
59  
60  	public QueryCaptureFilterGroup(WaybackRequest request, 
61  			UrlCanonicalizer canonicalizer) 
62  	throws BadQueryException {
63  		
64  		requestType = request.get(WaybackRequest.REQUEST_TYPE);
65  
66  		// URL-Filters:
67  		chain = new ObjectFilterChain<CaptureSearchResult>();
68  		try {
69  			keyUrl = canonicalizer.urlStringToKey(request.getRequestUrl());
70  		} catch (IOException e) {
71  			throw new BadQueryException("Bad request URL(" + 
72  					request.getRequestUrl() +")");
73  		}
74  		// Date-Filters:
75  		startDate = request.getStartTimestamp();
76  		if(startDate == null) {
77  				startDate = Timestamp.earliestTimestamp().getDateStr();
78  		}
79  		endDate = request.getEndTimestamp();
80  		if(endDate == null) {
81  				endDate = Timestamp.latestTimestamp().getDateStr();
82  		}
83  		if(request.isReplayRequest()) {
84  			exactDate = request.getReplayTimestamp();
85  			if(exactDate == null) {
86  				exactDate = Timestamp.latestTimestamp().getDateStr();
87  			}
88  			chain.addFilter(new UrlMatchFilter(keyUrl));
89  			chain.addFilter(new SelfRedirectFilter(canonicalizer));
90  			
91  			long wantMS = request.getReplayDate().getTime();
92  			if(request.getAccessPoint().isUseAnchorWindow()) {
93  				// use AnchorTimestamp, if specified:
94  				String anchorTS = request.getAnchorTimestamp();
95  				if(anchorTS != null) {
96  					wantMS = 
97  						Timestamp.parseBefore(anchorTS).getDate().getTime();
98  				}
99  			}
100 
101 		} else if(request.isCaptureQueryRequest()) {
102 			chain.addFilter(new UrlMatchFilter(keyUrl));
103 			chain.addFilter(new SelfRedirectFilter(canonicalizer));
104 			// OPTIMIZ: EndDateFilter is a hard stop: ABORT
105 			//          DateRangeFilter is an INCLUDE/EXCLUDE
106 			//          one class which EXCLUDEs before startDate, and ABORTs
107 			//              after endDate would save a compare..
108 			chain.addFilter(new EndDateFilter(endDate));
109 			chain.addFilter(new DateRangeFilter(startDate, endDate));
110 		} else if(request.isUrlQueryRequest()) {
111 			chain.addFilter(new UrlPrefixMatchFilter(keyUrl));
112 			chain.addFilter(new DateRangeFilter(startDate, endDate));
113 		}
114 
115 		
116 		// Other Filters:
117 		if(request.isExactHost()) {
118 			chain.addFilter(
119 					new HostMatchFilter(
120 							UrlOperations.urlToHost(request.getRequestUrl()),
121 							this)
122 					);
123 		}
124 
125 		if(request.isExactScheme()) {
126 			chain.addFilter(new SchemeMatchFilter(
127 					UrlOperations.urlToScheme(request.getRequestUrl()),this));
128 		}
129 	}
130 
131 	public List<ObjectFilter<CaptureSearchResult>> getFilters() {
132 		return chain.getFilters();
133 	}
134 
135 	public void annotateResults(SearchResults results) {
136 
137 		// set the filter properties on the results:
138 		results.putFilter(WaybackRequest.REQUEST_URL, keyUrl);
139 		results.putFilter(WaybackRequest.REQUEST_START_DATE, startDate);
140 		results.putFilter(WaybackRequest.REQUEST_END_DATE, endDate);
141 		if(exactDate != null) {
142 			results.putFilter(WaybackRequest.REQUEST_EXACT_DATE, exactDate);
143 		}
144 		results.putFilter(WaybackRequest.REQUEST_TYPE, requestType);
145 		if(!closeMatches.isEmpty()) {
146 			results.setCloseMatches(new ArrayList<String>(closeMatches.values()));
147 		}
148 	}
149 
150 	public void addCloseMatch(String host, String closeMatch) {
151 		closeMatches.put(host, closeMatch);
152 	}
153 }

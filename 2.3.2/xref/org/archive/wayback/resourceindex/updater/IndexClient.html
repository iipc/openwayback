47  public class IndexClient {
48  	private static final Logger LOGGER = Logger.getLogger(IndexClient
49  			.class.getName());
50  	
51  	private String target = null;
52  	private File tmpDir = null;
53  	
54  	private HttpClient client = new HttpClient();
55  
56  	/**
57  	 * @param cdx
58  	 * @return true if CDX was added to local or remote index
59  	 * @throws HttpException
60  	 * @throws IOException
61  	 */
62  	public boolean addCDX(File cdx) throws HttpException, IOException {
63  		boolean added = false;
64  		if(target == null) {
65  			throw new IOException("No target set");
66  		}
67  		String base = cdx.getName();
68  		if(target.startsWith("http://")) {
69  			String finalUrl = target;
70  			if(target.endsWith("/")) {
71  				finalUrl = target + base;
72  			} else {
73  				finalUrl = target + "/" + base;
74  			}
75  			PutMethod method = new PutMethod(finalUrl);
76  	        method.setRequestEntity(new InputStreamRequestEntity(
77  	        		new FileInputStream(cdx)));
78  
79  			int statusCode = client.executeMethod(method);
80  	        if (statusCode == HttpStatus.SC_OK) {
81  		        LOGGER.info("Uploaded cdx " + cdx.getAbsolutePath() + " to " +
82  		        		finalUrl);
83  				if(!cdx.delete()) {
84  					throw new IOException("FAILED delete " + 
85  							cdx.getAbsolutePath());
86  				}
87  
88  		        added = true;
89  	        } else {
90  	            throw new IOException("Method failed: " + method.getStatusLine()
91  	            		+ " for URL " + finalUrl + " on file " 
92  	            		+ cdx.getAbsolutePath());
93  	        }
94  			
95  		} else {
96  			// assume a local directory:
97  			File toBeMergedDir = new File(target);
98  			if(!toBeMergedDir.exists()) {
99  				toBeMergedDir.mkdirs();
100 			}
101 			if(!toBeMergedDir.exists()) {
102 				throw new IOException("Target " + target + " does not exist");
103 			}
104 			if(!toBeMergedDir.isDirectory()) {
105 				throw new IOException("Target " + target + " is not a dir");
106 			}
107 			if(!toBeMergedDir.canWrite()) {
108 				throw new IOException("Target " + target + " is not writable");
109 			}
110 			File toBeMergedFile = new File(toBeMergedDir,base);
111 			if(toBeMergedFile.exists()) {
112 				LOGGER.warning("WARNING: "+toBeMergedFile.getAbsolutePath() +
113 						"already exists!");
114 			} else {
115 				if(cdx.renameTo(toBeMergedFile)) {
116 					LOGGER.info("Queued " + toBeMergedFile.getAbsolutePath() + 
117 							" for merging.");
118 					added = true;
119 				} else {
120 					LOGGER.severe("FAILED rename("+cdx.getAbsolutePath()+
121 							") to ("+toBeMergedFile.getAbsolutePath()+")");
122 				}
123 			}
124 		}
125 		return added;
126 	}
127 	
128 	/**
129 	 * @param base
130 	 * @param itr
131 	 * @return true if data was added to local or remote index
132 	 * @throws HttpException
133 	 * @throws IOException
134 	 */
135 	public boolean addSearchResults(String base, Iterator<CaptureSearchResult> itr) 
136 	throws HttpException, IOException {
137 		
138 		if(tmpDir == null) {
139 			throw new IOException("No tmpDir argument");
140 		}
141 		File tmpFile = new File(tmpDir,base);
142 		if(tmpFile.exists()) {
143 			// TODO: is this safe?
144 			if(!tmpFile.delete()) {
145 				throw new IOException("Unable to remove tmp " + 
146 						tmpFile.getAbsolutePath());
147 			}
148 		}
149 		FileOutputStream os = new FileOutputStream(tmpFile);
150 		BufferedOutputStream bos = new BufferedOutputStream(os);
151 		PrintWriter pw = new PrintWriter(bos);
152 		
153 		Adapter<CaptureSearchResult,String> adapterSRtoS = 
154 			new SearchResultToCDXLineAdapter();
155 		Iterator<String> itrS = 
156 			new AdaptedIterator<CaptureSearchResult,String>(itr,adapterSRtoS);
157 		
158 		while(itrS.hasNext()) {
159 			pw.println(itrS.next());
160 		}
161 		pw.close();
162 		boolean added = addCDX(tmpFile);
163 		return added;
164 	}
165 
166 	/**
167 	 * @return the target
168 	 */
169 	public String getTarget() {
170 		return target;
171 	}
172 
173 	/**
174 	 * @param target the target to set
175 	 */
176 	public void setTarget(String target) {
177 		this.target = target;
178 	}
179 
180 	/**
181 	 * @return the tmpDir
182 	 */
183 	public String getTmpDir() {
184 		if(tmpDir == null) {
185 			return null;
186 		}
187 		return tmpDir.getAbsolutePath();
188 	}
189 
190 	/**
191 	 * @param tmpDir the tmpDir to set
192 	 */
193 	public void setTmpDir(String tmpDir) {
194 		this.tmpDir = new File(tmpDir);
195 		if(!this.tmpDir.isDirectory()) {
196 			this.tmpDir.mkdirs();
197 		}
198 	}

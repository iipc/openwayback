26  	private final static String ROBOT_SUFFIX = "/robots.txt";
27  	private final static Logger LOGGER = 
28  		Logger.getLogger(HRobotExclusionFilter.class.getName());
29  
30  	// TODO: this is not the right thing!
31  	private Charset cs = Charset.forName("UTF-8");

33  	private RobotsDirectiveAggregation aggregation = null;

35  
36  	private String userAgent = null;
37  	private boolean notifiedSeen = false;
38  	private boolean notifiedPassed = false;
39  	private static final FixedRobotsDirectives ALLOW_ROBOT_DIRECTIVE = 
40  		new FixedRobotsDirectives(true);

42  	/**
43  	 * Construct a new HRobotExclusionFilter that uses webCache to pull 
44  	 * robots.txt documents. filtering is based on userAgent, and cached 
45  	 * documents newer than maxCacheMS in the webCache are considered valid.
46  	 * 
47  	 * @param webCache LiveWebCache from which documents can be retrieved 
48  	 * @param userAgent String user agent to use for requests to the live web.
49  	 * @param maxCacheMS long number of milliseconds to cache documents in the
50  	 *                   LiveWebCache
51  	 */
52  	public HRobotExclusionFilter(LiveWebCache webCache, String userAgent,
53  			long maxCacheMS) {
54  		aggregation = new RobotsDirectiveAggregation();
55  		this.webCache = webCache;
56  		this.userAgent = userAgent;
57  	}
58  	
59  	private void updateAggregation(String host) 
60  	throws LiveWebCacheUnavailableException,
61  	LiveWebTimeoutException, MalformedURLException, IOException {
62  	
63  		List<String> missing = aggregation.getMissingRobotUrls(host);
64  		for(String robotUrl : missing) {
65  			long start = System.currentTimeMillis();
66  			Resource resource;
67  			try {
68  				resource = webCache.getCachedResource(new URL(robotUrl),
69  						0,true);
70  				if(resource.getStatusCode() != 200) {
71  					LOGGER.info("ROBOT: Non200("+robotUrl+")");
72  					// consider it an allow:
73  					aggregation.addDirectives(robotUrl, ALLOW_ROBOT_DIRECTIVE);
74  				} else {
75  					InputStreamReader isr = new InputStreamReader(resource, cs);
76  					BufferedReader br = new BufferedReader(isr);
77  					Robotstxt robotsTxt = new Robotstxt(br);
78  					RobotsDirectives directives = robotsTxt.getDirectivesFor(userAgent);
79  					aggregation.addDirectives(robotUrl, directives);
80  				}
81  			} catch (LiveDocumentNotAvailableException e) {
82  				if(LOGGER.isLoggable(Level.INFO)) {
83  					LOGGER.info("ROBOT: LiveDocumentNotAvailableException("
84  							+ robotUrl + ")");
85  				}
86  				// consider it an allow:
87  				aggregation.addDirectives(robotUrl, ALLOW_ROBOT_DIRECTIVE);
88  			}
89  			long elapsed = System.currentTimeMillis() - start;
90  			PerformanceLogger.noteElapsed("RobotRequest", elapsed, robotUrl);
91  		}
92  	}

94  	/* (non-Javadoc)
95  	 * @see org.archive.wayback.resourceindex.SearchResultFilter#filterSearchResult(org.archive.wayback.core.SearchResult)
96  	 */
97  	public int filterObject(CaptureSearchResult r) {
98  		if(!notifiedSeen) {
99  			if(filterGroup != null) {
100 				filterGroup.setSawRobots();
101 			}
102 			notifiedSeen = true;
103 		}
104 		String originalURL = r.getOriginalUrl();
105 		String path = UrlOperations.getURLPath(originalURL);
106 		if(path.equals(ROBOT_SUFFIX)) {
107 			if(!notifiedPassed) {
108 				if(filterGroup != null) {
109 					filterGroup.setPassedRobots();
110 				}
111 				notifiedPassed = true;
112 			}
113 			return ObjectFilter.FILTER_INCLUDE;
114 		}
115 		String host = UrlOperations.urlToHost(originalURL);
116 		boolean updated = false;
117 		try {
118 			updateAggregation(host);
119 			if(!aggregation.isBlocked(path)) {
120 				if(LOGGER.isLoggable(Level.INFO)) {
121 					LOGGER.fine("ROBOT: BLOCKED(" + originalURL + ")");
122 				}
123 				if(LOGGER.isLoggable(Level.FINE)) {
124 					LOGGER.finer("ROBOT: ALLOWED(" + originalURL + ")");
125 				}
126 				if(!notifiedPassed) {
127 					if(filterGroup != null) {
128 						filterGroup.setPassedRobots();
129 					}
130 					notifiedPassed = true;
131 				}
132 				return ObjectFilter.FILTER_INCLUDE;
133 			}
134 
135 //		} catch (LiveDocumentNotAvailableException e) {
136 		} catch (LiveWebCacheUnavailableException e) {
137 			LOGGER.severe("ROBOT: LiveWebCacheUnavailableException("
138 					+ originalURL + ")");
139 			filterGroup.setLiveWebGone();
140 
141 		} catch (LiveWebTimeoutException e) {
142 			LOGGER.severe("ROBOT: LiveDocumentTimedOutException("
143 					+ originalURL + ")");
144 			filterGroup.setRobotTimedOut();
145 
146 		} catch (MalformedURLException e) {
147 
148 			LOGGER.warning("ROBOT: MalformedURLException(" + 
149 					originalURL + ")");
150 
151 		} catch (IOException e) {
152 			e.printStackTrace();
153 			return ObjectFilter.FILTER_EXCLUDE;
154 		}
155 
156 		if(filterGroup.getRobotTimedOut() || filterGroup.getLiveWebGone()) {
157 			return ObjectFilter.FILTER_ABORT;
158 		}
159 		if(LOGGER.isLoggable(Level.INFO)) {
160 			LOGGER.fine("ROBOT: BLOCKED(" + originalURL + ")");
161 		}
162 		return ObjectFilter.FILTER_EXCLUDE;
163 	}

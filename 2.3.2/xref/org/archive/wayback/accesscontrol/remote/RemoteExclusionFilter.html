20  package org.archive.wayback.accesscontrol.remote;
21  
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.io.UnsupportedEncodingException;
25  import java.net.MalformedURLException;
26  import java.net.URL;
27  import java.net.URLEncoder;
28  
29  import java.util.logging.Logger;
30  import org.archive.wayback.core.CaptureSearchResult;
31  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
32  
33  /**
34   * SearchResultFilter which uses remote access control/exclusion service to
35   * filter results.
36   *
37   * @deprecated superseded by ExclusionOracle
38   * @author brad
39   * @version $Date$, $Revision$
40   */
41  public class RemoteExclusionFilter extends ExclusionFilter {
42  	private static final Logger LOGGER = Logger.getLogger(RemoteExclusionFilter.class
43  			.getName());
44  
45  
46  	private final static int BYTE_BUFFER_SIZE = 4096;
47  	
48  	private static String OPERATION_ARGUMENT = "operation";
49  	private static String CHECK_OPERATION = "check";
50  	
51  	private static String URL_ARGUMENT = "url";
52  	private static String USER_AGENT_ARGUMENT = "useragent";
53  	private static String TIMESTAMP_ARGUMENT = "timestamp"; 
54  	private static String OK_CONTENT = "OK";
55  	
56  	private String urlPrefix = null;
57  	private String userAgent = null;
58  
59  	// allocate byte buffer once
60  	private byte[] bbuffer = new byte[BYTE_BUFFER_SIZE];
61  	
62  	/**
63  	 * Constructor
64  	 * 
65  	 * @param urlPrefix String prefix for remote exclusion service
66  	 * @param userAgent String user agent to send to remote exclusion service
67  	 */
68  	public RemoteExclusionFilter(final String urlPrefix, final String userAgent) {
69  		this.urlPrefix = urlPrefix;
70  		this.userAgent = userAgent;
71  	}
72  	
73  	private boolean isBlocked(final String urlString, final String captureDate) {
74  		boolean blocked = true;
75  		
76  		StringBuffer finalUrl = new StringBuffer(urlPrefix);
77  		finalUrl.append("?");
78  
79  		finalUrl.append(OPERATION_ARGUMENT);
80  		finalUrl.append("=");
81  		finalUrl.append(CHECK_OPERATION);
82  
83  
84  		finalUrl.append("&");
85  		finalUrl.append(URL_ARGUMENT);
86  		finalUrl.append("=");
87  		try {
88  			finalUrl.append(URLEncoder.encode(urlString,"UTF-8"));
89  		} catch (UnsupportedEncodingException e) {
90  			// TODO what happenned here?
91  			e.printStackTrace();
92  		}
93  
94  		finalUrl.append("&");
95  		finalUrl.append(USER_AGENT_ARGUMENT);
96  		finalUrl.append("=");
97  		try {
98  			finalUrl.append(URLEncoder.encode(userAgent,"UTF-8"));
99  		} catch (UnsupportedEncodingException e) {
100 			// TODO what happenned here?
101 			e.printStackTrace();
102 		}
103 
104 		finalUrl.append("&");
105 		finalUrl.append(TIMESTAMP_ARGUMENT);
106 		finalUrl.append("=");
107 		try {
108 			finalUrl.append(URLEncoder.encode(captureDate,"UTF-8"));
109 		} catch (UnsupportedEncodingException e) {
110 			// TODO what happenned here?
111 			e.printStackTrace();
112 		}
113 		
114 		try {
115 
116 			// TODO: this is so ugly -- there must be a better/easy way.
117 			URL url = new URL(finalUrl.toString());
118 			InputStream is = url.openStream();
119 			// slurp the whole thing into RAM:
120 			// TODO: character encoding detection & handling!!
121 			StringBuilder sbuffer = new StringBuilder(BYTE_BUFFER_SIZE);
122 			for (int r = -1; (r = is.read(bbuffer, 0, BYTE_BUFFER_SIZE)) != -1;) {
123 				sbuffer.append(new String(bbuffer, 0, r));
124 			}
125 			String content = sbuffer.toString();
126 			if(content.equals(OK_CONTENT)) {
127 				blocked = false;
128 			} else {
129 				LOGGER.info("Disallowing " + urlString + " because " + content);
130 			}
131 		} catch (MalformedURLException e) {
132 			// TODO Auto-generated catch block
133 			e.printStackTrace();
134 		} catch (IOException e) {
135 			// TODO Auto-generated catch block
136 			e.printStackTrace();
137 		}
138 		return blocked;
139 	}
140 	
141 	/* (non-Javadoc)
142 	 * @see org.archive.wayback.resourceindex.SearchResultFilter#filterSearchResult(org.archive.wayback.core.SearchResult)
143 	 */
144 	public int filterObject(CaptureSearchResult r) {
145 		String captureDate = r.getCaptureTimestamp();
146 		String url = r.getOriginalUrl();
147 		return isBlocked(url,captureDate) ?	FILTER_EXCLUDE : FILTER_INCLUDE;
148 	}
149 }

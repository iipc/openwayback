20  package org.archive.wayback.accesscontrol.staticmap;
21  
22  import java.util.Map;
23  import java.util.logging.Logger;
24  
25  import org.apache.commons.httpclient.URIException;
26  import org.archive.wayback.UrlCanonicalizer;
27  import org.archive.wayback.core.CaptureSearchResult;
28  import org.archive.wayback.resourceindex.filterfactory.ExclusionCaptureFilterGroup;
29  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
30  import org.archive.wayback.surt.SURTTokenizer;
31  import org.archive.wayback.util.ObjectFilter;
32  
33  /**
34   *
35   *
36   * @author brad
37   * @version $Date$, $Revision$
38   */
39  public class StaticMapExclusionFilter extends ExclusionFilter {
40  	private static final Logger LOGGER = Logger.getLogger(
41  			StaticMapExclusionFilter.class.getName());
42  
43  	private String lastChecked = null;
44  	private boolean lastCheckedExcluded = false;
45  	private boolean notifiedSeen = false;
46  	private boolean notifiedPassed = false;
47  	Map<String,Object> exclusionMap = null;
48  	UrlCanonicalizer canonicalizer = null;
49  	/**
50  	 * @param map where each String key is a SURT that is blocked.
51  	 */
52  	public StaticMapExclusionFilter(Map<String,Object> map, UrlCanonicalizer canonicalizer) {
53  		exclusionMap = map;
54  		this.canonicalizer = canonicalizer;
55  	}
56  	
57  	
58  	// Set the canonicalizer from the filter, as it may be different from the default
59  	@Override
60  	public void setFilterGroup(ExclusionCaptureFilterGroup filterGroup) {
61  		super.setFilterGroup(filterGroup);
62  		if ((filterGroup != null) && (filterGroup.getCaptureFilterGroupCanonicalizer() != null)) {
63  			this.canonicalizer = filterGroup.getCaptureFilterGroupCanonicalizer();
64  		}
65  	}
66  	
67  	protected boolean isExcluded(String url) {
68  		try {
69  			SURTTokenizer st = new SURTTokenizer(url, canonicalizer.isSurtForm());
70  			while(true) {
71  				String nextSearch = st.nextSearch();
72  				if(nextSearch == null) {
73  					break;
74  				}
75  				LOGGER.fine("EXCLUSION-MAP:Checking " + nextSearch);
76  				if(exclusionMap.containsKey(nextSearch)) {
77  					LOGGER.info("EXCLUSION-MAP: EXCLUDED: \"" + nextSearch + "\" (" + url +")");
78  					return true;
79  				}
80  			}
81  		} catch (URIException e) {
82  			LOGGER.warning(e.toString());
83  			return true;
84  		}
85  		return false;
86  	}
87  	/* (non-Javadoc)
88  	 * @see org.archive.wayback.resourceindex.SearchResultFilter#filterSearchResult(org.archive.wayback.core.SearchResult)
89  	 */
90  	public int filterObject(CaptureSearchResult r) {
91  		if(!notifiedSeen) { 
92  			if(filterGroup != null) {
93  				filterGroup.setSawAdministrative();
94  			}
95  			notifiedSeen = true;
96  		}
97  		String url = r.getUrlKey();
98  		if(lastChecked != null) {
99  			if(lastChecked.equals(url)) {
100 				if(lastCheckedExcluded) { 
101 					return ObjectFilter.FILTER_EXCLUDE;
102 				} else {
103 					// don't need to: already did last time...
104 					//filterGroup.setPassedAdministrative();
105 					return ObjectFilter.FILTER_INCLUDE;
106 				}
107 			}
108 		}
109 		lastChecked = url;
110 		lastCheckedExcluded = isExcluded(url);
111 		if(lastCheckedExcluded) {
112 			return ObjectFilter.FILTER_EXCLUDE;
113 		} else {
114 			if(!notifiedPassed) {
115 				if(filterGroup != null) {
116 					filterGroup.setPassedAdministrative();
117 				}
118 				notifiedPassed = true;
119 			}
120 			return ObjectFilter.FILTER_INCLUDE;
121 		}
122 			
123 	}
124 }

10  import org.archive.format.cdx.FieldSplitFormat;

13  /**
14   * Matches a FieldSplitLine against a string of regex
15   * Supports matching against individual fields if specified
16   * eg:
17   * 
18   * ~<containsstr> = look for containing sting <containsstr> and not a regex
19   * 
20   * <regex> = match whole line
21   * <field>:<regex> = match <field> in FieldSplitLine, by name or number, and match only that field
22   * 
23   * Supports !<regex> for not matching
24   * 
25   * @author ilya
26   *
27   */
28  public class FieldRegexFilter implements CDXFilter {
29  	
30  	final static String INVERT_CHAR = "!";
31  	final static String CONTAINS_CHAR = "~";
32  	final static String FIELD_SEP_CHAR = ":";
33  	
34  	final protected FieldSplitFormat names;
35  	final protected List<RegexMatch> regexMatchers;
36  	
37  	class RegexMatch {
38  		final Pattern regex;
39  		final boolean inverted;
40  		final String containsStr;
41  		final int fieldIndex;
42  		
43  		RegexMatch(String str)
44  		{
45  			boolean contains = false;
46  			
47  			if (str.startsWith(CONTAINS_CHAR)) {
48  				str = str.substring(1);
49  				contains = true;
50  			}
51  			
52  			if (str.startsWith(INVERT_CHAR)) {
53  				str = str.substring(1);
54  				inverted = true;
55  			} else {
56  				inverted = false;
57  			}
58  			
59  			int sepIndex = str.indexOf(FIELD_SEP_CHAR);
60  			
61  			// Match entire line
62  			if (sepIndex < 0) {
63  				fieldIndex = -1;
64  				if (contains) {
65  					containsStr = str;
66  					regex = null;
67  				} else {
68  					containsStr = null;
69  					regex = Pattern.compile(str);
70  				}
71  				return;
72  			}
73  			
74  			String field = str.substring(0, sepIndex);
75  			String pattern = str.substring(sepIndex + 1);
76  			
77  			int index = -1;
78  			
79  			// First try parsing as int
80  			try {
81  				index = Integer.parseInt(field);
82  			} catch (NumberFormatException n) {
83  				
84  			}
85  			
86  			// Then try names if available
87  			if ((index < 0) && (names != null)) {
88  				index = names.getFieldIndex(field);
89  			}
90  			
91  			fieldIndex = index;
92  			
93  			if (contains) {
94  				containsStr = pattern;
95  				regex = null;
96  			} else {
97  				containsStr = null;
98  				regex = Pattern.compile(pattern);
99  			}		
100 		}
101 		
102 		boolean matches(FieldSplitLine line)
103 		{
104 			boolean matched;
105 			
106 			if (fieldIndex < 0) {
107 				if (containsStr != null) {
108 					matched = line.toString().contains(containsStr);
109 				} else {
110 					matched = regex.matcher(line.toString()).matches();	
111 				}
112 			} else {
113 				if (containsStr != null) {
114 					matched = line.getField(fieldIndex).contains(containsStr);
115 				} else {
116 					matched = regex.matcher(line.getField(fieldIndex)).matches();
117 				}
118 			}
119 			
120 			if (inverted) {
121 				matched = !matched;
122 			}
123 			
124 			return matched;
125 		}
126 	}
127 	
128 	
129 	public FieldRegexFilter(String[] regexs, FieldSplitFormat names)
130 	{
131 	  this.names = names;
132 		this.regexMatchers = new ArrayList<RegexMatch>(regexs.length);
133 		
134 		for (String regex : regexs) {
135 		    if (!regex.isEmpty()) {
136 		        regexMatchers.add(new RegexMatch(regex));
137 		    }
138 		}
139 	}
140 	
141 	public boolean include(CDXLine line)
142 	{
143 		for (RegexMatch regexMatch : regexMatchers) 
144 		{
145 			if (!regexMatch.matches(line)) {
146 				return false;
147 			}
148 		}
149 		
150 		return true;
151 	}
152 }

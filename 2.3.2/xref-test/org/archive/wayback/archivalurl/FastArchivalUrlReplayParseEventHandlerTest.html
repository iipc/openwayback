258      * Now all attribute values are unescaped before processing, and then escaped back
259      * before writing out.  This has a side-effect: bare "{@code &}" gets rewritten to

265 		// note "&amp;amp" - it should appear in translated URL as it does in the original.

269 				+ "tracks/135768597%3Ftoken%3Dsss&amp;amp;auto_play=false&bare=1\"></iframe>"
270 				+ "</body>"
271 				+ "</html>";
272 		final String expected = "<html>"
273 				+ "<body>"
274 				+ "<iframe src=\"http://replay.archive.org/2001if_/https://example.com/player/?url=https%3A//api.example.com/"
275 				+ "tracks/135768597%3Ftoken%3Dsss&amp;amp;auto_play=false&amp;bare=1\"></iframe>"
276 				+ "</body>"
277 				+ "</html>";
278 		assertEquals(expected, doEndToEnd(input));

1083     public String doEndToEnd(String input) throws Exception {
1084 		ByteArrayInputStream bais = new ByteArrayInputStream(input.getBytes(charSet));
1085 		
1086 		// To make sure we get the length, we have to buffer it all up...
1087 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
1088 		context.setOutputStream(baos);
1089 		context.setJspExec(jspExec);
1090 
1091 		// and finally, parse, using the special lexer that knows how to
1092 		// handle javascript blocks containing unescaped HTML entities:
1093 		Page lexPage = new Page(bais,charSet);
1094 		Lexer lexer = new Lexer(lexPage);
1095 		Lexer.STRICT_REMARKS = false;
1096 		ContextAwareLexer lex = new ContextAwareLexer(lexer, context);
1097 
1098 		Node node;
1099 		while ((node = lex.nextNode()) != null) {
1100 			delegator.handleNode(context, node);
1101 		}
1102 		delegator.handleParseComplete(context);
1103 
1104 		// At this point, baos contains the utf-8 encoded bytes of our result:
1105 		return new String(baos.toByteArray(), outputCharset);
1106 	}

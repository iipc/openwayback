14  /**
15   * Test of {@link CustomPolicyOracleFilter}.
16   * <p>
17   * Has provision of testing error notification through
18   * {@link ExclusionCaptureFilterGroup}, but test is not implemented, as it looks
19   * like unused in current deployment.
20   * </p>
21   */
22  public class CustomPolicyOracleFilterTest extends TestCase {
23  
24  	CustomPolicyOracleFilter cut;
25  	ExclusionCaptureFilterGroup filterGroup = null;
26  	TestAccessControlClient acClient;
27  
28  	class TestAccessControlClient extends AccessControlClient {
29  		String policyToReturn = "allow";
30  		public TestAccessControlClient() {
31  			super("");
32  		}
33  		@Override
34  		public String getPolicy(String url, Date captureDate,
35  				Date retrievalDate, String who)
36  						throws RobotsUnavailableException,
37  						RuleOracleUnavailableException {
38  			return policyToReturn;
39  		}
40  	}
41  
42  	/* (non-Javadoc)
43  	 * @see junit.framework.TestCase#setUp()
44  	 */
45  	protected void setUp() throws Exception {
46  		cut = new CustomPolicyOracleFilter("", "group", null);
47  		cut.client = (acClient = new TestAccessControlClient());
48  	}
49  
50  	public void testAllow() {
51  		// object properties are not really used except for originalUrl.
52  		CaptureSearchResult capture = new FastCaptureSearchResult();
53  		capture.setOriginalUrl("http://www.example.com/");
54  
55  		int rv = cut.filterObject(capture);
56  
57  		assertEquals(CustomPolicyOracleFilter.FILTER_INCLUDE, rv);
58  	}
59  
60  	public void testBlock() {
61  		acClient.policyToReturn = "block";
62  
63  		// object properties are not really used except for originalUrl.
64  		CaptureSearchResult capture = new FastCaptureSearchResult();
65  		capture.setOriginalUrl("http://www.example.com/");
66  
67  		int rv = cut.filterObject(capture);
68  
69  		// Now "block" returns FILTER_INCLUDE, "X" flag in robotflags.
70  		assertEquals(CustomPolicyOracleFilter.FILTER_INCLUDE, rv);
71  		assertEquals(
72  			Character.toString(CaptureSearchResult.CAPTURE_ROBOT_BLOCKED),
73  			capture.getRobotFlags());
74  	}
75  
76  	public void testBlockMessage() {
77  		acClient.policyToReturn = "block-message";
78  
79  		// object properties are not really used except for originalUrl.
80  		CaptureSearchResult capture = new FastCaptureSearchResult();
81  		capture.setOriginalUrl("http://www.example.com/");
82  
83  		int rv = cut.filterObject(capture);
84  
85  		// Now "block" returns FILTER_INCLUDE, "X" flag in robotflags.
86  		assertEquals(CustomPolicyOracleFilter.FILTER_EXCLUDE, rv);
87  	}
88  
89  	public void testRobots() {
90  		// AccessControlClient translates "robots" policy into either
91  		// "allow" or "block" when robotLookupsEnable is true - which
92  		// are tested above. "robots" policy is considered as "allow".
93  		acClient.policyToReturn = "robots";
94  
95  		// object properties are not really used except for originalUrl.
96  		CaptureSearchResult capture = new FastCaptureSearchResult();
97  		capture.setOriginalUrl("http://www.example.com/");
98  
99  		int rv = cut.filterObject(capture);
100 
101 		assertEquals(CustomPolicyOracleFilter.FILTER_INCLUDE, rv);
102 	}
103 }

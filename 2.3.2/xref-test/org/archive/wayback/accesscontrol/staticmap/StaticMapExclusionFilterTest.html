141 		String bases[] = {"http://www.peagreenboat.com/",
142 							"http://peagreenboat.com/"};
143 //		setTmpContents(bases);
144 		ObjectFilter<CaptureSearchResult> filter = getFilter(bases);
145 		assertTrue("unmassaged",isBlocked(filter,"http://www.peagreenboat.com"));
146 		assertTrue("unmassaged",isBlocked(filter,"http://peagreenboat.com"));
147 		assertFalse("other1",isBlocked(filter,"http://peagreenboatt.com"));
148 		assertFalse("other2",isBlocked(filter,"http://peagreenboat.org"));
149 		assertFalse("other3",isBlocked(filter,"http://www.peagreenboat.org"));
150 		// there is a problem with the SURTTokenizer... deal with ports!
151 //		assertFalse("other4",isBlocked(filter,"http://www.peagreenboat.com:8080"));
152 		assertTrue("subpath",isBlocked(filter,"http://www.peagreenboat.com/foo"));
153 		assertTrue("emptypath",isBlocked(filter,"http://www.peagreenboat.com/"));

156 	private boolean isBlocked(ObjectFilter<CaptureSearchResult> filter, String url) throws URIException {
157 		CaptureSearchResult result = new CaptureSearchResult();
158 		result.setOriginalUrl(url);
159 		result.setUrlKey(canonicalizer.urlStringToKey(url));
160 		int filterResult = filter.filterObject(result);
161 		if(filterResult == ObjectFilter.FILTER_EXCLUDE) {
162 			return true;
163 		}
164 		return false;
165 	}
166 	
167 	private ObjectFilter<CaptureSearchResult> getFilter(String lines[]) 
168 		throws IOException {
169 		
170 		setTmpContents(lines);
171 		Map<String,Object> map = factory.loadFile(tmpFile.getAbsolutePath());
172 		return new StaticMapExclusionFilter(map,canonicalizer);
173 	}
174 
175 	private void setTmpContents(String[] lines) throws IOException {
176 		if(tmpFile != null && tmpFile.exists()) {
177 			tmpFile.delete();
178 		}
179 //		tmpFile = File.createTempFile("range-map","tmp");
180 		FileWriter writer = new FileWriter(tmpFile);
181 		StringBuilder sb = new StringBuilder();
182 		for(int i=0; i<lines.length; i++) {
183 			sb.append(lines[i]).append("\n");
184 		}
185 		String contents = sb.toString();
186 		writer.write(contents);
187 		writer.close();
188 		//factory.reloadFile();
189 	}

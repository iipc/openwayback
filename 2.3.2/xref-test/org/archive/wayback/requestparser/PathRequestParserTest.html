46              new ReplayRequestParser(brp),

53      private ArchivalUrl parseAU(String path) 
54      throws BetterRequestException, BadQueryException {
55          WaybackRequest wbRequest = null;
56          for(int i = 0; i < parsers.length; i++) {
57              wbRequest = parsers[i].parse(path, null);
58              if (wbRequest != null) {
59                  break;
60              }
61          }
62          return new ArchivalUrl(wbRequest);
63      }
64      
65      private void trt(String want, String src) {
66          try {
67              assertEquals(want,parseAU(src).toString());
68          } catch (BetterRequestException e) {
69              e.printStackTrace();
70              fail(e.getMessage());
71          } catch (BadQueryException e) {
72              e.printStackTrace();
73              fail(e.getMessage());
74          }
75      }

110     public void testReplayRequestParser() throws Exception {
111         ReplayRequestParser parser = new ReplayRequestParser(brp);
112         // note AccessPoint arg is unused
113         {
114             WaybackRequest wbr = parser.parse("20100101000000/http://www.yahoo.com/", null);
115             assertEquals("replayTimestamp", "20100101000000", wbr.getReplayTimestamp());
116             assertNull("startTimestamp", wbr.getStartTimestamp());
117             // endTimestamp is current timestamp
118             assertNotNull("endTimestamp", wbr.getEndTimestamp());
119         }
120         // old Alexa ARCs has 12-digits dates - padded with "00" -> same as prev case
121         {
122             WaybackRequest wbr = parser.parse("201001010000/http://www.yahoo.com/", null);
123             assertEquals("replayTimestamp", "20100101000000", wbr.getReplayTimestamp());
124             assertNull("startTimestamp", wbr.getStartTimestamp());
125             // endTimestamp is current timestamp
126             assertNotNull("endTimestamp", wbr.getEndTimestamp());
127         }
128         {
129             WaybackRequest wbr = parser.parse("20100101/http://www.yahoo.com/", null);
130             assertEquals("replayTimestamp", "20100101235959", wbr.getReplayTimestamp());
131             assertEquals("startTimestamp", "20100101000000", wbr.getStartTimestamp());
132             assertEquals("endTimestamp", "20100101235959", wbr.getEndTimestamp());
133         }

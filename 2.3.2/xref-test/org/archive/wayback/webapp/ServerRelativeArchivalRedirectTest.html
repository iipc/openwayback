43  		EasyMock.expect(request.getRequestURI()).andStubReturn(requestURI);
44  		EasyMock.expect(request.getContextPath()).andStubReturn("");

48  	 * test clean case with well-formed target URL in {@code Referer}.
49  	 */
50  	public void testRegular() throws Exception {
51  
52  		final String TARGET_URL = "http://example.com/index.html";
53  		final String REPLAY_BASE = "http://web.archive.org/web/20010203040506/";
54  		final String REFERER = REPLAY_BASE + TARGET_URL;
55  		final String REQUEST_URI = "/js/s_code.js";
56  
57  		final String EXPECTED_REDIRECT_URL = REPLAY_BASE +
58  				"http://example.com/js/s_code.js";
59  
60  		// mocks
61  		setUpRequest(REQUEST_URI, REFERER);
62  
63  		HttpServletResponse response = EasyMock
64  			.createMock(HttpServletResponse.class);
65  		response.addHeader("Vary", "Referer");
66  		EasyMock.expectLastCall().once();
67  		response.sendRedirect(EXPECTED_REDIRECT_URL);
68  		EasyMock.expectLastCall().once();
69  
70  		EasyMock.replay(request, response);
71  
72  		boolean handled = cut.handleRequest(request, response);
73  
74  		EasyMock.verify(response);
75  		assertTrue(handled);
76  	}

112 	public static class Extended extends ServerRelativeArchivalRedirect {
113 		ArchivalUrlRef ref;
114 		public Extended(ArchivalUrlRef ref) {
115 			this.ref = ref;
116 		}
117 		protected ArchivalUrlRef getOrigin(HttpServletRequest httpRequest) {
118 			ArchivalUrlRef ref = super.getOrigin(httpRequest);
119 			if (ref == null) {
120 				// imaging getting this value from somewhere...
121 				ref = this.ref;
122 			}
123 			return ref;
124 		};
125 	}

139 				DATESPEC + "/" + TARGET_BASE + REQUEST_URI;

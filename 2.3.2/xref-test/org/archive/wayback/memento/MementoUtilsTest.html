98  	final static Pattern LINK_ELEMENT_PATTERN = Pattern.compile("(<.*?>(?:;\\s*[a-z]+=\"(.*?)\")+)(,\\s*)?");
99  
100 	public void testGenerateMementoLinkHeaders() {
101 		final CaptureSearchResults results = new CaptureSearchResults();
102 		results.addSearchResult(createCapture("20140101000000"));
103 		results.addSearchResult(createCapture("20140125000000"));
104 		results.setClosest(results.getResults().getLast());
105 
106 		final WaybackRequest wbr = new WaybackRequest();
107 		wbr.setAccessPoint(accessPoint);
108 		wbr.setRequestUrl("http://example.com/");
109 
110 		String result = MementoUtils.generateMementoLinkHeaders(results, wbr, true, true);
111 
112 		List<String> elements = new ArrayList<String>();
113 		Matcher m = LINK_ELEMENT_PATTERN.matcher(result);
114 		while (m.regionStart() < result.length() && m.lookingAt()) {
115 			elements.add(m.group(1));
116 			m.region(m.end(), result.length());
117 		}
118 		assertEquals(result.length(), m.regionStart());
119 
120 		// TODO: order doesn't really matter
121 		assertEquals("<http://example.com/>; rel=\"original\"",
122 			elements.get(0));
123 		assertEquals("</web/timemap/link/http://example.com/>; rel=\"timemap\"; type=\"application/link-format\"",
124 			elements.get(1));
125 		assertEquals("</web/http://example.com/>; rel=\"timegate\"",
126 			elements.get(2));
127 		assertEquals("</web/20140125000000/http://example.com/>; rel=\"last memento\"; datetime=\"Sat, 25 Jan 2014 00:00:00 GMT\"",
128 			elements.get(3));
129 		assertEquals("</web/20140101000000/http://example.com/>; rel=\"prev first memento\"; datetime=\"Wed, 01 Jan 2014 00:00:00 GMT\"",
130 			elements.get(4));
131 	}

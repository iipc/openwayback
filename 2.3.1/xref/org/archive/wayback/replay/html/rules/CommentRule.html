20  package org.archive.wayback.replay.html.rules;
21  
22  import java.io.IOException;
23  import java.io.OutputStream;
24  
25  import org.archive.wayback.replay.html.ReplayParseEventDelegator;
26  import org.archive.wayback.replay.html.ReplayParseEventDelegatorVisitor;
27  import org.archive.wayback.replay.html.ReplayParseContext;
28  import org.archive.wayback.util.htmllex.ParseContext;
29  import org.archive.wayback.util.htmllex.handlers.CloseTagHandler;
30  import org.archive.wayback.util.htmllex.handlers.OpenTagHandler;
31  import org.htmlparser.Node;
32  import org.htmlparser.nodes.TagNode;
33  
34  public class CommentRule implements ReplayParseEventDelegatorVisitor, 
35  	OpenTagHandler, CloseTagHandler {
36  
37  	private final static byte[] startComment = "<!--".getBytes();
38  	private final static byte[] endComment = "-->".getBytes();
39  	
40  	public void emit(ReplayParseContext context, Node node) throws IOException {
41  		OutputStream os = context.getOutputStream();
42  		if(os != null) {
43  			os.write(startComment);
44  			os.write(node.toHtml(true).getBytes());
45  			os.write(endComment);
46  		}
47  	}
48  
49  	public void visit(ReplayParseEventDelegator rules) {
50  		rules.getPreModifyDelegator().addOpenTagHandler(this);
51  		rules.getPreModifyDelegator().addCloseTagHandler(this, "A");
52  	}
53  
54  	public void handleOpenTagNode(ParseContext context, TagNode node) throws IOException {
55  		emit((ReplayParseContext)context,node);
56  	}
57  
58  	public void handleCloseTagNode(ParseContext context, TagNode node)
59  			throws IOException {
60  		emit((ReplayParseContext)context,node);
61  	}
62  }

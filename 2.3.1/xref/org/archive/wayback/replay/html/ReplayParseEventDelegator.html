20  package org.archive.wayback.replay.html;
21  
22  import java.io.IOException;
23  import java.io.OutputStream;
24  import java.io.UnsupportedEncodingException;
25  import java.util.List;
26  
27  import org.archive.wayback.util.htmllex.ParseEventHandler;
28  import org.archive.wayback.util.htmllex.ParseEventDelegator;
29  import org.archive.wayback.util.htmllex.ParseContext;
30  import org.htmlparser.Node;
31  
32  public class ReplayParseEventDelegator implements ParseEventHandler {
33  	
34  	public static final int PHASE_PRE_MODIFY = 0;
35  	public static final int PHASE_MODIFY = 1;
36  	public static final int PHASE_POST_OUTPUT = 2;
37  
38  	private ParseEventDelegator preModifyDelegator = null;
39  	private ParseEventDelegator modifyDelegator = null;
40  	private ParseEventDelegator postModifyDelegator = null;
41  	private List<ReplayParseEventDelegatorVisitor> parserVisitors = null;
42  	
43  	protected void emit(ParseContext context, Node node) throws IOException {
44  		ReplayParseContext rContext = (ReplayParseContext) context;
45  		OutputStream out = rContext.getOutputStream();
46  		// no-op, override to actually output something:
47  		if(out != null) {
48  			String charset = rContext.getOutputCharset();
49  			String rawHTML = node.toHtml(true);
50  			byte[] bytes = null;
51  			try {
52  				bytes = rawHTML.getBytes(charset);
53  			} catch (UnsupportedEncodingException e) {
54  				bytes = rawHTML.getBytes();
55  			}
56  			out.write(bytes);
57  		}
58  	}
59  
60  
61  	public void init() {
62  		preModifyDelegator = new ParseEventDelegator();
63  		modifyDelegator = new ParseEventDelegator();
64  		postModifyDelegator = new ParseEventDelegator();
65  		if(parserVisitors != null) {
66  			for(ReplayParseEventDelegatorVisitor visitor : parserVisitors) {
67  				visitor.visit(this);
68  			}
69  		}
70  	}
71  
72  	
73  	public void handleNode(ParseContext pContext, Node node) 
74  		throws IOException {
75  		ReplayParseContext context = (ReplayParseContext) pContext;
76  		context.setPhase(PHASE_PRE_MODIFY);
77  		preModifyDelegator.handleNode(context,node);
78  		context.setPhase(PHASE_MODIFY);
79  		modifyDelegator.handleNode(context,node);
80  		emit(context, node);
81  		context.setPhase(PHASE_POST_OUTPUT);
82  		postModifyDelegator.handleNode(context,node);
83  
84  
85  	}
86  
87  	public void handleParseStart(ParseContext context) throws IOException {
88  		preModifyDelegator.handleParseStart(context);
89  		modifyDelegator.handleParseStart(context);
90  		postModifyDelegator.handleParseStart(context);
91  	}
92  
93  	public void handleParseComplete(ParseContext context) throws IOException {
94  		preModifyDelegator.handleParseComplete(context);
95  		modifyDelegator.handleParseComplete(context);
96  		postModifyDelegator.handleParseComplete(context);
97  	}
98  	/**
99  	 * @return the preModifyDelegator
100 	 */
101 	public ParseEventDelegator getPreModifyDelegator() {
102 		return preModifyDelegator;
103 	}
104 
105 
106 	/**
107 	 * @param preModifyDelegator the preModifyDelegator to set
108 	 */
109 	public void setPreModifyDelegator(ParseEventDelegator preModifyDelegator) {
110 		this.preModifyDelegator = preModifyDelegator;
111 	}
112 
113 
114 	/**
115 	 * @return the modifyDelegator
116 	 */
117 	public ParseEventDelegator getModifyDelegator() {
118 		return modifyDelegator;
119 	}
120 
121 
122 	/**
123 	 * @param modifyDelegator the modifyDelegator to set
124 	 */
125 	public void setModifyDelegator(ParseEventDelegator modifyDelegator) {
126 		this.modifyDelegator = modifyDelegator;
127 	}
128 
129 
130 	/**
131 	 * @return the postModifyDelegator
132 	 */
133 	public ParseEventDelegator getPostModifyDelegator() {
134 		return postModifyDelegator;
135 	}
136 
137 
138 	/**
139 	 * @param postModifyDelegator the postModifyDelegator to set
140 	 */
141 	public void setPostModifyDelegator(ParseEventDelegator postModifyDelegator) {
142 		this.postModifyDelegator = postModifyDelegator;
143 	}
144 
145 
146 	/**
147 	 * @param parserVisitors the parserVisitors to set
148 	 */
149 	public void setParserVisitors(List<ReplayParseEventDelegatorVisitor> parserVisitors) {
150 		this.parserVisitors = parserVisitors;
151 	}
152 }

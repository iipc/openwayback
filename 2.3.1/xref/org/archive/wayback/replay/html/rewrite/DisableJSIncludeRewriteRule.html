1   package org.archive.wayback.replay.html.rewrite;

3   import org.archive.wayback.archivalurl.FastArchivalUrlReplayParseEventHandler;
4   import org.archive.wayback.replay.html.ReplayParseContext;
5   
6   /**
7    * Special {@link RewriteRule} for disabling script inclusion
8    * ({@code <SCRIPT SRC="..."></SCRIPT>}).
9    * Policy name shall have script filename, followed by "{@code =}".
10   * If {@code SRC} attribute value contains the filename,
11   * {@code SCRIPT} tag will be disabled.
12   * <p>
13   * {@link FastArchivalUrlReplayParseEventHandler} has special
14   * handling of {@code SCRIPT} tag. It passes the value of
15   * {@code SRC} attribute as {@code inputSrc}. When this rule
16   * returns {@code null} or empty string, {@code SCRIPT} tag
17   * is disabled by replacing {@code SRC} attribute value with
18   * empty text.
19   * </p>
20   * @see FastArchivalUrlReplayParseEventHandler
21   * @see DisableJSBlocksRewriteRule
22   * @see RewritingStringTransformer
23   */
24  public class DisableJSIncludeRewriteRule extends RewriteRule {
25  
26  	@Override
27  	public String rewrite(ReplayParseContext context, String policy,
28  			String inputSrc) {
29  
30  		int index = policy.indexOf('=', getName().length());
31  
32  		if (index >= 0) {
33  			String filename = policy.substring(index + 1);
34  
35  			// If inputSrc contains filename, exclude it!
36  			if ((inputSrc != null) && inputSrc.indexOf(filename) >= 0) {
37  				return null;
38  			}
39  		}
40  
41  		return inputSrc;
42  	}
43  }

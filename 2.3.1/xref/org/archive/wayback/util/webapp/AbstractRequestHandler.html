24  import javax.servlet.ServletContext;

30  /**
31   * Abstract RequestHandler implementation which performs the minimal behavior
32   * for self registration with a RequestMapper, requiring subclasses to implement
33   * only handleRequest(). 
34   * 
35   * @author brad
36   *
37   */
38  public abstract class AbstractRequestHandler implements RequestHandler {
39  	private String beanName = null;
40  	private String accessPointPath = null;
41  	private ServletContext servletContext = null;
42  	private int internalPort = 0;
43  
44  	public void setBeanName(final String beanName) {
45  		this.beanName = beanName;
46  	}
47  
48  	public String getBeanName() {
49  		return beanName;
50  	}
51  
52  	public int getInternalPort() {
53  		return internalPort;
54  	}
55  
56  	public void setInternalPort(int internalPort) {
57  		this.internalPort = internalPort;
58  	}
59  
60  	public String getAccessPointPath() {
61  		return accessPointPath;
62  	}
63  
64  	public void setAccessPointPath(String accessPointPath) {
65  		this.accessPointPath = accessPointPath;
66  	}
67  
68  	public void setServletContext(ServletContext servletContext) {
69  		this.servletContext = servletContext;
70  	}
71  
72  	public ServletContext getServletContext() {
73  		return servletContext;
74  	}
75  
76  	// Refactor: this method is only called by RequestMapper, and eventually
77  	// calls RequestMapper.addRequestHandler() through static method BeanNameRegistrar.registerHandler().
78  	// AbstractRequestHandler does not play any active role there.  Move this code to RequestMapper.
79  	/**
80  	 * @deprecated 2014-04-24 call {@link BeanNameRegistrar#registerHandler(RequestHandler, RequestMapper)} directly.
81  	 */
82  	public void registerPortListener(RequestMapper requestMapper) {
83  		BeanNameRegistrar.registerHandler(this, requestMapper);
84  	}
85  
86  	/**
87  	 * @deprecated 2014-04-23 use {@link RequestMapper#getRequestContextPath(HttpServletRequest)} directly.
88  	 */
89  	public String translateRequestPath(HttpServletRequest httpRequest) {
90  		return RequestMapper.getRequestContextPath(httpRequest);
91  	}
92  
93  	/**
94  	 * @deprecated 2014-04-23 use {@link RequestMapper#getRequestContextPathQuery(HttpServletRequest)} directly
95  	 */
96  	public String translateRequestPathQuery(HttpServletRequest httpRequest) {
97  		return RequestMapper.getRequestContextPathQuery(httpRequest);
98  	}
99  
100 	// Refactor: move getMapParam, getRequiredMapParam and getMapParamOrEmpty
101 	// to RequestParser base class.
102 	/**
103 	 * Extract the first value in the array mapped to by field in queryMap
104 	 * @param queryMap the Map in which to search
105 	 * @param field the field value desired
106 	 * @return the first value in the array mapped to field in queryMap, if 
107 	 * present, null otherwise
108 	 */
109 	public static String getMapParam(Map<String,String[]> queryMap,
110 			String field) {
111 		String arr[] = queryMap.get(field);
112 		if (arr == null || arr.length == 0) {
113 			return null;
114 		}
115 		return arr[0];
116 	}
117 
118 	/**
119 	 * Extract the first value in the array mapped to by field in queryMap
120 	 * @param queryMap the Map in which to search
121 	 * @param field the field value desired
122 	 * @return the first value in the array mapped to field in queryMap, if 
123 	 * present. A BadQueryException is thrown if there is no appropriate value
124 	 * @throws BadQueryException if there is nothing mapped to field, or if the
125 	 * Array mapped to field is empty
126 	 */
127 	public static String getRequiredMapParam(Map<String, String[]> queryMap,
128 			String field) throws BadQueryException {
129 		// TODO: Throw something different, org.archive.wayback.util should have
130 		// no references outside of org.archive.wayback.util 
131 		String value = getMapParam(queryMap, field);
132 		if (value == null) {
133 			throw new BadQueryException("missing field " + field);
134 		}
135 		if (value.length() == 0) {
136 			throw new BadQueryException("empty field " + field);
137 		}
138 		return value;
139 	}
140 
141 	/**
142 	 * Extract the first value in the array mapped to by field in queryMap
143 	 * @param map the Map in which to search
144 	 * @param param the field value desired
145 	 * @return the first value in the array mapped to field in queryMap, if 
146 	 * present, or an empty string otherwise
147 	 */
148 	public static String getMapParamOrEmpty(Map<String,String[]> map, 
149 			String param) {
150 		String val = getMapParam(map, param);
151 		return (val == null) ? "" : val;
152 	}
153 }

20  package org.archive.wayback.util.graph;
21  
22  import java.awt.Graphics2D;
23  
24  /**
25   * @author brad
26   *
27   */
28  public class RegionGraphElement extends RectangularGraphElement {
29  	private RegionData data = null;
30  	private ValueGraphElements values = null;
31  	private GraphConfiguration config = null;
32  
33  	/**
34  	 * @param x the x coordinate for the region, in the global coordinate space
35  	 * of the containing Graph
36  	 * @param y the y coordinate for the region, in the global coordinate space
37  	 * of the containing Graph
38  	 * @param width the width of this region, in pixels
39  	 * @param height the height of this region, in pixels
40  	 * @param data the data to use for this region
41  	 * @param config the GraphConfiguration for this Graph
42  	 */
43  	public RegionGraphElement(int x, int y, int width, int height,
44  			RegionData data, GraphConfiguration config) {
45  		super(x,y,width,height);
46  //		System.err.format("Created region (%d,%d)-(%d,%d)\n",x,y,width,height);
47  		this.data = data;
48  		this.config = config;
49  //		int labelHeight = config.regionFontSize + (config.fontPadY * 2);
50  		int labelHeight = 0;
51  		int valuesHeight = height - labelHeight;
52  		this.values = new ValueGraphElements(x+1, y+1, width - 1, valuesHeight,
53  				data.getHighlightedValue(), data.getValues(), 
54  				data.getMaxValue(), config);
55  	}
56  	
57  	/**
58  	 * @return the RegionData for this region
59  	 */
60  	public RegionData getData() {
61  		return data;
62  	}
63  	
64  	public void draw(Graphics2D g2d) {
65  		
66  		if(data.hasHighlightedValue()) {
67  			g2d.setColor(config.regionHighlightColor);
68  			g2d.fillRect(x + 1, y+1, width - 1, height-2);
69  		}
70  		
71  		g2d.setColor(config.regionBorderColor);
72  		g2d.setStroke(config.regionBorderStroke);
73  		g2d.drawLine(x, y, x, y + height);
74  
75  //		int fontY = (y + height) - config.fontPadY;
76  //		
77  //		g2d.setColor(config.regionLabelColor);
78  //		g2d.setFont(config.getRegionLabelFont());
79  //		g2d.drawString(data.getLabel(), x + config.fontPadX, fontY);
80  		values.draw(g2d);
81  	}
82  }

20  package org.archive.wayback.resourcestore.locationdb;
21  
22  import java.io.BufferedReader;
23  import java.io.File;
24  import java.io.FileInputStream;
25  import java.io.FileWriter;
26  import java.io.IOException;
27  import java.io.InputStreamReader;
28  import java.io.RandomAccessFile;
29  
30  import org.archive.util.iterator.CloseableIterator;
31  import org.archive.wayback.util.ByteOp;
32  import org.archive.wayback.util.flatfile.RecordIterator;
33  
34  /**
35   * Simple log file tracking new names being added to a ResourceFileLocationDB.
36   * 
37   * Also supports returning an iterator of Strings to a byte range of the log, to
38   * simplify tracking deltas to the DB.
39   * 
40   * @author brad
41   * @version $Date$, $Revision$
42   */
43  public class ResourceFileLocationDBLog extends File {
44  
45  	/**
46  	 * 
47  	 */
48  	private static final long serialVersionUID = -9128222006544481378L;
49  
50  	/**
51  	 * @param pathname
52  	 * @throws IOException
53  	 */
54  	public ResourceFileLocationDBLog(String pathname) throws IOException {
55  		super(pathname);
56  		if (!isFile()) {
57  			if (exists()) {
58  				throw new IOException("path(" + pathname
59  						+ ") exists but is not a file!");
60  			}
61  			try {
62  				if (!createNewFile()) {
63  					throw new IOException(
64  							"Unable to create empty file " + pathname);
65  				}
66  			} catch (IOException e) {
67  				e.printStackTrace();
68  				throw new IOException("Unable to create empty file "
69  						+ pathname);
70  			}
71  		}
72  	}
73  
74  	/**
75  	 * @return long value indicating the current end position of the log
76  	 */
77  	public long getCurrentMark() {
78  		return length();
79  	}
80  
81  	/**
82  	 * @param start
83  	 * @param end
84  	 * @return CleanableIterator that returns all names between start and end
85  	 * @throws IOException
86  	 */
87  	public CloseableIterator<String> getNamesBetweenMarks(long start, long end)
88  			throws IOException {
89  
90  		RandomAccessFile raf = new RandomAccessFile(this, "r");
91  		raf.seek(start);
92  		FileInputStream fis = new FileInputStream(raf.getFD());
93  		InputStreamReader isr = new InputStreamReader(fis,ByteOp.UTF8);
94  		BufferedReader is = new BufferedReader(isr);
95  		return new BufferedRangeIterator(new RecordIterator(is),end - start);
96  	}
97  
98  	/**
99  	 * @param name
100 	 * @throws IOException
101 	 */
102 	public synchronized void addName(String name) throws IOException {
103 		FileWriter writer = new FileWriter(this, true);
104 		writer.write(name + "\n");
105 		writer.flush();
106 		writer.close();
107 	}
108 
109 	private class BufferedRangeIterator implements CloseableIterator<String> {
110 		private RecordIterator itr;
111 		private long bytesToSend;
112 		private long bytesSent;
113 		private String next;
114 		private boolean done;
115 		/**
116 		 * @param itr
117 		 * @param bytesToSend
118 		 */
119 		public BufferedRangeIterator(RecordIterator itr, long bytesToSend) {
120 			this.itr = itr;
121 			this.bytesToSend = bytesToSend;
122 			bytesSent = 0;
123 			next = null;
124 			done = false;
125 		}
126 		/* (non-Javadoc)
127 		 * @see org.archive.wayback.util.CleanableIterator#clean()
128 		 */
129 		public void close() throws IOException {
130 			if(done == false) {
131 				itr.close();
132 				done = true;
133 			}
134 		}
135 
136 		/* (non-Javadoc)
137 		 * @see java.util.Iterator#hasNext()
138 		 */
139 		public boolean hasNext() {
140 			if(done) return false;
141 			if(next != null) return true;
142 			if((bytesSent >= bytesToSend) || !itr.hasNext()) {
143 				try {
144 					close();
145 				} catch (IOException e) {
146 					// TODO This is lame. What is the right way?
147 					throw new RuntimeException(e);
148 				}
149 				return false;
150 			}
151 			next = (String) itr.next();
152 			return true;
153 		}
154 
155 		/* (non-Javadoc)
156 		 * @see java.util.Iterator#next()
157 		 */
158 		public String next() {
159 			String returnString = next;
160 			next = null;
161 			bytesSent += returnString.length() + 1; // TODO: not X-platform!
162 			return returnString;
163 		}
164 
165 		/* (non-Javadoc)
166 		 * @see java.util.Iterator#remove()
167 		 */
168 		public void remove() {
169 			throw new UnsupportedOperationException();
170 		}
171 	}
172 }

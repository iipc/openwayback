37  /**
38   * @author brad
39   *
40   */
41  public class ZiplinesChunkIterator implements CloseableIterator<String> {
42  	private static final Logger LOGGER = Logger.getLogger(
43  			ZiplinesChunkIterator.class.getName());
44  
45  	private BufferedReader br = null;
46  	private Iterator<ZiplinedBlock> blockItr = null;
47  	private String cachedNext = null;
48  	private boolean truncated = false;
49  	/**
50  	 * @param blocks which should be fetched and unzipped, one after another
51  	 */
52  	public ZiplinesChunkIterator(List<ZiplinedBlock> blocks) {
53  		LOGGER.info("initialized with " + blocks.size() + " blocks");
54  		blockItr = blocks.iterator();
55  	}
56  	/* (non-Javadoc)
57  	 * @see java.util.Iterator#hasNext()
58  	 */
59  	public boolean hasNext() {
60  		if(cachedNext != null) {
61  			return true;
62  		}
63  		while(cachedNext == null) {
64  			if(br != null) {
65  				// attempt to read the next line from this:
66  				try {
67  					cachedNext = br.readLine();
68  					if(cachedNext == null) {
69  						br = null;
70  						// next loop:
71  					} else {
72  						return true;
73  					}
74  				} catch (IOException e) {
75  					e.printStackTrace();
76  					br = null;
77  				}
78  			} else {
79  				// do we have more blocks to use?
80  				if(blockItr.hasNext()) {
81  					try {
82  						br = blockItr.next().readBlock();
83  					} catch (IOException e) {
84  						throw new RuntimeIOException();
85  					}
86  				} else {
87  					return false;
88  				}
89  			}
90  		}
91  		
92  		return false;
93  	}
94  
95  	/* (non-Javadoc)
96  	 * @see java.util.Iterator#next()
97  	 */
98  	public String next() {
99  		String tmp = cachedNext;
100 		cachedNext = null;
101 		return tmp;
102 	}
103 
104 	/* (non-Javadoc)
105 	 * @see java.util.Iterator#remove()
106 	 */
107 	public void remove() {
108 		throw new UnsupportedOperationException();
109 	}
110 
111 	/* (non-Javadoc)
112 	 * @see java.io.Closeable#close()
113 	 */
114 	public void close() throws IOException {
115 		if(br != null) {
116 			br.close();
117 		}
118 	}
119 	public static void main(String[] args) {
120 		if(args.length != 1) {
121 			System.err.println("Usage: ZIPLINES_PATH");
122 			System.exit(1);
123 		}
124 		File f = new File(args[0]);
125 		long size = f.length();
126 		long numBlocks = (long) (size / ZiplinedBlock.BLOCK_SIZE);
127 		long size2 = numBlocks * ZiplinedBlock.BLOCK_SIZE;
128 		if(size != size2) {
129 			System.err.println("File size of " + args[0] + " is not a mulitple"
130 					+ " of " + ZiplinedBlock.BLOCK_SIZE);
131 		}
132 		try {
133 			RandomAccessFile raf = new RandomAccessFile(f, "r");
134 			for(int i = 0; i < numBlocks; i++) {
135 				long offset = i * ZiplinedBlock.BLOCK_SIZE;
136 				raf.seek(offset);
137 //				BufferedReader br = new BufferedReader(new InputStreamReader(
138 //						new GZIPInputStream(new FileInputStream(raf.getFD())),ByteOp.UTF8));
139 				BufferedReader br = new BufferedReader(new InputStreamReader(
140 						new OpenJDK7GZIPInputStream(new FileInputStream(raf.getFD())),ByteOp.UTF8));
141 				String line = br.readLine();
142 				if(line == null) {
143 					System.err.println("Bad block at " + offset + " in " + args[0]);
144 					System.exit(1);
145 				}
146 				System.out.println(args[0] + " " + offset + " " + line);
147 			}
148 		} catch (IOException e) {
149 			e.printStackTrace();
150 			System.exit(1);
151 		}
152 	}
153 	/**
154 	 * @return the truncated
155 	 */
156 	public boolean isTruncated() {
157 		return truncated;
158 	}
159 	/**
160 	 * @param truncated the truncated to set
161 	 */
162 	public void setTruncated(boolean truncated) {
163 		this.truncated = truncated;
164 	}
165 }

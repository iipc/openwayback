20  package org.archive.wayback.archivalurl.requestparser;
21  
22  import java.io.UnsupportedEncodingException;
23  import java.net.URLDecoder;
24  
25  import org.archive.wayback.core.WaybackRequest;
26  import org.archive.wayback.requestparser.BaseRequestParser;
27  import org.archive.wayback.requestparser.PathRequestParser;
28  import org.archive.wayback.webapp.AccessPoint;
29  
30  /**
31   * PathRequestParser that expects <i>TIMESTAMP/URL</i> path pattern.
32   * <p>TIMESTAMP component is %-decoded before passed to 
33   * @author Kenji Nagahashi
34   *
35   */
36  public abstract class DateUrlPathRequestParser extends PathRequestParser {
37  
38  	public DateUrlPathRequestParser(BaseRequestParser wrapped) {
39  		super(wrapped);
40  	}
41  	
42  	private String urldecode(String s) {
43  		try {
44  			return URLDecoder.decode(s, "UTF-8");
45  		} catch (UnsupportedEncodingException ex) {
46  			return s;
47  		}
48  	}
49  
50  	/* (non-Javadoc)
51  	 * @see org.archive.wayback.requestparser.PathRequestParser#parse(java.lang.String, org.archive.wayback.webapp.AccessPoint)
52  	 */
53  	@Override
54  	public WaybackRequest parse(String requestPath, AccessPoint accessPoint) {
55  		int p = requestPath.indexOf('/');
56  		if (p < 1)
57  			return null;
58  		String ts = urldecode(requestPath.substring(0, p));
59  		// TODO: skip multiple "/"?
60  		String url = requestPath.substring(p + 1);
61  		return parseDateUrl(ts, url);
62  	}
63  
64  	/**
65  	 * parse {@code dateStr} and {@code urlStr} and return {@code WaybackRequest} object.
66  	 * @param dateStr the first path component, %-decoded.
67  	 * @param urlStr everything after the first slash, non-decoded.
68  	 * @return {@code WaybackRequest} with parse result, or {@code null} if either
69  	 *   {@code dateStr} or {@code urlStr} does not match expected syntax. 
70  	 */
71  	protected abstract WaybackRequest parseDateUrl(String dateStr, String urlStr);
72  }

4   package org.archive.wayback.accesspoint.proxy;
5   
6   import javax.servlet.http.HttpServletRequest;
7   import javax.servlet.http.HttpServletResponse;
8   
9   import junit.framework.TestCase;
10  
11  import org.archive.wayback.ResultURIConverter;
12  import org.easymock.EasyMock;
13  import org.easymock.IAnswer;
14  
15  /**
16   * Test for {@link ProxyAccessPoint}.
17   */
18  public class ProxyAccessPointTest extends TestCase {
19  
20  	/* (non-Javadoc)
21  	 * @see junit.framework.TestCase#setUp()
22  	 */
23  	protected void setUp() throws Exception {
24  		super.setUp();
25  	}
26  	
27  	/**
28  	 * {@code ProxyAccessPoint} recognizes Archival-URL style request and redirects
29  	 * it to either calendar view (if timestamp is {@code "*"}) or replay of specific capture
30  	 * (if timestamp is explicit).
31  	 * Check if queryString part is property included in the redirect URL (ARI-3903).
32  	 * @throws Exception
33  	 */
34  	public void testHandleProxy() throws Exception {
35  		// request.getRequestURL() returns URL without queryString.
36  		final String REQUEST_PROTOCOL_AND_HOST = "http://wayback.archive-it.org:8081";
37  		// request.getRequestURI() returns path part only.
38  		final String REQUEST_URI = "/3548/20140430194857/http://www.example.edu/index.php";
39  		// request.getQueryString() does not include "?"
40  		final String QUERY_STRING = "id=860233";
41  
42  		HttpServletRequest request = EasyMock.createNiceMock(HttpServletRequest.class);
43  		EasyMock.expect(request.getRequestURL()).andStubAnswer(new IAnswer<StringBuffer>() {
44  			@Override
45  			public StringBuffer answer() throws Throwable {
46  				return new StringBuffer(REQUEST_PROTOCOL_AND_HOST + REQUEST_URI);
47  			}
48  		});
49  		EasyMock.expect(request.getRequestURI()).andStubReturn(REQUEST_URI);
50  		EasyMock.expect(request.getQueryString()).andStubReturn(QUERY_STRING);
51  
52  		ResultURIConverter archivalToProxy = EasyMock.createMock(ResultURIConverter.class);
53  		EasyMock.expect(
54  				archivalToProxy.makeReplayURI("20140430194857",
55  						"http://www.example.edu/index.php?id=860233"))
56  				.andReturn("http://got-it-right/");
57  		HttpServletResponse response = EasyMock.createMock(HttpServletResponse.class);
58  		response.sendRedirect("http://got-it-right/");
59  
60  		ProxyConfigSelector configSelector = EasyMock.createNiceMock(ProxyConfigSelector.class);
61  		EasyMock.expect(configSelector.resolveConfig(request)).andReturn("3548");
62  
63  
64  		// not setting nonProxyAccessPoint
65  		ProxyAccessPoint cut = new ProxyAccessPoint();
66  		cut.setReplayPrefix("//wayback.archive-it.org:8081");
67  		cut.setConfigSelector(configSelector);
68  		cut.setArchivalToProxyConverter(archivalToProxy);
69  		
70  		EasyMock.replay(request, response, archivalToProxy, configSelector);
71  		
72  		boolean r = cut.handleProxy(request, response);
73  		
74  		EasyMock.verify(response, archivalToProxy);
75  		
76  		assertTrue("handleProxy return value", r);
77  	}

103 	protected void init() {
104 		CaptureSearchResults cResults = results.getCaptureResults();
105 		WaybackRequest wbRequest = results.getWbRequest();
106 		String searchUrl =
107 			UrlOperations.stripDefaultPortFromUrl(wbRequest.getRequestUrl());
108 
109 		searchUrlForHTML = StringEscapeUtils.escapeHtml(searchUrl);
110 		searchUrlForJS = StringEscapeUtils.escapeJavaScript(searchUrl);
111 		firstResultDate = cResults.getFirstResultDate();
112 		firstResultReplayUrl = StringEscapeUtils.escapeHtml(results.resultToReplayUrl(cResults.getResults().getFirst()));
113 		lastResultDate = cResults.getLastResultDate();
114 		lastResultReplayUrl = StringEscapeUtils.escapeHtml(results.resultToReplayUrl(cResults.getResults().getLast()));
115 		Date searchStartDate = wbRequest.getStartDate();
116 		Date searchEndDate = wbRequest.getEndDate();
117 		months = capturePartitioner.getRange(monthSize, searchStartDate, searchEndDate);
118 		years =	partitionPartitioner.getRange(yearSize, searchStartDate, searchEndDate);
119 
120 
121 		// To build the graph, we need to break all the results into 1 month
122 		// partitions, so partition all the results into the months:
123 		capturePartitioner.populate(months, cResults.iterator());
124 
125 		// To fill in the calendar, we need to break the current year into day
126 		// sized partitions, so first partition those months into years:
127 		partitionPartitioner.populate(years, months.iterator());
128 		// find the active year:
129 		Partition<Partition<CaptureSearchResult>> activeYear = null;
130 		for (Partition<Partition<CaptureSearchResult>> year : years) {
131 			if (year.isContainsClosest()) {
132 				activeYear = year;
133 				break;
134 			}
135 		}
136 		// if there's no activeYear, something is quite wrong...
137 		// TODO: check anyways:
138 		if (activeYear == null) {
139 			activeYear = years.get(years.size() - 1);
140 		}
141 //		String yearStr = fmt.format("{0,date,yyyy}",activeYear.getStart());
142 //		yearNum = Integer.parseInt(yearStr);
143 		Calendar cal = Calendar.getInstance(calendarTimeZone);
144 		cal.setTime(activeYear.getStart());
145 		yearNum = cal.get(Calendar.YEAR);
146 
147 		// now unroll the months in the active year into day-sized partitions:
148 		List<Partition<CaptureSearchResult>> days =
149 			capturePartitioner.getRange(daySize,
150 					activeYear.getStart(),activeYear.getEnd());
151 		for (Partition<CaptureSearchResult> month : activeYear.list()) {
152 			capturePartitioner.populate(days,month.iterator());
153 		}
154 		// finally, spool the days of the current year into 12 month-sized
155 		// partitions:
156 		monthsByDay = partitionPartitioner.getRange(monthSize, activeYear.getStart(), activeYear.getEnd());
157 		partitionPartitioner.populate(monthsByDay, days.iterator());
158 		dataStartMSSE = years.get(0).getStart().getTime();
159 		dataEndMSSE = years.get(years.size() - 1).getEnd().getTime();
160 		numResults = cResults.getMatchingCount();
161 	}

163 		final String yearFormatKey = "PartitionSize.dateHeader.yearGraphLabel";
164 		Graph yearGraph = PartitionsToGraph.partsOfPartsToGraph(
165 				years, results.getFormatter(), yearFormatKey,
166 				imgWidth, imgHeight);
167 		for (RegionGraphElement rge : yearGraph.getRegions()) {
168 			if (rge.getData().hasHighlightedValue()) {
169 //				rge.getData().setHighlightedValue(-1);
170 			}
171 		}
172 
173 		return GraphEncoder.encode(yearGraph);
174 	}

253 	/**
254 	 * return {@code CaptureSearchResult} partitions for the month grouped
255 	 * by week.
256 	 * Call this method after designating desired month with {@link #setMonth(int)}.
257 	 * Each week group has exactly seven elements, where out-of-month slots are
258 	 * {@code null}.
259 	 * @return Weekly list of {@code Partition}s
260 	 * @exception IllegalStateException if {@code month} is not in 0..11 range.
261 	 */
262 	public List<List<Partition<CaptureSearchResult>>> getCaptureCalendar() {
263 		if (month < 0 || month > 11)
264 			throw new IllegalStateException("invalid month");
265 		Partition<Partition<CaptureSearchResult>> curMonth = monthsByDay.get(month);
266 		List<Partition<CaptureSearchResult>> monthDays = curMonth.list();
267 
268 		Calendar cal = Calendar.getInstance(calendarTimeZone);
269 		cal.setTime(curMonth.getStart());
270 		// DAY_OF_WEEK field has 1 for SUNDAY. Hence this makes week start Sunday.
271 		int skipDays = cal.get(Calendar.DAY_OF_WEEK) - 1;
272 
273 		List<List<Partition<CaptureSearchResult>>> weeks = new ArrayList<List<Partition<CaptureSearchResult>>>();
274 		List<Partition<CaptureSearchResult>> week = new ArrayList<Partition<CaptureSearchResult>>(7);
275 		for (int i = 0; i < skipDays; i++) {
276 			week.add(null);
277 		}
278 		for (Partition<CaptureSearchResult> p : monthDays) {
279 			if (week == null)
280 				week = new ArrayList<Partition<CaptureSearchResult>>(7);
281 			week.add(p);
282 			if (week.size() == 7) {
283 				weeks.add(week);
284 				week = null;
285 			}
286 		}
287 		if (week != null) {
288 			while (week.size() < 7)
289 				week.add(null);
290 			weeks.add(week);
291 		}
292 		return weeks;
293 	}

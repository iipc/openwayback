1   package org.archive.cdxserver.processor;
2   
3   import org.apache.commons.lang.math.NumberUtils;
4   import org.archive.cdxserver.CDXServer;
5   import org.archive.format.cdx.CDXLine;
6   
7   /**
8    * Performs <i>timestamp-based collapsing</i>, that is to group
9    * CDX lines by {@code timestamp} prefix and filter out all but just
10   * one CDX line per group.
11   * <p>Timestamp prefix is specified in terms of the number of digits
12   * (from left). If two CDX lines have {@code timestamp}s whose prefixes are
13   * identical, they are considered to be in the same group.</p>
14   * <p>It picks the first CDX line with the best (i.e. smallest)
15   * {@code statuscode} field within each group.</p>
16   * <p>CDX lines with {@code filename} that starts with any of prefixes
17   * specified in {@code noCollapsePrefix} are written out regardless of its
18   * {@code timestamp} or {@code statuscode}, in addition to the one picked for
19   * the group.</p>
20   * <p>Instantiated by {@link CDXServer} as part of CDX line processing pipeline.</p>
21   */
22  public class DupeTimestampBestStatusFilter extends WrappedProcessor {
23  	final static int WORST_HTTP_CODE = 9999;
24  
25  	protected String lastTimestamp;
26  
27  	protected int bestHttpCode = WORST_HTTP_CODE;
28  	protected int timestampDedupLength;
29  
30  	protected String[] noCollapsePrefix;
31  
32  	public DupeTimestampBestStatusFilter(BaseProcessor output,
33  			int timestampDedupLength, String[] noCollapsePrefix) {
34  		super(output);
35  		this.timestampDedupLength = timestampDedupLength;
36  		this.noCollapsePrefix = noCollapsePrefix;
37  	}
38  
39  	/**
40  	 * Return {@code true} if {@code line} is to be passed through,
41  	 * as specified by {@code noCollapsePrefix}.
42  	 * <p>Soft-blocked captures are also passed-through.</p>
43  	 * @param line CDX line
44  	 * @return boolean
45  	 */
46  	protected final boolean passThrough(CDXLine line) {
47  		return isBlocked(line) || noCollapse(line);
48  	}
49  
50  	protected final boolean isBlocked(CDXLine line) {
51  		String robotflags = line.getRobotFlags();
52  		// TODO: give 'X' a constant symbol - CaptureSearchResult.CAPTURE_ROBOT_BLOCKED
53  		// is exactly that, but wayback-cdx-server cannot use it.
54  		return robotflags != null && robotflags.indexOf('X') >= 0;
55  	}
56  
57  	protected final boolean noCollapse(CDXLine line) {
58  		if (noCollapsePrefix != null) {
59  			for (String prefix : noCollapsePrefix) {
60  				if (line.getFilename().startsWith(prefix)) {
61  					return true;
62  				}
63  			}
64  		}
65  		return false;
66  	}
67  
68  	@Override
69  	public int writeLine(CDXLine line) {
70  		if (include(line)) {
71  			return super.writeLine(line);
72  		} else {
73  			return 0;
74  		}
75  	}
76  
77  	protected boolean include(CDXLine line) {
78  
79  		if (timestampDedupLength <= 0) {
80  			return true;
81  		}
82  
83  		// If starts with special no collapse prefix, then always include
84  		if (passThrough(line)) return true;
85  
86  		String timestamp = line.getTimestamp();
87  		timestamp = timestamp.substring(0, Math.min(timestampDedupLength, timestamp.length()));
88  		int httpCode = NumberUtils.toInt(line.getStatusCode(), WORST_HTTP_CODE);
89  
90  		boolean isDupe = false;
91  
92  		if ((lastTimestamp != null) && timestamp.equals(lastTimestamp)) {
93  			if (httpCode < bestHttpCode) {
94  				bestHttpCode = httpCode;
95  			} else {
96  				isDupe = true;
97  			}
98  		} else {
99  			bestHttpCode = httpCode;
100 		}
101 
102 		lastTimestamp = timestamp;
103 
104 		return !isDupe;
105 	}
106 }

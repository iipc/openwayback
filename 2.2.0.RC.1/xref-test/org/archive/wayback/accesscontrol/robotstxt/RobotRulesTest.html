51  	protected RobotRules load(String txt) throws IOException {
52  		rr.parse(new ByteArrayInputStream(txt.getBytes("UTF-8")));
53  		assertFalse(rr.hasSyntaxErrors());
54  		return rr;
55  	}

147 	/**
148 	 * multiple records for different User-agent's.
149 	 * <p>while RFC states "the format logically consists of a non-empty set
150 	 * or records, separated by blank lines", Google's documentation has no
151 	 * mention to blank lines as group separator - instead, it recognizes a sequence 
152 	 * of User-agent: as the start of "group". So this sample is syntax error per
153 	 * RFC, but okay according to Google/Yahoo/Bing/Ask convention.
154 	 * @throws Exception
155 	 */
156 	public void testNonBlocksPathForUA() throws Exception {
157 		final String testString =
158 				"User-agent: *\n" +
159 				"Allow: /\n" +
160 				"User-agent: Google-bot\n" +
161 				"Disallow: /\n";
162 		load(testString);
163 
164 		assertFalse(rr.blocksPathForUA("/", WB_UA));
165 	}

167 	/**
168 	 * there's sitemap (non-allow/disallow) directive after User-agent.
169 	 * next user-agent should start a new group.
170 	 * ia_archiver ALLOWED as disallow rule only applies to B.
171 	 */
172 	public void testMultiUAWithOtherLinesLine() throws Exception {
173 		final String testString =
174 				"User-agent: *\n" +
175 				"Sitemap: X\n" +
176 				"\n" +
177 				"User-agent: B\n" +
178 				"Disallow: /\n";
179 		load(testString);

202 	/**
203 	 * this is a syntax error per RFC, but ok with Google/Yahoo/Bing/Ask convention.
204 	 * blank lines are permitted within a group to improve readability. they are simply
205 	 * ignored, don't end group.
206 	 */
207 	public void testBlankLineInGroup() throws Exception {
208 		final String testString =
209 				"User-agent: *\n" +
210 				"Disallow: /media/\n" +
211 				"\n" +
212 				"Disallow: /cgi-bin/\n" +
213 				"# images\n" + 
214 				"Disallow: /img/\n" +
215 				"Disallow: /icon/\n" +
216 				"\n" +
217 				"\n" +
218 				"Disallow: /actions/\n" +
219 				"\n" +
220 				"# sensitive stuff\n" +
221 				"Disallow: /etc\n";
222 		load(testString);
223 		
224 		assertTrue(rr.blocksPathForUA("/cgi-bin/", WB_UA));

248 	/**
249 	 * path matching basics. substring-based, <code>/</code> is no special,
250 	 * and case-sensitive.
251 	 */
252 	public void testSubpath() throws Exception {
253 		String testString =
254 				"User-agent: *\n" +
255 				"Disallow: /media\n" +
256 				"Disallow: /hidden/\n";
257 		load(testString);
258 		
259 		assertFalse(rr.blocksPathForUA("/", WB_UA));
260 		assertTrue(rr.blocksPathForUA("/media/theme.mp3", WB_UA));
261 		assertTrue(rr.blocksPathForUA("/media?order=2", WB_UA));
262 		assertFalse(rr.blocksPathForUA("/images/logo.png", WB_UA));
263 		assertFalse(rr.blocksPathForUA("/hidden", WB_UA));
264 		assertTrue(rr.blocksPathForUA("/hidden/index.html", WB_UA));
265 		
266 		// path match is case-sensitive
267 		assertFalse(rr.blocksPathForUA("/Hidden/notreally.mp3", WB_UA));
268 	}

295 	/**
296 	 * <p>By GYBA convention, if multiple
297 	 * disallow and allow directives matches the URL, the most specific
298 	 * rule based on the length of the path will win over less specific
299 	 * (shorter) ones.</p> 
300 	 * <p>RFC says differently: "a robot must attempt to match
301 	 * the paths in Allow and Disallow lines against the URL, in the
302 	 * order they occur in the record. The first match found is used."</p>
303 	 * <p>we follow GYBA convention here.</p>
304 	 * @throws Exception
305 	 */
306 	public void testMostSpecificPathPrevails() throws Exception {
307 		final String testString =
308 				"User-agent: *\n" +
309 				"Disallow: /\n" +
310 				"Allow: /media\n" +
311 				"Disallow: /media/private\n";
312 		load(testString);
313 		
314 		// (some assertions are disabled until we decide to fix it)
315 		assertTrue(rr.blocksPathForUA("/index.html", WB_UA));
316 //		assertFalse(rr.blocksPathForUA("/media/toc.html", WB_UA));
317 //		assertFalse(rr.blocksPathForUA("/media/priv", WB_UA));
318 		assertTrue(rr.blocksPathForUA("/media/private.html", WB_UA));
319 		assertTrue(rr.blocksPathForUA("/media/private/voice.mp3", WB_UA));
320 	}

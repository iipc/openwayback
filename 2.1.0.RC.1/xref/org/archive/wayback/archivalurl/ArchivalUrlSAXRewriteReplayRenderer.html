99  	@Override
100 	public void renderResource(HttpServletRequest httpRequest,
101 			HttpServletResponse httpResponse, WaybackRequest wbRequest,
102 			CaptureSearchResult result, Resource httpHeadersResource,
103 			Resource payloadResource, ResultURIConverter uriConverter,
104 			CaptureSearchResults results) throws ServletException, IOException,
105 			WaybackException {
106 
107 		Resource decodedResource = TextReplayRenderer.decodeResource(httpHeadersResource, payloadResource);
108 
109 		// The URL of the page, for resolving in-page relative URLs: 
110 //		URL url = null;
111 //		try {
112 //			url = new URL(result.getOriginalUrl());
113 //		} catch (MalformedURLException e1) {
114 //			// TODO: this shouldn't happen...
115 //			e1.printStackTrace();
116 //			throw new IOException(e1.getMessage());
117 //		}
118 		// determine the character set used to encode the document bytes:
119 		String charSet = charsetDetector.getCharset(httpHeadersResource, decodedResource, wbRequest);
120 
121 		ContextResultURIConverterFactory fact = createConverterFactory(uriConverter, httpRequest, wbRequest);
122 		
123 		// set up the context:
124 		ReplayParseContext context = 
125 //				new ReplayParseContext(fact,url,result.getCaptureTimestamp());
126 				new ReplayParseContext(fact, result);
127 		
128 		context.setRewriteHttpsOnly(rewriteHttpsOnly);
129 
130 		// XXX same code in ArchivalUrlJSStringReplayRenderer
131 		String policy = result.getOraclePolicy();
132 		if (policy == null) {
133 			AccessPoint accessPoint = wbRequest.getAccessPoint();
134 			if (accessPoint != null) {
135 				policy = accessPoint.getRewriteDirective(result);
136 			}
137 		}
138 		if (policy != null) {
139 			context.setOraclePolicy(policy);
140 		}
141 
142 		if (!wbRequest.isFrameWrapperContext()) {
143 			// in case this is an HTML page with FRAMEs, peek ahead an look:
144 			// TODO: make ThreadLocal:
145 			byte buffer[] = new byte[FRAMESET_SCAN_BUFFER_SIZE];
146 
147 			decodedResource.mark(FRAMESET_SCAN_BUFFER_SIZE);
148 			int amtRead = decodedResource.read(buffer);
149 			decodedResource.reset();
150 
151 			if(amtRead > 0) {
152 				StringBuilder foo = new StringBuilder(new String(buffer,charSet));
153 				int frameIdx = TagMagix.getEndOfFirstTag(foo, "FRAMESET");
154 				if(frameIdx != -1) {
155 					// insert flag so we don't add FRAMESET:
156 					context.putData(FastArchivalUrlReplayParseEventHandler.FERRET_DONE_KEY,"");
157 
158 //					// top-level Frameset: Draw the frame wrapper thingy:
159 //					frameWrappingRenderer.renderResource(httpRequest, 
160 //							httpResponse, wbRequest, result, resource, 
161 //							uriConverter, results);
162 //					return;
163 				}
164 			}
165 		}
166 
167 
168 		// copy the HTTP response code:
169 		HttpHeaderOperation.copyHTTPMessageHeader(httpHeadersResource, httpResponse);
170 
171 		// transform the original headers according to our headerProcessor:
172 		Map<String,String> headers = HttpHeaderOperation.processHeaders(
173 				httpHeadersResource, result, uriConverter, httpHeaderProcessor);
174 
175 		// prepare several objects for the parse:
176 
177 		// a JSPExecutor:
178 //		JSPExecutor jspExec = new JSPExecutor(uriConverter, httpRequest,
179 //				httpResponse, wbRequest, results, result, decodedResource);
180 		UIResults uiResults = new UIResults(wbRequest, uriConverter, results, result, decodedResource);
181 		JSPExecutor jspExec = new JSPExecutor(httpRequest, httpResponse, uiResults);
182 
183 		// To make sure we get the length, we have to buffer it all up...
184 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
185 
186 		context.setOutputCharset(OUTPUT_CHARSET);
187 		context.setOutputStream(baos);
188 		context.setJspExec(jspExec);
189 
190 
191 		// and finally, parse, using the special lexer that knows how to
192 		// handle javascript blocks containing unescaped HTML entities:
193 		Page lexPage = new Page(decodedResource,charSet);
194 		Lexer lexer = new Lexer(lexPage);
195 		Lexer.STRICT_REMARKS = false;
196 		ContextAwareLexer lex = new ContextAwareLexer(lexer, context);
197 		Node node;
198 		try {
199 			delegator.handleParseStart(context);
200 			while((node = lex.nextNode()) != null) {
201 				delegator.handleNode(context, node);
202 			}
203 			delegator.handleParseComplete(context);
204 		} catch (ParserException e) {
205 			e.printStackTrace();
206 			throw new IOException(e.getMessage());
207 		}
208 
209 		// At this point, baos contains the utf-8 encoded bytes of our result:
210 		byte[] utf8Bytes = baos.toByteArray();
211 		// set the corrected length:
212 		headers.put(HttpHeaderOperation.HTTP_LENGTH_HEADER, 
213 				String.valueOf(utf8Bytes.length));
214 		headers.put(TextReplayRenderer.GUESSED_CHARSET_HEADER, charSet);
215 
216 		// send back the headers:
217 		HttpHeaderOperation.sendHeaders(headers, httpResponse);
218 		// Tomcat will always send a charset... It's trying to be smarter than
219 		// we are. If the original page didn't include a "charset" as part of
220 		// the "Content-Type" HTTP header, then Tomcat will use the default..
221 		// who knows what that is, or what that will do to the page..
222 		// let's try explicitly setting it to what we used:
223 		httpResponse.setCharacterEncoding(OUTPUT_CHARSET);
224 		httpResponse.getOutputStream().write(utf8Bytes);
225 	}

227 	protected ContextResultURIConverterFactory createConverterFactory(ResultURIConverter uriConverter, HttpServletRequest httpRequest, WaybackRequest wbRequest)
228 	{
229 		// sam ecode in ArchivalURLJSStringTransformerReplayRenderer
230 		ContextResultURIConverterFactory fact = null;
231 		
232 		if (uriConverter instanceof ArchivalUrlResultURIConverter) {
233 			fact = new ArchivalUrlContextResultURIConverterFactory(
234 					(ArchivalUrlResultURIConverter) uriConverter);
235 		} else if (converterFactory != null) {
236 			fact = converterFactory;
237 		} else {
238 			fact = new IdentityResultURIConverterFactory(uriConverter);			
239 		}
240 		
241 		return fact;
242 	}

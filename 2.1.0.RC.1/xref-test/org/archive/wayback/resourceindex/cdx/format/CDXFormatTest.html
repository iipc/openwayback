20  package org.archive.wayback.resourceindex.cdx.format;
21  
22  import org.archive.wayback.core.CaptureSearchResult;
23  import org.archive.wayback.resourceindex.cdx.format.CDXFormat;
24  import org.archive.wayback.resourceindex.cdx.format.CDXFormatException;
25  
26  import junit.framework.TestCase;
27  
28  public class CDXFormatTest extends TestCase {
29  	public void testParseSpec() {
30  		CaptureSearchResult c;
31  		CDXFormat f = OKFormat(" CDX a V");
32  		c = OKParse(f,"http://foo.com 12");
33  		assertEquals("http://foo.com",c.getOriginalUrl());
34  		assertEquals(c.getOffset(), 12);
35  
36  		
37  		f = OKFormat(" CDX a V k");
38  		c = OKParse(f,"http://foo.com 12 10");
39  		assertEquals("http://foo.com",c.getOriginalUrl());
40  		assertEquals(12,c.getOffset());
41  		assertEquals("10",c.getDigest());
42  		
43  		
44  		exceptionFormat("CDX a k");
45  		exceptionFormat("\tCDX a k");
46  		exceptionFormat("\tCDX a k ");
47  		exceptionFormat(" CDX\ta k");
48  		exceptionFormat(" CDX\ta k\t");
49  		exceptionFormat(" CDX\ta\tk\t");
50  		
51  		f = OKFormat(" CDX\ta\tV\tk");
52  		c = OKParse(f,"http://foo.com\t12\t10");
53  		assertEquals("http://foo.com",c.getOriginalUrl());
54  		assertEquals(12,c.getOffset());
55  		assertEquals("10",c.getDigest());
56  
57  		c = OKParse(f,"http://foo .com\t12\t10");
58  		assertEquals("http://foo .com",c.getOriginalUrl());
59  		assertEquals(12,c.getOffset());
60  		assertEquals("10",c.getDigest());
61  	}
62  	private CaptureSearchResult OKParse(CDXFormat f, String line) {
63  		CaptureSearchResult r = null;
64  		try {
65  			r = f.parseResult(line);
66  		} catch (CDXFormatException e) {
67  			fail(e.getLocalizedMessage());
68  		}
69  		return r;
70  	}
71  	private CDXFormat OKFormat(String format) {
72  		CDXFormat f = null;
73  		try {
74  			f = new CDXFormat(format);
75  		} catch (CDXFormatException e) {
76  			fail("Format '" + format + "' should NOT have thrown exception");
77  		}
78  		return f;
79  	}
80  	private void exceptionFormat(String format) {
81  		try {
82  			new CDXFormat(format);
83  			fail("Format '" + format + "' should have thrown exception");
84  		} catch (CDXFormatException e) {
85  		}
86  		
87  	}
88  }

50  	public static Resource createTestJSResource(byte[] payloadBytes) throws IOException {
51  		WARCRecordInfo recinfo = TestWARCRecordInfo.createHttpResponse("text/javascript", payloadBytes);
52  		TestWARCReader ar = new TestWARCReader(recinfo);
53  		WARCRecord rec = ar.get(0);
54  		WarcResource resource = new WarcResource(rec, ar);
55  		resource.parseHeaders();
56  		return resource;
57  	}

59  	/**
60  	 * Rewrite directive shall be passed from {@link AccessPoint#getRewriteDirective(CaptureSearchResult)}
61  	 * to {@link ReplayParseContext}.
62  	 * @throws Exception
63  	 */
64  	public void testRewriteDirectiveIsPassedToParseContext() throws Exception {
65  		HttpHeaderProcessor httpHeaderProcessor = new IdentityHttpHeaderProcessor();
66  		ArchivalURLJSStringTransformerReplayRenderer renderer = new ArchivalURLJSStringTransformerReplayRenderer(httpHeaderProcessor);
67  
68  		final String rewriteDirective = "rewrite-it";
69  
70  		StringTransformer transformer = new StringTransformer() {
71  			@Override
72  			public String transform(ReplayParseContext context, String input) {
73  				String directive = context.getOraclePolicy();
74  				assertEquals(rewriteDirective, directive);
75  				return input;
76  			}
77  		};
78  		renderer.setTransformer(transformer);
79  
80  		AccessPoint accessPoint = new AccessPoint() {
81  			public String getRewriteDirective(CaptureSearchResult capture) {
82  				return rewriteDirective;
83  			}
84  		};
85  		WaybackRequest wbRequest = WaybackRequest.createReplayRequest(
86  			"http://example.com/", "20140101000000", null, null);
87  		wbRequest.setAccessPoint(accessPoint);
88  
89  		Resource payloadResource = createTestJSResource("aaa".getBytes("UTF-8"));
90  
91  		// ResultURIConverter argument is passed down from AccessPoint#getUriConverter().
92  		// it is typically ProxyHttpsResultURIConverter(), the same class for converterFactory
93  		// (but a separate instance) - we reuse proxyURIConverter above.
94  		HttpServletResponse response = EasyMock.createNiceMock(HttpServletResponse.class);
95  		TestServletOutputStream servletOutput = new TestServletOutputStream();
96  		EasyMock.expect(response.getOutputStream()).andStubReturn(servletOutput);
97  
98  		HttpServletRequest request = null; // assuming unused
99  		CaptureSearchResult result = new CaptureSearchResult();
100 		result.setOriginalUrl("http://example.com/");
101 
102 		ResultURIConverter uriConverter = new ArchivalUrlResultURIConverter();
103 
104 		EasyMock.replay(response);
105 
106 		renderer.renderResource(request, response, wbRequest, result, payloadResource, uriConverter, null);
107 	}

109 	/**
110 	 * multi-component test with typical usage pattern:
111 	 * rewriting {@code https://} to {@code http://} in proxy-mode.
112 	 * (ArchivalURLJSStringTransformerReplayRenderer is also used in
113 	 * proxy-mode despite its name.)
114 	 * 
115 	 * @see org.archive.wayback.replay.html.transformer.JSStringTransformerTest
116 	 */
117 	public void testProxyHttpsTranslation() throws Exception {
118 		HttpHeaderProcessor httpHeaderProcessor = new RedirectRewritingHttpHeaderProcessor();
119 		ArchivalURLJSStringTransformerReplayRenderer renderer = new ArchivalURLJSStringTransformerReplayRenderer(
120 				httpHeaderProcessor);
121 		// not testing jspInserts - TODO
122 		
123 		// in production transformer is a MultiRegexReplaceStringTransformer
124 		// running other rewrites besides JSStringTransformer.  We are
125 		// only testing JSStringTransformer here.
126 		JSStringTransformer transformer = new JSStringTransformer();
127 		renderer.setTransformer(transformer);
128 		
129 		ResultURIConverter proxyURIConverter = new ProxyHttpsResultURIConverter();
130 		ContextResultURIConverterFactory converterFactory =
131 				new IdentityResultURIConverterFactory(proxyURIConverter);
132 		renderer.setConverterFactory(converterFactory);
133 		renderer.setRewriteHttpsOnly(true);
134 		
135 		final String payload =
136 				"var img1 = 'https://home.archive.org/~hstern/ARI-3745/happy_face.jpg';\n" +
137 				"var el1 = document.createElement('img');\n" +
138 				"el1.src = img;\n" +
139 				"document.getElementById('imgdiv').appendChild(el1)\n";
140 		final String expected =
141 				"var img1 = 'http://home.archive.org/~hstern/ARI-3745/happy_face.jpg';\n" +
142 				"var el1 = document.createElement('img');\n" +
143 				"el1.src = img;\n" +
144 				"document.getElementById('imgdiv').appendChild(el1)\n";
145 		
146 		final byte[] payloadBytes = payload.getBytes("UTF-8");
147 		Resource payloadResource = createTestJSResource(payloadBytes);
148 		
149 		// ResultURIConverter argument is passed down from AccessPoint#getUriConverter().
150 		// it is typically ProxyHttpsResultURIConverter(), the same class for converterFactory
151 		// (but a separate instance) - we reuse proxyURIConverter above.
152 		HttpServletResponse response = EasyMock.createNiceMock(HttpServletResponse.class);
153 		TestServletOutputStream servletOutput = new TestServletOutputStream();
154 		EasyMock.expect(response.getOutputStream()).andStubReturn(servletOutput);
155 		
156 		HttpServletRequest request = null; // assuming unused
157 		WaybackRequest wbRequest = new WaybackRequest();
158 		CaptureSearchResult result = new CaptureSearchResult();
159 		result.setOriginalUrl("http://home.archive.org/~hstern/ARI-3745/");
160 		
161 		EasyMock.replay(response);
162 		
163 		renderer.renderResource(request, response, wbRequest, result, payloadResource, payloadResource, proxyURIConverter, null);
164 		
165 		String out = servletOutput.getString();
166 		assertEquals("servlet output", expected, out);
167 	}

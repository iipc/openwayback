20  package org.archive.wayback;
21  
22  import java.io.IOException;
23  
24  import javax.servlet.ServletException;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  
28  import org.archive.wayback.core.CaptureSearchResults;
29  import org.archive.wayback.core.UrlSearchResults;
30  import org.archive.wayback.core.WaybackRequest;
31  
32  /**
33   * {@code QueryRenderer} generates response for two types of resource index
34   * query: <em>capture search</em> and <em>URL search</em>.
35   *
36   * <h3>REFACTORING THOUGHTS</h3>
37   * <p>There's no good reason two types of query should be bundled in one object.
38   * Probably we should split this interface into two interfaces, or for the better,
39   * just one interface covering the two. It'll open up a possibility of unifying
40   * {@link ReplayRenderer} and {@code QueryRendere} into one interface to simplify
41   * {@code AccessPoint}.
42   * </p>
43   * @author brad
44   * @see ReplayRenderer
45   */
46  public interface QueryRenderer {
47  
48  	/**
49  	 * Display matching SearchResults for the WaybackRequest to the user.
50  	 *
51  	 * @param httpRequest the HttpServletRequest
52  	 * @param httpResponse the HttpServletResponse
53  	 * @param wbRequest the WaybackRequest that returned the results
54  	 * @param results the SearchResults that the WaybackRequest matched
55  	 * @param uriConverter the URI converter to use to translate matching
56  	 *        results into replayable URLs
57  	 * @throws ServletException per usual
58  	 * @throws IOException per usual
59  	 */
60  	public void renderCaptureResults(HttpServletRequest httpRequest,
61  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
62  			CaptureSearchResults results, ResultURIConverter uriConverter)
63  			throws ServletException, IOException;
64  
65  	/**
66  	 * Show the SearchResults of the request which may have resulted in multiple
67  	 * matching URLs.
68  	 *
69  	 * @param httpRequest the HttpServletRequest
70  	 * @param response the HttpServletResponse
71  	 * @param wbRequest the WaybackRequest that returned the results
72  	 * @param results the SearchResults that the WaybackRequest matched
73  	 * @param uriConverter the URI converter to use to translate matching
74  	 *        results into replayable URLs
75  	 * @throws ServletException per usual
76  	 * @throws IOException per usual
77  	 */
78  	public void renderUrlResults(HttpServletRequest httpRequest,
79  			HttpServletResponse response, WaybackRequest wbRequest,
80  			UrlSearchResults results, ResultURIConverter uriConverter)
81  			throws ServletException, IOException;
82  
83  }

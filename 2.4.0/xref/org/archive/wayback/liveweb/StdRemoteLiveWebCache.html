50  /**
51   * This class fetches resource from live web. 
52   * It works with standard proxy server e.g. Squid.
53   * 
54   * @author Mohamed Elsayed
55   * @see LiveWebCache
56   * @see ArcRemoteLiveWebCache
57   */
58  public class StdRemoteLiveWebCache implements LiveWebCache
59  {
60      private static final Logger LOGGER = Logger.getLogger(
61  			StdRemoteLiveWebCache.class.getName() );
62  
63      protected MultiThreadedHttpConnectionManager connectionManager;
64      protected HostConfiguration hostConfiguration;
65      protected HttpClient httpClient; 
66      protected String requestPrefix;
67      private CloseableHttpResponse response;
68      private ArcResource ar;
69      
70      /**
71       * StdRemoteLiveWebCache constructor initializes and configures connection objects.
72       */
73      public StdRemoteLiveWebCache() 
74      {
75      	connectionManager = new MultiThreadedHttpConnectionManager();
76      	hostConfiguration = new HostConfiguration();
77  	HttpClientParams params = new HttpClientParams();
78          params.setParameter( HttpClientParams.RETRY_HANDLER, 
79                  new NoRetryHandler() );
80      	httpClient = new HttpClient( params, connectionManager );
81      	httpClient.setHostConfiguration( hostConfiguration );
82      }
83      
84      /**
85       * Gets resource object from the live web. Configure timeout to 10 seconds.
86       *  
87       * @param url to fetch from the live web.
88       * @param maxCacheMS maximum age of resource to return - optionally honored
89       * @param bUseOlder if true, return documents older than maxCacheMS if
90       *                  a more recent copy is not available.
91       * 
92       * @return Resource for url
93       * 
94       * @throws LiveDocumentNotAvailableException if the resource cannot be
95       *         retrieved from the live web, but all proxying and caching 
96       *         mechanisms functioned properly
97       * @throws LiveWebCacheUnavailableException if there was a problem either
98       * 		   accessing the live web, in proxying to the live web, or in
99       * 		   maintaining the cache for the live web
100      * @throws LiveWebTimeoutException if there is no response from the live
101      * 		   web cache before a timeout occurred.
102      * @throws IOException for the usual reasons
103      * 
104      * @see org.archive.wayback.liveweb.LiveWebCache#getCachedResource(java.net.URL, long, boolean)
105      * @inheritDoc org.archive.wayback.liveweb.LiveWebCache#getCachedResource
106      */
107     @Override
108     public Resource getCachedResource( URL url, long maxCacheMS, 
109             boolean bUseOlder ) 
110             throws LiveDocumentNotAvailableException,
111             LiveWebCacheUnavailableException, LiveWebTimeoutException, IOException 
112     {
113         String urlStr = url.toExternalForm();
114         
115         if (requestPrefix != null)
116                 urlStr = requestPrefix + urlStr;
117         
118         HttpHost proxy = new HttpHost( hostConfiguration.getProxyHost(),
119                 hostConfiguration.getProxyPort() );
120         
121         // Set socketTimeout and connectionTimeout to 10 seconds.
122         RequestConfig reqConf = RequestConfig.custom().setProxy( proxy )
123                 .setSocketTimeout( 10000 )
124                 .setConnectTimeout( 10000 )
125                 .setConnectionRequestTimeout( 10000 )
126                 .build();
127         CloseableHttpClient httpclient = HttpClients.custom().
128                 setDefaultRequestConfig(reqConf).build();
129         HttpGet httpGet = new HttpGet( urlStr );
130         
131         try 
132         {
133             // The following line gets robots.txt from live web
134             response= httpclient.execute( httpGet );
135             
136             String httpHeaderStr = "";
137             String bodyStr = "";
138             
139             /* If it fails to get robots.txt (http status code is 404),
140                then display contents and don't throw exception
141                (socketTimeOutException or connectTimeOutException)
142             */
143             if ( response.getStatusLine().getStatusCode() == 404 )
144             {
145                 httpHeaderStr = "HTTP/1.0 200 OK\n";
146                 bodyStr = String.format( "%s\n%s\n",
147                         "User-agent: *", "Allow: /" );
148             }
149             else if ( response.getStatusLine().getStatusCode() == 200 )
150             {
151                 // The following line represents first line in http header
152                 httpHeaderStr = String.format( "%s %d %s\n", 
153                        response.getStatusLine().getProtocolVersion(),
154                        response.getStatusLine().getStatusCode(),
155                        response.getStatusLine().getReasonPhrase() );
156 
157                 // Get robots.txt contents and store it into bodyStr
158                HttpEntity entity = response.getEntity();
159                bodyStr = EntityUtils.toString(entity);
160             }
161             
162             // Get Http Header and store complete http header in httpHeaderStr
163             for ( Header header : response.getAllHeaders() )
164                 httpHeaderStr += header.toString() + "\n";
165             
166             httpHeaderStr += "\n";
167             int length = httpHeaderStr.length() + bodyStr.length();
168             
169             /*
170                 Using httpHeaderStr and bodyStr to construct responseStr.
171                 First line in responseStr should exist.
172             */
173             
174             // TODO: the following line should be enhanced, 
175             //       especially the first line in responseStr. 
176             String responseStr = String.format( "%s %s %d\n%s%s", urlStr, 
177                     "0.0.0.0 10000000000000 text/plain", length,
178                     httpHeaderStr, bodyStr );
179             
180             ByteArrayInputStream bais = new ByteArrayInputStream(
181                     responseStr.getBytes() );
182             
183             // TODO: Should not use ARCRecord
184             ARCRecord r = new ARCRecord( bais, "id", 0L, false, false, true );
185             ar = ( ArcResource ) ResourceFactory.ARCArchiveRecordToResource( r, null );
186             
187             if ( ar.getStatusCode() == 502 ) 
188             {
189                     throw new LiveDocumentNotAvailableException( urlStr );
190             } 
191             else if ( ar.getStatusCode() == 504 ) 
192             {
193                     throw new LiveWebTimeoutException( "Timeout:" + urlStr );
194             }
195 
196             return ar;
197         }
198         catch( ResourceNotAvailableException e ) 
199         {
200             throw new LiveDocumentNotAvailableException( urlStr );
201         }
202         catch( NoHttpResponseException e ) 
203         {
204             throw new LiveWebCacheUnavailableException( "No Http Response for " +
205                     urlStr );
206         }
207         catch( ConnectException e ) 
208         {
209             throw new LiveWebCacheUnavailableException( e.getLocalizedMessage() +
210                     " : " + urlStr );
211         }
212         catch ( SocketException e ) 
213         {
214             throw new LiveWebCacheUnavailableException( e.getLocalizedMessage() +
215                     " : " + urlStr );
216         }
217         catch ( SocketTimeoutException e ) 
218         {
219             throw new LiveWebTimeoutException( e.getLocalizedMessage() + " : " +
220                     urlStr );
221         }
222         catch( ConnectTimeoutException e ) 
223         {
224             throw new LiveWebTimeoutException( e.getLocalizedMessage() + " : " +
225                     urlStr );
226         }
227         finally 
228         {
229             response.close();
230         }
231     }

233     /**
234      * Sets proxy and port (proxy:port).
235      * 
236      * @param hostPort to proxy requests through - ex. "localhost:3128"
237      */
238     public void setProxyHostPort( String hostPort ) 
239     {
240     	int colonIdx = hostPort.indexOf( ':' );
241     	if(colonIdx > 0) 
242         {
243             String host = hostPort.substring( 0,colonIdx );
244             int port = Integer.valueOf( hostPort.substring( colonIdx+1 ) );
245             hostConfiguration.setProxy( host, port );
246     	}
247     }

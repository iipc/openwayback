38  /**
39   * {@code ServerRelativeArchivalRedirect} is a {@link RequestHandler}
40   * that redirects <i>leaked</i> server-relative URL back to replay request
41   * URL.
42   * <p>For example, assuming {@code Referer} is
43   * {@code http://web.archive.org/web/20010203040506/http://example.com/index.html},
44   * it redirects request {@code http://web.archive.org/js/foo.js}
45   * to {@code http://web.archive.org/web/20010203040506/http://example.com/js/foo.js}.</p>
46   * <p>It is typically set up as catch-all {@code RequestHandler}</p>
47   * <p>
48   * Refactoring Thoughts: parsing and construction of Archival-URL in this
49   * class must match that of Archival-URL access point. Consider delegating
50   * those to AccessPoint for easier configuration.
51   * </p>
52   * @author brad
53   */
54  public class ServerRelativeArchivalRedirect extends AbstractRequestHandler {
55  	private static final Logger LOGGER = Logger
56  		.getLogger(ServerRelativeArchivalRedirect.class.getName());
57  
58  	boolean useCollection = false;
59  	private String matchHost = null;
60  	private int matchPort = -1;
61  	private String replayPrefix;

82  	/**
83  	 * Return the Archival-URL projection of request origin, that
84  	 * should give a context necessary to redirect the leaked request
85  	 * back to Archival-URL space.
86  	 * <p>
87  	 * Default implementation parses {@code Referer} header.
88  	 * Sub class may override this method to use alternative method
89  	 * of obtaining equivalent information. Typically you want to
90  	 * call super method first, then resort to alternative method
91  	 * if super method returns {@code null} or incomplete info.
92  	 * </p>
93  	 * @param httpRequest request object
94  	 * @return ArchivalUrlCaptureRef object or {@code null}
95  	 * if valid information cannot be found.
96  	 */
97  	protected ArchivalUrlRef getOrigin(HttpServletRequest httpRequest) {
98  		// TODO: check if this works with non-empty context path.
99  		// I believe requestURI starts with context path. So non-empty
100 		// context path will break collection and timestamp extraction
101 		// code below. It'd be desirable for this class to get a pattern
102 		// of replay URL from somewhere - replayPreifx?
103 
104 		// hope that it's a server relative request, with a valid referrer:
105 		String referer = httpRequest.getHeader("Referer");
106 		if (referer == null) return null;
107 
108 		final UsableURI refuri;
109 		final String host;
110 		final String authority;
111 		String path;
112 		try {
113 			refuri = UsableURIFactory.getInstance(referer);
114 			host = refuri.getHost();
115 			authority = refuri.getAuthority();
116 			path = refuri.getPath();
117 		} catch (URIException ex) {
118 			LOGGER.info("Ignoring unparsable Referer: " + referer);
119 			return null;
120 		}
121 
122 		// Check that the Referer is our current wayback path
123 		// before attempting to use referer as base archival url
124 
125 		if ((matchHost != null && !matchHost.equals(host)) ||
126 				(matchPort != -1 && refuri.getPort() != -1 && matchPort != refuri
127 				.getPort())) {
128 			LOGGER.info("Server-Relative-Redirect: Skipping, Referer " +
129 					host + ":" + refuri.getPort() +
130 					" not from matching wayback host:port\t");
131 			return null;
132 		}
133 
134 		String collection = null;
135 		if (useCollection) {
136 			int colSlash = path.indexOf('/', 1);
137 			if (colSlash == -1) return null;
138 
139 			// !! collection has leading '/'. It's bad, but existing
140 			// sub-class may break if we change it.
141 			collection = modifyCollection(path.substring(0, colSlash));
142 			path = path.substring(colSlash + 1);
143 		} else {
144 			// next line expects path does not start with "/"
145 			path = path.substring(1);
146 		}
147 
148 		int tsSlash = path.indexOf('/');
149 		if (tsSlash == -1) return null;
150 
151 		String datespec = path.substring(0, tsSlash);
152 		if (!datespec.isEmpty() &&
153 				!Character.isDigit(datespec.charAt(0))) {
154 			datespec = null;
155 		}
156 
157 		String url = path.substring(tsSlash + 1);
158 		url = UrlOperations.fixupScheme(url);
159 		url = ArchiveUtils.addImpliedHttpIfNecessary(url);
160 
161 		final String root = refuri.getScheme() + "://" + authority;
162 		return new ArchivalUrlRef(root, collection, datespec, url);
163 	}

165 	private String handleRequestWithCollection(HttpServletRequest httpRequest,
166 			HttpServletResponse httpResponse) throws ServletException,
167 			IOException {
168 		ArchivalUrlRef ref = getOrigin(httpRequest);
169 		if (ref == null) return null;
170 
171 		String thisPath = httpRequest.getRequestURI();
172 		String queryString = httpRequest.getQueryString();
173 		if (queryString != null) {
174 			thisPath += "?" + queryString;
175 		}
176 		String resolved = UrlOperations.resolveUrl(ref.url, thisPath);
177 
178 		String contextPath = httpRequest.getContextPath();
179 		StringBuilder sb = new StringBuilder(ref.root);
180 		sb.append(contextPath);
181 		sb.append(ref.collection);
182 		sb.append("/");
183 		if (ref.datespec != null) {
184 			sb.append(ref.datespec);
185 			sb.append("/");
186 		}
187 		sb.append(resolved);
188 
189 		return sb.toString();
190 	}

203 	private String handleRequestWithoutCollection(
204 			HttpServletRequest httpRequest, HttpServletResponse httpResponse)
205 			throws ServletException, IOException {
206 
207 		ArchivalUrlRef ref = getOrigin(httpRequest);
208 		if (ref == null) return null;
209 
210 		String thisPath = httpRequest.getRequestURI();
211 		String queryString = httpRequest.getQueryString();
212 		if (queryString != null) {
213 			thisPath += "?" + queryString;
214 		}
215 
216 		String resolved = UrlOperations.resolveUrl(ref.url, thisPath);
217 		String contextPath = httpRequest.getContextPath();
218 		String finalUrl = ref.root + //uri.getScheme() + "://" + uri.getAuthority() +
219 				contextPath + "/" + ref.datespec + "/" + resolved;
220 
221 		return finalUrl;
222 	}

224 	@Override
225 	public boolean handleRequest(HttpServletRequest httpRequest,
226 			HttpServletResponse httpResponse) throws ServletException,
227 			IOException {
228 		if (matchHost != null) {
229 			if (!matchHost.equals(httpRequest.getServerName())) {
230 				LOGGER.fine("Wrong host for ServerRelativeRed(" +
231 						httpRequest.getServerName() + ")");
232 				return false;
233 			}
234 		}
235 		if (matchPort != -1) {
236 			if (matchPort != httpRequest.getLocalPort()) {
237 				LOGGER.fine("Wrong port for ServerRealtiveRed(" +
238 						httpRequest.getServerName() + ")(" +
239 						httpRequest.getLocalPort() + ") :" +
240 						httpRequest.getRequestURI());
241 				return false;
242 			}
243 		}
244 		String replayUrl = (useCollection ?
245 				handleRequestWithCollection(httpRequest, httpResponse) :
246 					handleRequestWithoutCollection(httpRequest, httpResponse));
247 		if (replayUrl == null && replayPrefix != null) {
248 			String thisPath = httpRequest.getRequestURI();
249 			String queryString = httpRequest.getQueryString();
250 			if (queryString != null) {
251 				thisPath += "?" + queryString;
252 			}
253 			// TODO: rethink this fallback, for now adding https support as
254 			// well
255 			if (thisPath.startsWith("/http://") ||
256 					thisPath.startsWith("/https://")) {
257 				// assume a replay request:
258 				StringBuilder sb = new StringBuilder(thisPath.length() +
259 					replayPrefix.length() + 16);
260 				sb.append(replayPrefix);
261 				sb.append(Timestamp.currentTimestamp().getDateStr());
262 				sb.append(thisPath);
263 				replayUrl = sb.toString();
264 			}
265 		}
266 		if (replayUrl != null) {
267 			// Gotta make sure this is properly cached, or
268 			// weird things happen:
269 			httpResponse.addHeader("Vary", "Referer");
270 			httpResponse.sendRedirect(replayUrl);
271 			return true;
272 		}
273 		return false;
274 	}

20  package org.archive.wayback.webapp;
21  
22  import java.util.logging.Level;
23  import java.util.logging.Logger;
24  
25  /**
26   * Brutally simple, barely functional class to allow simple recording of 
27   * millisecond level timing within a particular request, enabling rough logging
28   * of the time spent in various parts of the handling of a WaybackRequest
29   * @author brad
30   *
31   */
32  public class PerformanceLogger {
33  	private static final Logger LOGGER = Logger.getLogger(
34  			PerformanceLogger.class.getName());
35  
36  	private static char delim = '\t';
37  	
38  	private String type = null;
39  	private long start = 0;
40  	private long query = 0;
41  	private long retrieve = -1;
42  	private long render = 0;
43  	/**
44  	 * Construct a Performance logger with the specified String "type"
45  	 * @param type the String type to report with the logged output
46  	 */
47  	public PerformanceLogger(String type) {
48  		this.type = type;
49  		this.start = System.currentTimeMillis();
50  	}
51  	/**
52  	 * record the time when the query associated with this request completed
53  	 */
54  	public void queried() {
55  		this.query = System.currentTimeMillis();
56  	}
57  	/**
58  	 * record the time when the retrieval of a Resource required for this 
59  	 * request completed, implies a Replay request...
60  	 */
61  	public void retrieved() {
62  		this.retrieve = System.currentTimeMillis();
63  	}
64  	/**
65  	 * record the time when the replayed resource, or the query results were
66  	 * returned to the client, implies the bulk of the request processing is 
67  	 * complete.
68  	 */
69  	public void rendered() {
70  		this.render = System.currentTimeMillis();
71  	}
72  	/**
73  	 * Produce a debug message to this classes logger, computing the time
74  	 * taken to query the index, retrieve the resource (if a replay request)
75  	 * and render the results to the client. 
76  	 * @param info String suffix to append to the log message 
77  	 */
78  	public void write(String info) {
79  		StringBuilder sb = new StringBuilder(40);
80  		sb.append(type).append(delim);
81  		sb.append(query - start).append(delim);
82  		if(retrieve == -1) {
83  			sb.append(render - query).append(delim);
84  		} else {
85  			sb.append(retrieve - query).append(delim);
86  			sb.append(render - retrieve).append(delim);
87  		}
88  		sb.append(info);
89  		LOGGER.finer(sb.toString());
90  	}
91  	public static void noteElapsed(String message, long elapsed, String note) {
92  		if(LOGGER.isLoggable(Level.INFO)) {
93  			StringBuilder sb = new StringBuilder();
94  			sb.append("WB-PERF\t").append(message).append("\t").append(elapsed);
95  			if(note != null) {
96  				sb.append("\t").append(note);
97  			}
98  			LOGGER.info(sb.toString());
99  		}
100 	}
101 	
102 	public static void noteElapsed(String message, long elapsed) {
103 		noteElapsed(message,elapsed,null);
104 	}
105 }

20  package org.archive.wayback.resourcestore;
21  
22  import java.io.IOException;
23  import java.util.logging.Logger;
24  
25  import org.archive.wayback.ResourceStore;
26  import org.archive.wayback.core.CaptureSearchResult;
27  import org.archive.wayback.core.Resource;
28  import org.archive.wayback.exception.ResourceNotAvailableException;
29  import org.archive.wayback.resourcestore.resourcefile.ArcWarcFilenameFilter;
30  import org.archive.wayback.resourcestore.resourcefile.ResourceFactory;
31  
32  
33  /**
34   * Implements ResourceStore where ARC/WARCs are accessed via a local file or an
35   * HTTP 1.1 range request. All files are assumed to be "rooted" at a particular
36   * HTTP URL, or within a single local directory. The HTTP version may imply a
37   * file reverse-proxy to connect through to actual HTTP ARC/WARC locations.
38   *
39   * @author brad
40   * @version $Date$, $Revision$
41   */
42  public class SimpleResourceStore implements ResourceStore {
43  
44  	private final static Logger LOGGER = Logger.getLogger(
45  			SimpleResourceStore.class.getName());
46  	private static String HTTP_ERROR = "HTTP";
47  	private static String HTTP_502 = "502";
48  	private String prefix = null;
49  	private String regex;
50  	private String replace;
51  	private String includeFilter;
52    
53  	private int retries = 2;
54  
55  	public Resource retrieveResource(CaptureSearchResult result)
56  		throws ResourceNotAvailableException {
57  
58  		// extract ARC filename
59  		String fileName = result.getFile();
60  		if(fileName == null || fileName.length() < 1) {
61  			throw new ResourceNotAvailableException("No ARC/WARC name in search result...", fileName);
62  		}
63  		
64          // If includeFilter is provided, filter out paths that don't contain the include filter
65          if (includeFilter != null) {
66          	if (!fileName.contains(includeFilter)) {
67          		throw new ResourceNotAvailableException("Resource " + fileName + " not found in this store", fileName);
68          	}
69          }		
70  
71  		final long offset = result.getOffset();
72  		if(!fileName.endsWith(ArcWarcFilenameFilter.ARC_SUFFIX)
73  				&& !fileName.endsWith(ArcWarcFilenameFilter.ARC_GZ_SUFFIX)
74  				&& !fileName.endsWith(ArcWarcFilenameFilter.WARC_SUFFIX)
75  				&& !fileName.endsWith(ArcWarcFilenameFilter.WARC_GZ_SUFFIX)) {
76  			fileName = fileName + ArcWarcFilenameFilter.ARC_GZ_SUFFIX;
77  		}
78  				
79          String fileUrl;
80          if ( regex != null && replace != null )
81            {
82              fileUrl = fileName.replaceAll( regex, replace );
83            }
84          else
85            {
86              fileUrl = prefix + fileName;
87            }
88          
89  		Resource r = null;
90  		try {
91  			int attempts = retries;
92  	        while(attempts-- > 0) {
93  	        	try {
94  	        		r = ResourceFactory.getResource(fileUrl, offset);
95  	        		break;
96  	        	} catch (IOException e) {
97  	        		String message = e.getMessage();
98  	        		if(attempts > 0 
99  	        				&& message.contains(HTTP_ERROR) 
100 	        				&& message.contains(HTTP_502)) {
101 	        			
102 	        			LOGGER.info(String.format(
103 	        					"Failed attempt for (%s) retrying with" +
104 	        					" (%d) attempts left",fileUrl,attempts));
105 	        		} else {
106 	        			throw e;
107 	        		}
108 	        	}
109 	        }
110 
111 		} catch (IOException e) {
112 			String msg = fileUrl + " - " + e;
113 			LOGGER.info("Unable to retrieve:" + msg);
114 			
115 			throw new ResourceNotAvailableException(msg, fileUrl, e);
116 		}
117 		return r;
118 	}
119 
120 	/**
121 	 * @return the prefix
122 	 */
123 	public String getPrefix() {
124 		return prefix;
125 	}
126 
127 	/**
128 	 * @param prefix the prefix to set
129 	 */
130 	public void setPrefix(String prefix) {
131 		this.prefix = prefix;
132 	}
133 
134 	public String getRegex() {
135 		return regex;
136 	}
137 
138 	public void setRegex(String regex) {
139 		this.regex = regex;
140         }
141 
142 	public String getReplace() {
143 		return replace;
144 	}
145 
146 	public void setReplace(String replace) {
147 		this.replace = replace;
148         }
149 
150 	public void shutdown() throws IOException {
151 		// no-op
152 	}
153 
154 	/**
155 	 * @return the number of attempts to fetch resources with an HTTP 502 
156 	 * failure.
157 	 */
158 	public int getRetries() {
159 		return retries;
160 	}
161 
162 	/**
163 	 * @param retries the number of attempts to fetch resources with an HTTP 502 
164 	 * failure.
165 	 */
166 	public void setRetries(int retries) {
167 		this.retries = retries;
168 	}
169 
170 	
171 	public String getIncludeFilter() {
172 		return includeFilter;
173 	}
174 
175 	public void setIncludeFilter(String includeFilter) {
176 		this.includeFilter = includeFilter;
177 	}
178 }

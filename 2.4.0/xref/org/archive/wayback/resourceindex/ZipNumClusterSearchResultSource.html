17   * SearchResultSource wrapper for new implementation of CDX input source, including ZipNumCluster and CDX File

23  public class ZipNumClusterSearchResultSource implements SearchResultSource, Adapter<String,CaptureSearchResult> {
24  		
25  	protected ZipNumCluster cluster;
26  	protected ZipNumParams params = null;
27  	
28  	protected ZipNumParams oneBlockParams;
29  	
30  	protected int timestampDedupLength = 0;
31  	
32  	protected List<String> ignoreRobotPaths;
33  		
34  	public void init() throws IOException
35  	{
36  		//this.cluster = new ZipNumCluster(clusterUri, summaryFile, blockLoader);
37  		
38  		oneBlockParams = new ZipNumParams();
39  		oneBlockParams.setMaxBlocks(1);
40  	}
41  	
42  	@Override
43  	public CloseableIterator<CaptureSearchResult> getPrefixIterator(
44  			String urlkey) throws ResourceIndexNotAvailableException {
45  		
46  		try {			
47  			CloseableIterator<String> cdxIter = null;
48  			
49  			String prefix = urlkey;
50  			
51  			int space = prefix.indexOf(' ');
52  			
53  			// One-block query
54  			if (space >= 0) {
55  				prefix = prefix.substring(0, space);
56  				
57  				cdxIter = cluster.getCDXIterator(urlkey, prefix, true, oneBlockParams);
58  			// Exact Match
59  			} else if (!prefix.endsWith("*\t")) {
60  				cdxIter = cluster.getCDXIterator(urlkey, prefix, true, params);
61  			// Prefix Match
62  			} else {
63  				cdxIter = cluster.getCDXIterator(urlkey, prefix.substring(0, prefix.length() - 2), false, params);
64  			}
65  			
66  			if (timestampDedupLength > 0) {
67  				cdxIter = new TimestampBestPickDedupIterator(cdxIter, timestampDedupLength);
68  			}
69  			
70  			return new AdaptedIterator<String,CaptureSearchResult>(cdxIter, this);
71  			 
72  			
73  		} catch (IOException e) {
74  			throw new ResourceIndexNotAvailableException(e.toString());
75  		}
76  	}
77  
78  	@Override
79  	public CloseableIterator<CaptureSearchResult> getPrefixReverseIterator(
80  			String prefix) throws ResourceIndexNotAvailableException {
81  		
82  		throw new ResourceIndexNotAvailableException("Unsupported");
83  	}

113 	@Override
114 	public CaptureSearchResult adapt(String line) {
115 		CaptureSearchResult result = CDXFlexFormat.parseCDXLineFlexFast(line);
116 		
117 		if (ignoreRobotPaths != null) {
118 			for (String path : ignoreRobotPaths) {
119 				if (result.getUrlKey().startsWith(path)) {
120 					result.setRobotIgnore();
121 					break;
122 				}
123 			}
124 		}
125 		
126 		return result;
127 	}
128

130 		return ignoreRobotPaths;
131 	}
132

134 		this.ignoreRobotPaths = ignoreRobotPaths;
135 	}
136

138 		return timestampDedupLength;
139 	}
140 
141 	public void setTimestampDedupLength(int timestampDedupLength) {
142 		this.timestampDedupLength = timestampDedupLength;
143 	}	
144 }

1   package org.archive.wayback.replay.charset;
2   
3   import java.io.IOException;
4   
5   import org.archive.wayback.core.Resource;
6   import org.archive.wayback.replay.TagMagix;
7   
8   /**
9    * {@link EncodingSniffer} that pre-scan byte stream for
10   * {@code <meta http-equiv="content-type" ... >} tag.
11   * <p>This is step 6 of WHAT-NG prescription, but decodes pre-scanned
12   * content as {@code UTF-8} to simplify the code. That should okay
13   * for the purpose...</p>
14   * <p>CHANGE: 1.8.1 2014-07-07 override {@code UTF-16} encodings to
15   * {@code UTF-8}, and x-user-defined encoding to {@code Windows-1252},
16   * as prescribed by WHAT-NG.</p>
17   */
18  public class PrescanMetadataSniffer extends BaseEncodingSniffer {
19  	// hand off this many bytes to the chardet library
20  	protected final static int MAX_CHARSET_READAHEAD = 65536;
21  
22  	@Override
23  	public String sniff(Resource resource) {
24  		String charsetName = null;
25  		try {
26  			byte[] bbuffer = new byte[MAX_CHARSET_READAHEAD];
27  			resource.mark(MAX_CHARSET_READAHEAD);
28  			resource.read(bbuffer, 0, MAX_CHARSET_READAHEAD);
29  			resource.reset();
30  			// convert to UTF-8 String -- which hopefully will not mess up the
31  			// characters we're interested in...
32  			StringBuilder sb = new StringBuilder(new String(bbuffer,
33  					"UTF-8"));
34  			String metaContentType = TagMagix.getTagAttrWhere(sb, "META",
35  				"content", "http-equiv", "Content-Type");
36  			if (metaContentType != null) {
37  				charsetName = contentTypeToCharset(metaContentType);
38  				// override
39  				if (charsetName != null) {
40  					String upped = charsetName.toUpperCase();
41  					if (upped.startsWith("UTF-16"))
42  						charsetName = "UTF-8";
43  				}
44  			}
45  			return charsetName;
46  		} catch (IOException ex) {
47  			// TODO: log at FINE.
48  			return null;
49  		}
50  	}
51  }

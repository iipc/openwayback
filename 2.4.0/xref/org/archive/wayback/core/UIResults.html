43  /**
44   * Simple class which acts as the go-between between Java request handling code
45   * and .jsp files which actually draw various forms of results for end user
46   * consumption. Designed to be flexible enough to handle forward various types
47   * of data to the eventual .jsp files, and provides a handful of convenience
48   * method to simplify .jsp code.
49   * 
50   * 5 main "forms" of this object:
51   * 1) Generic: has WaybackRequest, uriConverter
52   * 2) Exception: has WaybackRequest, uriConverter, WaybackException
53   * 3) CaptureQuery: has WaybackRequest, uriConverter, CaptureSearchResults
54   * 4) UrlQuery: has WaybackRequest, uriConverter, UrlSearchResults
55   * 5) Replay: has WaybackRequest, uriConverter, CaptureSearchResult, 
56   * 				CaptureSearchResults, Resource
57   *
58   * There are constructors to create each of these forms from the appropriate
59   * component objects.
60   * 
61   * There is also a common method "forward()" which will store the UIResults
62   * object into an HttpServletRequest, for later retrieval by .jsp files.
63   *
64   * There are static methods to extract each of these from an HttpServletRequest,
65   * which will also verify that the appropriate internal objects are present.
66   * These methods are intended to be used by the target .jsp files.
67   * 
68   *
69   * @author brad
70   */

263 	 * Create a self-referencing URL that will perform a query for all copies
264 	 * of the given URL.
265 	 * <p>This method builds URL that passes target URL in CGI parameter,
266 	 * along with other parameters unnecessary for making simple capture
267 	 * query request. It is not suitable for simple links.
268 	 * {@link #makePlainCaptureQueryUrl(String)} generates clean and
269 	 * plain URL.</p>
270 	 * @param url to search for copies of
271 	 * @return String url that will make a query for all captures of an URL.
272 	 */
273 	public String makeCaptureQueryUrl(String url) {
274 		WaybackRequest newWBR = wbRequest.clone();
275 		
276 		newWBR.setCaptureQueryRequest();
277 		newWBR.setRequestUrl(url);
278 		return newWBR.getAccessPoint().getQueryPrefix() + "query?" +
279 			newWBR.getQueryArguments(1);
280 	}

393 	/**
394 	 * Store this UIResults object in the given HttpServletRequest, then
395 	 * forward the request to target, in this case, an image, html file, .jsp,
396 	 * any file which can return a complete document. Specifically, this means
397 	 * that if target is a .jsp, it must render it's own header and footer.
398 	 * @param request the HttpServletRequest
399 	 * @param response the HttpServletResponse
400 	 * @param target the String path to the .jsp to handle drawing the data, 
401 	 * relative to the contextRoot (ex. "/WEB-INF/query/foo.jsp")
402 	 * @throws ServletException for usual reasons...
403 	 * @throws IOException for usual reasons...
404 	 */
405 	public void forward(HttpServletRequest request,
406 			HttpServletResponse response, final String target)
407 			throws ServletException, IOException {
408 		if (target.startsWith("/WEB-INF/classes")
409 				|| target.startsWith("/WEB-INF/lib")
410 				|| target.matches("^/WEB-INF/.+\\.xml")) {
411 			throw new IOException("Not allowed");
412 		}
413 		this.contentJsp = target;
414 		this.originalRequestURL = request.getRequestURL().toString();
415 		request.setAttribute(FERRET_NAME, this);
416 		RequestDispatcher dispatcher = request.getRequestDispatcher(target);
417 		if(dispatcher == null) {
418 			throw new IOException("No dispatcher for " + target);
419 		}
420 				
421 		dispatcher.forward(request, response);
422 	}

601 	private static void replaceAll(StringBuffer s,
602 			final String o, final String n) {
603 		int olen = o.length();
604 		int nlen = n.length();
605 		int found = s.indexOf(o);
606 		while (found >= 0) {
607 			s.replace(found,found + olen,n);
608 			found = s.indexOf(o,found + nlen);
609 		}
610 	}

20  package org.archive.wayback.replay.html.transformer;
21  
22  import java.net.URL;
23  
24  import junit.framework.TestCase;
25  
26  import org.archive.wayback.replay.html.transformer.JSStringTransformerTest.RecordingReplayParseContext;
27  
28  /**
29   * unit test for {@link SrcsetStringTransformer}.
30   *
31   */
32  public class SrcsetStringTransformerTest extends TestCase {
33  
34      URL baseURL;
35      RecordingReplayParseContext rc;
36      SrcsetStringTransformer st;
37  
38      /* (non-Javadoc)
39      * @see junit.framework.TestCase#setUp()
40      */
41      protected void setUp() throws Exception {
42          baseURL = new URL("http://foo.com");
43          rc = new RecordingReplayParseContext(null, baseURL, null);
44          st = new SrcsetStringTransformer();
45          }
46  
47      public void testTransform() throws Exception {
48          final String[][] cases = new String[][] {
49              {
50                  "http://example.com/files/pic-150.jpg 150w, http://example.com/files/pic.jpg 160w",
51                  "http://example.com/files/pic-150.jpg",
52                  "http://example.com/files/pic.jpg"
53              },
54              {
55                  "examples/img/lg.jpg, examples/img/xl.jpg 2x",
56                  "examples/img/lg.jpg",
57                  "examples/img/xl.jpg"
58              }
59          };
60  
61          for (String[] c : cases) {
62              rc = new RecordingReplayParseContext(null, baseURL, null);
63              st.transform(rc, c[0]);
64              assertEquals(c[0], 2, rc.got.size());
65              assertEquals(c[0], c[1], rc.got.get(0));
66              assertEquals(c[0], c[2], rc.got.get(1));
67          }
68      }
69  
70      public void testTransformWithCommas() throws Exception {
71          final String[] c = new String[] {
72              "a,b,c,,, /images/foo1.jpg 800w,data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7 700w, /images/foo2.jpg 480w(data:,foo, ,), /images/foo3.jpg 96w(x ",
73              "a,b,c",
74              "/images/foo1.jpg",
75              "data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7",
76              "/images/foo2.jpg",
77              "/images/foo3.jpg"
78          };
79  
80          rc = new RecordingReplayParseContext(null, baseURL, null);
81          st.transform(rc, c[0]);
82          assertEquals(c[0], 5, rc.got.size());
83          assertEquals(c[0], c[1], rc.got.get(0));
84          assertEquals(c[0], c[2], rc.got.get(1));
85          assertEquals(c[0], c[3], rc.got.get(2));
86          assertEquals(c[0], c[4], rc.got.get(3));
87          assertEquals(c[0], c[5], rc.got.get(4));
88      }
89  }

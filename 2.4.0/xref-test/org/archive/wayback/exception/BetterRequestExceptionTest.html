14  /**
15   * unit test for {@link BetterRequestException}.
16   */
17  public class BetterRequestExceptionTest extends TestCase {
18  
19      /**
20       * Create a WaybackRequest for a Memento Timemap URI.
21       * @param mementoPrefix prefix to use for Memento requests
22       * @return new WaybackRequest
23       */
24      protected final WaybackRequest createMementoWaybackRequest(String mementoPrefix) {
25          AccessPoint accessPoint = new AccessPoint();
26          WaybackRequest wbRequest = new WaybackRequest();
27          wbRequest.setAccessPoint(accessPoint);
28          wbRequest.setMementoAcceptDatetime(true);
29          Properties configs = new Properties();
30          configs.setProperty(MementoConstants.AGGREGATION_PREFIX_CONFIG, mementoPrefix);
31          accessPoint.setConfigs(configs);
32          return wbRequest;
33      }
34  
35      protected final HttpServletResponse createResponseMock() {
36          HttpServletResponse response = EasyMock.createMock(HttpServletResponse.class);
37          response.setStatus(HttpServletResponse.SC_FOUND);
38          return response;
39      }
40  
41      /**
42       * Test the response location and headers are set.
43       */
44      public void testGenerateResponse() {
45          final String betterURI = "http://example.com/test";
46          final String headerName = "Vary";
47          final String headerValue = "Accept-Encoding";
48  
49          HttpServletResponse response = createResponseMock();
50          response.setHeader("Location", betterURI);
51          response.setHeader(headerName, headerValue);
52          EasyMock.replay(response);
53  
54          BetterRequestException bre = new BetterRequestException(betterURI);
55          bre.addHeader(headerName, headerValue);
56          bre.generateResponse(response, null);
57          EasyMock.verify(response);
58      }
59  
60      /**
61       * Test the redirect URI consists of all ASCII characters.
62       */
63      public void testGenerateResponseConvertsRedirectToASCII() {
64          final String nonASCII = "http://mawdoo3.com/خاص:اتصال";
65          final String uri = "http://mawdoo3.com/%D8%AE%D8%A7%D8%B5:%D8%A7%D8%AA%D8%B5%D8%A7%D9%84";
66  
67          HttpServletResponse response = createResponseMock();
68          response.setHeader("Location", uri);
69          EasyMock.replay(response);
70  
71          BetterRequestException bre = new BetterRequestException(nonASCII);
72          bre.generateResponse(response, null);
73          EasyMock.verify(response);
74      }
75  
76      /**
77       * Test a Memento redirect gets its prefix.
78       */
79      public void testGenerateResponseUsesMementoPrefix() {
80          final String betterURI = "/timemap/link/http://example.com";
81          final String mementoPrefix = "http://web.archive.org";
82  
83          HttpServletResponse response = createResponseMock();
84          response.setHeader("Location", mementoPrefix + betterURI);
85          EasyMock.replay(response);
86          WaybackRequest wbRequest = createMementoWaybackRequest(mementoPrefix);
87  
88          BetterRequestException bre = new BetterRequestException(betterURI);
89          bre.generateResponse(response, wbRequest);
90          EasyMock.verify(response);
91      }
92

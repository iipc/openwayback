80  	private Map<String, String[]> parseParameters(String url) {
81  		int iq = url.indexOf("?");
82  		if (iq < 0)
83  			return Collections.emptyMap();
84  
85  		Map<String, List<String>> map = new HashMap<String, List<String>>();
86  		String query = url.substring(iq + 1);
87  		String[] params = query.split("&");
88  		for (String kv : params) {
89  			int ieq = kv.indexOf("=");
90  			String k, v;
91  			if (ieq >= 0) {
92  				k = kv.substring(0, ieq);
93  				try {
94  					v = URLDecoder.decode(kv.substring(ieq + 1), "UTF-8");
95  				} catch (UnsupportedEncodingException ex) {
96  					v = kv.substring(ieq + 1);
97  				}
98  			} else {
99  				k = kv;
100 				v = "";
101 			}
102 			List<String> va = map.get(k);
103 			if (va == null) {
104 				map.put(k, va = new ArrayList<String>());
105 			}
106 			va.add(v);
107 		}
108 		Map<String, String[]> parameterMap = new HashMap<String, String[]>();
109 		for (Entry<String, List<String>> e : map.entrySet()) {
110 			List<String> va = e.getValue();
111 			parameterMap.put(e.getKey(), va.toArray(new String[va.size()]));
112 		}
113 		return parameterMap;
114 	}

524 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
525 			assertNotNull(wbr);
526 			assertTrue(wbr.isReplayRequest());
527 			assertFalse(wbr.isBestLatestReplayRequest());
528 			assertEquals(EXPECTED_START_TIMESTAMP, wbr.getStartTimestamp());
529 			assertEquals(EXPECTED_END_TIMESTAMP, wbr.getEndTimestamp());
530 			assertEquals(date, wbr.getReplayDate());
531 			assertEquals(date, wbr.getAnchorDate());
532 		}
533 		// invalid value in Accept-Timestamp header is silently ignored,
534 		// unless Accept-Datetime also has an invalid value.
535 		{
536 			acceptDatetimeHeader = null;
537 			acceptTimestampHeader = "*INVALID*";
538 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
539 			assertNotNull(wbr);
540 			assertTrue(wbr.isReplayRequest());
541 			assertTrue(wbr.isBestLatestReplayRequest());
542 			assertEquals(EXPECTED_START_TIMESTAMP, wbr.getStartTimestamp());
543 			assertEquals(EXPECTED_END_TIMESTAMP, wbr.getEndTimestamp());
544 		}

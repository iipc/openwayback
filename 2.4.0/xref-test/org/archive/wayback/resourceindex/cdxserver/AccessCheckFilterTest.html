22  public class AccessCheckFilterTest extends TestCase {

24  	AccessCheckFilter cut;
25  
26  	AuthToken authToken;
27  	ObjectFilter<CaptureSearchResult> adminFilter;
28  	ObjectFilter<CaptureSearchResult> robotsFilter;
29  	CDXFilter scopeFilter;

31  	boolean ignoreRobots = false;

33  	private static class MockExclusionFilter extends ExclusionFilter {
34  		ObjectFilter<CaptureSearchResult> mock;
35  		public MockExclusionFilter(ObjectFilter<CaptureSearchResult> mock) {
36  			this.mock = mock;
37  		}
38  		@Override
39  		public int filterObject(CaptureSearchResult o) {
40  			return mock.filterObject(o);
41  		}
42  	}
43  	/* (non-Javadoc)
44  	 * @see junit.framework.TestCase#setUp()
45  	 */
46  	@SuppressWarnings("unchecked")
47  	protected void setUp() throws Exception {
48  		super.setUp();
49  		authToken = new AuthToken() {
50  			public boolean isIgnoreRobots() {
51  				return ignoreRobots;
52  			};
53  		};
54  		adminFilter = EasyMock.createMock(ObjectFilter.class);
55  		robotsFilter = EasyMock.createMock(ObjectFilter.class);
56  		scopeFilter = EasyMock.createMock(CDXFilter.class);
57  
58  		cut = new AccessCheckFilter(authToken, new MockExclusionFilter(
59  			adminFilter), new MockExclusionFilter(robotsFilter), scopeFilter);
60  	}

62  	static class CheckCaptureSearchResult implements IAnswer<Integer> {
63  		String url;
64  		String urlkey;
65  		String timestamp;
66  
67  		public CheckCaptureSearchResult(String url, String urlkey, String timestamp) {
68  			this.url = url;
69  			this.urlkey = urlkey;
70  			this.timestamp = timestamp;
71  		}
72  		@Override
73  		public Integer answer() throws Throwable {
74  			CaptureSearchResult o = (CaptureSearchResult)EasyMock.getCurrentArguments()[0];
75  			assertEquals(url, o.getOriginalUrl());
76  			assertEquals(urlkey, o.getUrlKey());
77  			assertEquals(timestamp, o.getCaptureTimestamp());
78  			return ExclusionFilter.FILTER_INCLUDE;
79  		}
80  	}
81  
82  	public void testIncludeUrl() throws Exception {
83  		final String url = "http://example.com/";
84  		final String urlkey = "example.com)/";
85  
86  		EasyMock.expect(adminFilter.filterObject(EasyMock.<CaptureSearchResult>notNull())).andAnswer(
87  			new CheckCaptureSearchResult(url, urlkey, null));
88  		EasyMock.expect(robotsFilter.filterObject(EasyMock.<CaptureSearchResult>notNull())).andAnswer(
89  			new CheckCaptureSearchResult(url, urlkey, null));
90  		// expects no calls to scopeFilter
91  
92  		EasyMock.replay(adminFilter, robotsFilter, scopeFilter);
93  
94  		boolean result = cut.includeUrl(urlkey, url);
95  
96  		assertTrue(result);
97  		EasyMock.verify(adminFilter, robotsFilter, scopeFilter);
98  	}

100 	public void testIncludeCapture() throws Exception {
101 		final String url = "http://example.com/";
102 		final String urlkey = "example.com)/";
103 		final String timestamp = "20100429123456";
104 
105 		final String cdxLine = StringUtils.join(new Object[] {
106 			urlkey, timestamp, url, "text/html", 200, "DIGEST",
107 			"-", "-", 1024, 966357, "crawl-a/crawl-a-20100429000000.warc.gz"
108 		}, " ");
109 
110 		final CDXLine input = new CDXLine(cdxLine, CDXFieldConstants.CDX_ALL_NAMES);
111 
112 		EasyMock.expect(adminFilter.filterObject(EasyMock.<CaptureSearchResult>notNull())).andAnswer(
113 			new CheckCaptureSearchResult(url, urlkey, timestamp));
114 		EasyMock.expect(robotsFilter.filterObject(EasyMock.<CaptureSearchResult>notNull())).andAnswer(
115 			new CheckCaptureSearchResult(url, urlkey, timestamp));
116 		EasyMock.expect(scopeFilter.include(EasyMock.<CDXLine>notNull())).andAnswer(
117 			new IAnswer<Boolean>() {
118 				@Override
119 				public Boolean answer() throws Throwable {
120 					CDXLine line = (CDXLine)EasyMock.getCurrentArguments()[0];
121 					assertEquals(url, line.getOriginalUrl());
122 					assertEquals(urlkey, line.getUrlKey());
123 					return true;
124 				}
125 			});
126 
127 		EasyMock.replay(adminFilter, robotsFilter, scopeFilter);
128 
129 		boolean result = cut.includeCapture(input);
130 
131 		assertTrue(result);
132 		EasyMock.verify(adminFilter, robotsFilter, scopeFilter);
133 	}

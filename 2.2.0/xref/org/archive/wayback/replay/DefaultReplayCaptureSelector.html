6   import org.archive.wayback.core.CaptureSearchResults;
7   import org.archive.wayback.core.WaybackRequest;
8   import org.archive.wayback.util.Timestamp;

10  /**
11   * Default implementation of {@link ReplayCaptureSelector}.
12   * <p>
13   * This is what used to be embedded in {@code AccessPoint hadleReplay} method.
14   * It compares previous and next capture of the current one, picks closer capture
15   * time-wise. Captures are removed as it traverses the captures list, so that
16   * the same capture will not be returned twice.
17   * </p>
18   * <p>
19   * It also favors successful (200) captures over closer redirects and errors for
20   * embeds and <em>bestLatestReplayRequest</em> (see {@link DatelessReplayRequestParser}).
21   * </p>
22   * <p>
23   * New feature 2014-00-18: It also skips captures with any of ROBOT_FLAGS_SKIPPED
24   * in {@code robotflags} field. This is for new <em>soft-blocked captures</em>
25   * feature.
26   * </p>
27   * <p>
28   * For backward compatibility, this implementation delegates closest-selection (i.e.
29   * selecting the first capture to return) to {@link ReplayDispatcher} passed to
30   * its constructor. {@code ReplayCaptureSelector} is not ready for factory-based
31   * customization because of this. When {@code ReplayCaptureSelector}
32   * takes over closest-selection functionality from {@code ReplayDispatcher},
33   * dependency on {@code replayDispatcher} will be removed.
34   * </p>
35   */
36  public class DefaultReplayCaptureSelector implements ReplayCaptureSelector {
37  	/**
38  	 * captures with these flags in {@code robotflags} field are skipped.
39  	 * (TODO: make this customizable?)
40  	 */
41  	public static final String ROBOT_FLAGS_SKIPPED = "X";
42  
43  	private WaybackRequest wbRequest;
44  	private CaptureSearchResults captures;
45  	private long requestMS;
46  
47  	// Current implementation delegates closest-selection part to
48  	// ReplayDispatcher. It will eventually be removed.
49  	private ReplayDispatcher replayDispatcher;
50  
51  	private CaptureSearchResult currentClosest;
52  
53  	/**
54  	 * Initialize object with {@link ReplayDispatcher}, to which
55  	 * closest-selection is delegated.
56  	 * @param replayDispatcher {@code ReplayDispatcher}, cannot be null.
57  	 */
58  	public DefaultReplayCaptureSelector(ReplayDispatcher replayDispatcher) {
59  		this.replayDispatcher = replayDispatcher;
60  	}
61  
62  	@Override
63  	public void setRequest(WaybackRequest wbRequest) {
64  		this.wbRequest = wbRequest;
65  		requestMS = Timestamp
66  				.parseBefore(wbRequest.getReplayTimestamp()).getDate()
67  				.getTime();
68  	}
69  
70  	@Override
71  	public void setCaptures(CaptureSearchResults captures) {
72  		this.captures = captures;
73  		currentClosest = null;
74  	}
75  
76  	/**
77  	 * set {@link ReplayDispatcher} for selecting the best capture.
78  	 * @param replayDispatcher
79  	 */
80  	public void setReplayDispatcher(ReplayDispatcher replayDispatcher) {
81  		this.replayDispatcher = replayDispatcher;
82  	}
83  
84  	protected static boolean hasAnyRobotFlags(CaptureSearchResult capture, String flags) {
85  		// most capture have no robot flag - do a shortcut.
86  		if (capture.getRobotFlags() != null) {
87  			for (int i = 0; i < flags.length(); i++) {
88  				if (capture.isRobotFlagSet(flags.charAt(i)))
89  					return true;
90  			}
91  		}
92  		return false;
93  	}
94  
95  	@Override
96  	public CaptureSearchResult next() {
97  		if (currentClosest == null)
98  			currentClosest = replayDispatcher.getClosest(wbRequest,
99  				captures);
100 		else
101 			currentClosest = findNextClosest();
102 		while (currentClosest != null) {
103 			// Attempt to resolve any not-found embedded content with next-best.
104 			// For "best last" capture, skip not-founds and redirects, hoping to
105 			// find the best 200 response.
106 			if (wbRequest.isAnyEmbeddedContext() &&
107 					currentClosest.isHttpError() ||
108 					wbRequest.isBestLatestReplayRequest() &&
109 					!currentClosest.isHttpSuccess()) {
110 				CaptureSearchResult capture;
111 				while ((capture = findNextClosest()) != null) {
112 					if (capture.isHttpRedirect()) {
113 						// save redirects, but keep looking; it'll be used if no
114 						// better capture is found (caveat: picks the last, i.e. farthest,
115 						// redirect capture.)
116 						currentClosest = capture;
117 					} else if (capture.isHttpSuccess()) {
118 						currentClosest = capture;
119 						break;
120 					}
121 				}
122 			}
123 			break;
124 		}
125 		return currentClosest;
126 	}
127 
128 	protected CaptureSearchResult findNextClosest() {
129 		CaptureSearchResult prev = currentClosest.getPrevResult();
130 		CaptureSearchResult next = currentClosest.getNextResult();
131 
132 		currentClosest.removeFromList();
133 
134 		if (prev == null) {
135 			return next;
136 		} else if (next == null) {
137 			return prev;
138 		}
139 
140 		long prevMS = prev.getCaptureDate().getTime();
141 		long nextMS = next.getCaptureDate().getTime();
142 		long prevDiff = Math.abs(prevMS - requestMS);
143 		long nextDiff = Math.abs(requestMS - nextMS);
144 
145 		if (prevDiff == 0) {
146 			return prev;
147 		} else if (nextDiff == 0) {
148 			return next;
149 		}
150 
151 		String currHash = currentClosest.getDigest();
152 		String prevHash = prev.getDigest();
153 		String nextHash = next.getDigest();
154 		boolean prevSameHash = (prevHash.equals(currHash));
155 		boolean nextSameHash = (nextHash.equals(currHash));
156 
157 		if (prevSameHash != nextSameHash) {
158 			return prevSameHash ? prev : next;
159 		}
160 
161 		String prevStatus = prev.getHttpCode();
162 		String nextStatus = next.getHttpCode();
163 		boolean prev200 = (prevStatus != null) && prevStatus.equals("200");
164 		boolean next200 = (nextStatus != null) && nextStatus.equals("200");
165 
166 		// If only one is a 200, prefer the entry with the 200
167 		if (prev200 != next200) {
168 			return (prev200 ? prev : next);
169 		}
170 
171 		if (prevDiff < nextDiff) {
172 			return prev;
173 		} else {
174 			return next;
175 		}
176 	}
177 }

22  import org.archive.wayback.RequestParser;
23  import org.archive.wayback.archivalurl.requestparser.ArchivalUrlFormRequestParser;
24  import org.archive.wayback.archivalurl.requestparser.DatelessReplayRequestParser;
25  import org.archive.wayback.archivalurl.requestparser.PathDatePrefixQueryRequestParser;
26  import org.archive.wayback.archivalurl.requestparser.PathDateRangeQueryRequestParser;
27  import org.archive.wayback.archivalurl.requestparser.PathPrefixDatePrefixQueryRequestParser;
28  import org.archive.wayback.archivalurl.requestparser.PathPrefixDateRangeQueryRequestParser;
29  import org.archive.wayback.archivalurl.requestparser.ReplayRequestParser;
30  import org.archive.wayback.memento.TimeMapRequestParser;
31  import org.archive.wayback.requestparser.CompositeRequestParser;
32  import org.archive.wayback.requestparser.OpenSearchRequestParser;

34  /**
35   * CompositeRequestParser that handles Archival Url Replay and Query requests,
36   * in addition to "standard" OpenSearch and Form RequestParsers.
37   *
38   * <p>Typically, this class is set to AccessPoint to configure ArchivalUrl
39   * access point.</p>
40   *
41   * <h4>Refactoring Note</h4>
42   * <p>It doesn't make much sense to split Archival-Url request parsing
43   * into such fine-grained classes. It just make code less efficient, more difficult to maintain.
44   * It is hard to think of the situation where these sub-parsers are customized separately, and
45   * order is very important.
46   * It also sounds weird to include TimeMapRequestParser in <em>ArchivalUrlRequestParser</em>, even if
47   * TimeMapRequestParser works only in Archival-Url space.
48   * Refactor these classes into one ArchivalUrl replay/query request parser, and rename this class.
49   * Probably this class may be bundled with other ArchivalUrl-related classes for better abstraction.</p>
50   *
51   * @see org.archive.wayback.webapp.AccessPoint#getParser
52   * @see org.archive.wayback.webapp.AccessPoint#handleRequest
53   * @author brad
54   */
55  public class ArchivalUrlRequestParser extends CompositeRequestParser {
56  	// TODO: move these constants to where they are actually used.
57  	// ArchivalUrl? ReplayRequestParser?
58  	/**
59  	 * delimiter character for datespec flags
60  	 */
61  	public final static String FLAG_DELIM = "_";
62  	/**
63  	 * text/javascript context
64  	 */
65  	public final static String JS_CONTEXT = "js"; 
66  	/**
67  	 * text/css context
68  	 */
69  	public final static String CSS_CONTEXT = "cs";
70  	/**
71  	 * image/* context
72  	 */
73  	public final static String IMG_CONTEXT = "im";
74  	/**
75  	 * raw/identity context
76  	 */
77  	public final static String IDENTITY_CONTEXT = "id";
78  	/**
79  	 * frame-wrapper context
80  	 */
81  	public final static String FRAME_WRAPPED_CONTEXT = "fw";
82  	/**
83  	 * iframe-wrapped context
84  	 */
85  	public final static String IFRAME_WRAPPED_CONTEXT = "if";
86  	/**
87  	 * object/embed wrapped context
88  	 */
89  	public final static String OBJECT_EMBED_WRAPPED_CONTEXT = "oe";	
90  	/**
91  	 * Charset detection strategy context - should be followed by an integer
92  	 * indicating which strategy to use 
93  	 */
94  	public final static String CHARSET_MODE = "cm";
95  
96  	protected RequestParser[] getRequestParsers() {
97  		RequestParser[] theParsers = {
98  				new ReplayRequestParser(this),
99  				new TimeMapRequestParser(this),			
100 				new PathDatePrefixQueryRequestParser(this),
101 				new PathDateRangeQueryRequestParser(this),
102 				new PathPrefixDatePrefixQueryRequestParser(this),
103 				new PathPrefixDateRangeQueryRequestParser(this),
104 				new OpenSearchRequestParser(this),
105 				new ArchivalUrlFormRequestParser(this),
106 				new DatelessReplayRequestParser(this)
107 				};
108 		return theParsers;
109 	}
110 }

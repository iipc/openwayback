20  package org.archive.wayback.archivalurl;
21  
22  import java.util.logging.Logger;
23  
24  import org.archive.wayback.ResultURIConverter;
25  import org.archive.wayback.util.url.UrlOperations;
26  import org.archive.wayback.webapp.AccessPoint;
27  import org.archive.wayback.webapp.AccessPointAware;
28  
29  /**
30   *
31   *
32   * @author brad
33   * @version $Date$, $Revision$
34   */
35  public class ArchivalUrlResultURIConverter implements ResultURIConverter, 
36  	AccessPointAware {
37  	private static final Logger LOGGER =
38  		Logger.getLogger(ArchivalUrlResultURIConverter.class.getName());
39  	
40  	/**
41  	 * configuration name for URL prefix of replay server
42  	 */
43  	private String replayURIPrefix = null;
44  	
45  	/* (non-Javadoc)
46  	 * @see org.archive.wayback.ResultURIConverter#makeReplayURI(java.lang.String, java.lang.String)
47  	 */
48  	public String makeReplayURI(String datespec, String url) {
49  		StringBuilder sb = null;
50  
51  		if(replayURIPrefix == null) {
52  			sb = new StringBuilder(url.length() + datespec.length());
53  			sb.append(datespec);
54  			sb.append("/");
55  			sb.append(UrlOperations.stripDefaultPortFromUrl(url));
56  			return sb.toString();
57  		}
58  		if(url.startsWith(replayURIPrefix)) {
59  			return url;
60  		}
61  		sb = new StringBuilder(url.length() + datespec.length());
62  		sb.append(replayURIPrefix);
63  		sb.append(datespec);
64  		sb.append("/");
65  		sb.append(UrlOperations.stripDefaultPortFromUrl(url));
66  		return sb.toString();
67  	}
68  
69  	/**
70  	 * @param replayURIPrefix the replayURIPrefix to set
71  	 */
72  	public void setReplayURIPrefix(String replayURIPrefix) {
73  		this.replayURIPrefix = replayURIPrefix;
74  	}
75  
76  	/**
77  	 * @return the replayURIPrefix
78  	 */
79  	public String getReplayURIPrefix() {
80  		return replayURIPrefix;
81  	}
82  
83  	public void setAccessPoint(AccessPoint accessPoint) {
84  		if(replayURIPrefix == null) {
85  			String apReplayPrefix = accessPoint.getReplayPrefix(); 
86  			LOGGER.warning("No replayURIPrefix configured - using accessPoint:"
87  					+ apReplayPrefix);
88  			replayURIPrefix = apReplayPrefix;
89  		} else {
90  			LOGGER.info("replayURIPrefix is configured: " + replayURIPrefix);
91  		}
92  	}
93  }

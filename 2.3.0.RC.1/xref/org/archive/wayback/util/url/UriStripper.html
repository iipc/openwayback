6   import java.util.regex.Matcher;

8   /**
9    *
10   * @author ngiraud
11   *
12   */
13  public class UriStripper extends PatternBasedTextProcessor {
14  
15      public UriStripper() {
16          super();
17      }
18  
19      @Override
20      public String process(final String uriAsString) {
21          Matcher m = getMatcher(uriAsString);
22          StringBuffer result = new StringBuffer();
23          while (m.find()) {
24              m.appendReplacement(result, "");;
25          }
26          m.appendTail(result);
27          return result.toString();
28      }
29  
30  }

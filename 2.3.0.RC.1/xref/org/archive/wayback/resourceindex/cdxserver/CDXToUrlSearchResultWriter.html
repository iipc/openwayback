1   package org.archive.wayback.resourceindex.cdxserver;
2   
3   import org.archive.cdxserver.CDXQuery;
4   import org.archive.cdxserver.processor.GroupCountProcessor;
5   import org.archive.format.cdx.CDXLine;
6   import org.archive.wayback.core.SearchResults;
7   import org.archive.wayback.core.UrlSearchResult;
8   import org.archive.wayback.core.UrlSearchResults;
9   
10  public class CDXToUrlSearchResultWriter extends CDXToSearchResultWriter {
11  
12  	/**
13  	 * Initialize with query parameters.
14  	 * @param query CDXQuery
15  	 */
16  	public CDXToUrlSearchResultWriter(CDXQuery query) {
17  		super(query);
18  	}
19  
20  	UrlSearchResults results;
21  	int count;
22  
23  	@Override
24  	public void begin() {
25  		results = new UrlSearchResults();
26  		count = 0;
27  	}
28  
29  	@Override
30  	public int writeLine(CDXLine line) {
31  		UrlSearchResult result = new UrlSearchResult();
32  
33  		result.setUrlKey(line.getUrlKey());
34  		result.setOriginalUrl(line.getOriginalUrl());
35  		result.setFirstCapture(line.getTimestamp());
36  
37  		result.setLastCapture(line.getField(GroupCountProcessor.endtimestamp));
38  		result.setNumCaptures(line.getField(GroupCountProcessor.groupcount));
39  		result.setNumVersions(line.getField(GroupCountProcessor.uniqcount));
40  
41  		results.addSearchResult(result);
42  		++count;
43  
44  		return 1;
45  	}
46  
47  	@Override
48  	public void end() {
49  		results.setReturnedCount(count);
50  		results.setMatchingCount(count);
51  	}
52  
53  	@Override
54  	public SearchResults getSearchResults() {
55  		return results;
56  	}
57  
58  }

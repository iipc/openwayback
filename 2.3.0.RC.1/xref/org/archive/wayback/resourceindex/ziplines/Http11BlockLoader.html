35  /**
36   * Class which wraps most of the complexity of an apache commons httpclient
37   * MultiThreaderHttpConnectionManager, exposing common configuration elements
38   * to Spring configuration.
39   * 
40   * This class is a near direct copy of RemoteLiveWebCache: refactoring needed.
41   * 
42   * @author brad
43   *
44   */
45  public class Http11BlockLoader implements BlockLoader {
46  	private static final Logger LOGGER = Logger.getLogger(
47  			Http11BlockLoader.class.getName());
48  
49      private MultiThreadedHttpConnectionManager connectionManager = null;
50      private HostConfiguration hostConfiguration = null;
51      private HttpClient http = null; 
52  
53      /**
54       * 
55       */
56      public Http11BlockLoader() {
57      	connectionManager = new MultiThreadedHttpConnectionManager();
58      	hostConfiguration = new HostConfiguration();
59  		HttpClientParams params = new HttpClientParams();
60  //        params.setParameter(HttpClientParams.RETRY_HANDLER, new NoRetryHandler());
61      	http = new HttpClient(params,connectionManager);
62      	http.setHostConfiguration(hostConfiguration);
63      }
64  
65  	/**
66  	 * Fetch a range of bytes from a particular URL. Note that the bytes are
67  	 * read into memory all at once, so care should be taken with the length
68  	 * argument.
69  	 * 
70  	 * @param url String URL to fetch
71  	 * @param offset byte start offset of the desired range
72  	 * @param length number of octets to fetch
73  	 * @return a new byte[] containing the octets fetched
74  	 * @throws IOException on HTTP and Socket failures, as well as Timeouts
75  	 */
76  	public byte[] getBlock(String url, long offset, int length) 
77  	throws IOException {
78  
79  		HttpMethod method = null;
80  		try {
81  			method = new GetMethod(url);
82  		} catch(IllegalArgumentException e) {
83  			LOGGER.warning("Bad URL for block fetch:" + url);
84  			throw new IOException("Url:" + url + " does not look like an URL?");
85  		}
86  		StringBuilder sb = new StringBuilder(16);
87  		sb.append(ZiplinedBlock.BYTES_HEADER).append(offset);
88  		sb.append(ZiplinedBlock.BYTES_MINUS).append((offset + length)-1);
89  		String rangeHeader = sb.toString();
90  		method.addRequestHeader(ZiplinedBlock.RANGE_HEADER, rangeHeader);
91  		//uc.setRequestProperty(RANGE_HEADER, sb.toString());
92  		long start = System.currentTimeMillis();
93  	    try {
94  			LOGGER.fine("Reading block:" + url + "("+rangeHeader+")");
95  	    	int status = http.executeMethod(method);
96  	    	if((status == 200) || (status == 206)) {
97  	    		InputStream is = method.getResponseBodyAsStream();
98  	    		byte[] block = new byte[length];
99  	    		ByteStreams.readFully(is, block);
100 	    		long elapsed = System.currentTimeMillis() - start;
101 				PerformanceLogger.noteElapsed("CDXBlockLoad",elapsed,url);
102 	    		return block;
103 	    		
104 	    	} else {
105 	    		throw new IOException("Bad status for " + url);
106 	    	}
107 		} finally {
108 	    	method.releaseConnection();
109 	    }
110 	}

112     /**
113      * @param hostPort to proxy requests through - ex. "localhost:3128"
114      */
115     public void setProxyHostPort(String hostPort) {
116     	int colonIdx = hostPort.indexOf(':');
117     	if(colonIdx > 0) {
118     		String host = hostPort.substring(0,colonIdx);
119     		int port = Integer.valueOf(hostPort.substring(colonIdx+1));
120     		
121     		hostConfiguration.setProxy(host, port);
122     	}
123     }

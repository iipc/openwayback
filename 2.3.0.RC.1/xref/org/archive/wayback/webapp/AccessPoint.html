204 			HttpServletResponse httpResponse) throws ServletException,
205 			IOException {
206 		if (LOGGER.isLoggable(Level.FINE)) {
207 			LOGGER.fine("Local dispatch /" + translateRequestPath(httpRequest));
208 		}
209 		if (!serveStatic) {
210 			return false;
211 		}
212 //		String contextRelativePath = httpRequest.getServletPath();
213 		String translatedNoQuery = "/" + translateRequestPath(httpRequest);
214 //		String absPath = getServletContext().getRealPath(contextRelativePath);
215 		String absPath = getServletContext().getRealPath(translatedNoQuery);
216 
217 		if (this.isEnableMemento()) {
218 			MementoUtils.addDoNotNegotiateHeader(httpResponse);
219 		}
220 
221 		//IK: added null check for absPath, it may be null (ex. on jetty)
222 		if (absPath != null) {
223 			File test = new File(absPath);
224 			if((test != null) && !test.exists()) {
225 				return false;
226 			}
227 		}
228 
229 		String translatedQ = "/" + translateRequestPathQuery(httpRequest);
230 
231 		WaybackRequest wbRequest = new WaybackRequest();
232 //			wbRequest.setContextPrefix(getUrlRoot());
233 		wbRequest.setAccessPoint(this);
234 		wbRequest.extractHttpRequestInfo(httpRequest);
235 		UIResults uiResults = new UIResults(wbRequest,uriConverter);
236 		try {
237 			uiResults.forward(httpRequest, httpResponse, translatedQ);
238 			return true;
239 		} catch(IOException e) {
240 			// TODO: figure out if we got IO because of a missing dispatcher
241 		}
242 
243 		return false;
244 	}

247 	 * @param httpRequest HttpServletRequest which is being handled
248 	 * @param httpResponse HttpServletResponse which is being handled
249 	 * @return true if the request was actually handled
250 	 * @throws ServletException per usual
251 	 * @throws IOException per usual
252 	 */
253 	public boolean handleRequest(HttpServletRequest httpRequest,
254 			HttpServletResponse httpResponse) throws ServletException,
255 			IOException {
256 
257 		WaybackRequest wbRequest = null;
258 		boolean handled = false;
259 
260 		try {
261 			PerfStats.clearAll();
262 
263 			if (this.isEnablePerfStatsHeader() && (perfStatsHeader != null)) {
264 				PerfStats.timeStart(PerfStat.Total);
265 				httpResponse = new PerfWritingHttpServletResponse(httpRequest,
266 					httpResponse, PerfStat.Total, perfStatsHeader,
267 					perfStatsHeaderFormat);
268 			}
269 
270 			String inputPath = translateRequestPathQuery(httpRequest);
271 			Thread.currentThread().setName("Thread " +
272 					Thread.currentThread().getId() + " " + getBeanName() +
273 					" handling: " + inputPath);
274 			LOGGER.fine("Handling translated: " + inputPath);
275 			wbRequest = getParser().parse(httpRequest, this);
276 
277 			if (wbRequest != null) {
278 				handled = true;
279 
280 				// TODO: refactor this code into RequestParser implementations
281 				wbRequest.setAccessPoint(this);
282 //				wbRequest.setContextPrefix(getAbsoluteLocalPrefix(httpRequest));
283 //				wbRequest.setContextPrefix(getUrlRoot());
284 				wbRequest.extractHttpRequestInfo(httpRequest);
285 				// end of refactor
286 
287 				if (getAuthentication() != null) {
288 					if (!getAuthentication().isTrue(wbRequest)) {
289 						throw new AuthenticationControlException(
290 							"Unauthorized", isRequestAuth());
291 					}
292 				}
293 
294 				// set exclusionFilter on wbRequest only if not set externally
295 				if (wbRequest.getExclusionFilter() == null) {
296 					wbRequest.setExclusionFilter(createExclusionFilter());
297 				}
298 
299 				// TODO: refactor this into RequestParser implementations, so a
300 				// user could alter requests to change the behavior within a
301 				// single AccessPoint. For now, this is a simple way to expose
302 				// the feature to configuration.g
303 				wbRequest.setExactScheme(isExactSchemeMatch());
304 
305 				if (wbRequest.isReplayRequest()) {
306 					if (bounceToReplayPrefix) {
307 						// we don't accept replay requests on this AccessPoint
308 						// bounce the user to the right place:
309 						String suffix = translateRequestPathQuery(httpRequest);
310 						String replayUrl = replayPrefix + suffix;
311 						httpResponse.sendRedirect(replayUrl);
312 						return true;
313 					}
314 					handleReplay(wbRequest, httpRequest, httpResponse);
315 				} else {
316 					if (bounceToQueryPrefix) {
317 						// we don't accept replay requests on this AccessPoint
318 						// bounce the user to the right place:
319 						String suffix = translateRequestPathQuery(httpRequest);
320 						String replayUrl = queryPrefix + suffix;
321 						httpResponse.sendRedirect(replayUrl);
322 						return true;
323 					}
324 					wbRequest.setExactHost(isExactHostMatch());
325 					handleQuery(wbRequest, httpRequest, httpResponse);
326 				}
327 			} else {
328 				handled = dispatchLocal(httpRequest, httpResponse);
329 			}
330 		} catch (BetterRequestException e) {
331 			e.generateResponse(httpResponse, wbRequest);
332 			httpResponse.getWriter(); // cause perf headers to be committed
333 			handled = true;
334 		} catch (WaybackException e) {
335 			if (httpResponse.isCommitted()) {
336 				return true;
337 			}
338 
339 			if (wbRequest == null) {
340 				wbRequest = new WaybackRequest();
341 				wbRequest.setAccessPoint(this);
342 			}
343 
344 			logError(httpResponse, errorMsgHeader, e, wbRequest);
345 
346 			LiveWebState liveWebState = LiveWebState.NOT_FOUND;
347 
348 			if ((getLiveWebRedirector() != null) &&
349 					!wbRequest.hasMementoAcceptDatetime() && !wbRequest.isMementoTimemapRequest()) {
350 				liveWebState = getLiveWebRedirector().handleRedirect(e, wbRequest, httpRequest, httpResponse);
351 			}
352 
353 			// If not liveweb redirected, then render current exception
354 			if (liveWebState != LiveWebState.REDIRECTED) {
355 				e.setLiveWebAvailable(liveWebState == LiveWebState.FOUND);
356 				getException().renderException(httpRequest, httpResponse, wbRequest, e, getUriConverter());
357 			}
358 
359 			handled = true;
360 
361 		} catch (Exception other) {
362 			logError(httpResponse, errorMsgHeader, other, wbRequest);
363 		} finally {
364 			//Slightly hacky, but ensures that all block loaders are closed
365 			ZipNumBlockLoader.closeAllReaders();
366 		}
367 
368 		return handled;
369 	}

426 	public void logError(HttpServletResponse httpResponse, String header,
427 			Exception e, WaybackRequest request) {
428 		if (e instanceof ResourceNotInArchiveException) {
429 			if (LOGGER.isLoggable(Level.INFO)) {
430 				this.logNotInArchive((ResourceNotInArchiveException)e, request);
431 			}
432 		} else if (e instanceof AccessControlException) {
433 			// While StaticMapExclusionFilter#isExcluded(String) reports
434 			// exclusion at INFO level, RobotExclusionFilter logs exclusion
435 			// at FINE level only. I believe here is the better place to log
436 			// exclusion. Unfortunately, AccessControlException has no
437 			// detailed info (TODO). we don't need a stack trace.
438 			if (LOGGER.isLoggable(Level.INFO)) {
439 				LOGGER.log(Level.INFO, "Access Blocked:" + request.getRequestUrl() + ": "+ e.getMessage());
440 			}
441 		} else {
442 			if (LOGGER.isLoggable(Level.WARNING)) {
443 				LOGGER.log(Level.WARNING, "Runtime Error", e);
444 			}
445 		}
446 
447 		if (!this.isEnableErrorMsgHeader()) {
448 			return;
449 		}
450 
451 		String message = (e != null ? e.toString() : "");
452 
453 		if (message == null) {
454 			message = "";
455 		} else {
456 			// Get substring from exception name
457 			int index = message.indexOf(':');
458 			if (index > 0) {
459 				index = message.lastIndexOf('.', index);
460 				if (index > 0) {
461 					message = message.substring(index + 1);
462 				}
463 			}
464 
465 			if (message.length() > MAX_ERR_HEADER_LEN) {
466 				message = message.substring(0, MAX_ERR_HEADER_LEN);
467 			}
468 			message = message.replace('\n', ' ');
469 		}
470 
471 		httpResponse.setHeader(header, message);
472 	}

508 				u.append(interstitialJsp);

514 				u.append("&");
515 				u.append(INTERSTITIAL_URL).append("=");

517 					u.append(URLEncoder.encode(wbRequest.getRequestUrl(),
518 						"UTF-8"));
519 				} catch (UnsupportedEncodingException e) {

522 				}
523 				u.append("&");
524 				u.append(INTERSTITIAL_TARGET).append("=");

531

537 	protected boolean isSelfRedirect(Resource resource,
538 			CaptureSearchResult closest, WaybackRequest wbRequest,
539 			String canonRequestURL) {
540 		int status = resource.getStatusCode();
541 
542 		// Only applies to redirects
543 		if ((status < 300) || (status >= 400)) {
544 			return false;
545 		}
546 
547 		String location = resource.getHeader("Location");
548 
549 		if (location == null) {
550 			return false;
551 		}
552 
553 //		if (!closest.getCaptureTimestamp().equals(wbRequest.getReplayTimestamp())) {
554 //			return false;
555 //		}
556 
557 		String redirScheme = UrlOperations.urlToScheme(location);
558 
559 		try {
560 			if (redirScheme == null && isExactSchemeMatch()) {
561 				location = UrlOperations.resolveUrl(closest.getOriginalUrl(), location);
562 				redirScheme = UrlOperations.urlToScheme(location);
563 			} else if (location.startsWith("/")) {
564 				location = UrlOperations.resolveUrl(closest.getOriginalUrl(), location);
565 			}
566 
567 			if (getSelfRedirectCanonicalizer() != null) {
568 				location = getSelfRedirectCanonicalizer().urlStringToKey(location);
569 			}
570 		} catch (IOException e) {
571 			return false;
572 		}
573 
574 		if (location.equals(canonRequestURL)) {
575 			// if not exact scheme, don't do scheme compare, must be equal
576 			if (!isExactSchemeMatch()) {
577 				return true;
578 			}
579 
580 			String origScheme = UrlOperations.urlToScheme(wbRequest
581 				.getRequestUrl());
582 
583 			if ((origScheme != null) && (redirScheme != null) &&
584 					(origScheme.compareTo(redirScheme) == 0)) {
585 				return true;
586 			}
587 		}
588 
589 		return false;
590 	}

1446 				return getQueryPrefix() + startdate + "*/" + url;
1447 			}
1448 		} else {

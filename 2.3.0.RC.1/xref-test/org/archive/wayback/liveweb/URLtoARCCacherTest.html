16  public class URLtoARCCacherTest extends TestCase {
17  	public void testSocketTimeout() throws Exception {
18  	    MultiThreadedHttpConnectionManager connectionManager = null;
19  	    HostConfiguration hostConfiguration = null;
20  	    HttpClient http = null;
21      	connectionManager = new MultiThreadedHttpConnectionManager();
22      	hostConfiguration = new HostConfiguration();
23      	http = new HttpClient(connectionManager);
24      	http.setHostConfiguration(hostConfiguration);
25  		HttpMethod method = null;
26  //		String urlString = "http://wayback.archive-it.org:6100/one";
27  		String urlString = "http://netpreserve.org/";
28  		int socketTimeoutMS = 100;
29  		int connectTimeoutMS = 200;
30      	connectionManager.getParams().setSoTimeout(socketTimeoutMS);
31      	connectionManager.getParams().setConnectionTimeout(connectTimeoutMS);
32  		try {
33  			method = new GetMethod(urlString);
34  		} catch(IllegalArgumentException e) {
35  			throw new LiveDocumentNotAvailableException("Url:" + urlString +
36  					"does not look like an URL?");
37  		}
38  	    try {
39  	    	int status = http.executeMethod(method);
40  	    	System.out.println("Got response code: " + status);
41  	    	ByteOp.copyStream(method.getResponseBodyAsStream(), System.out);
42  	    } catch (SocketTimeoutException e) {
43  	    	// OK
44  	    } catch (Exception e) {
45  	    	e.printStackTrace();
46  	    }
47  
48  	}
49  }

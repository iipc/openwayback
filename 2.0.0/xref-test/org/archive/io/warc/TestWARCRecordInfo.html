33      public TestWARCRecordInfo(byte[] content) {
34          this.type = WARCRecordType.response;
35          this.url = "http://test.example.com/";
36          this.mimetype = "application/http; msgtype=response";
37          try {
38              this.recordId = new URI("uri:recordidentifier");
39          } catch (URISyntaxException ex) {
40              throw new RuntimeException("unexpected error", ex);
41          }
42          this.contentStream = new ByteArrayInputStream(content);
43          this.contentLength = content.length;
44          // NB: create14DigitDate must be in ISOZ format (name "14DigitDate" is confusing)
45          this.create14DigitDate = DateUtils.getLog14Date();
46      }

48      /**
49       * translates DT14 (YYYYmmddHHMMSS) to ISOZ format used in WARC-Date header.
50       * @param dt14
51       * @return ISOZ (YYYY-mm-ddTHH:MM:SSZ)
52       * @exception IOException dt14 is in bad format (wrapping ParseException to simply error handling)
53       */
54      public static String dt14ToISOZ(String dt14) throws IOException {
55          try {
56              Date date = DateUtils.parse14DigitDate(dt14);
57              return DateUtils.getLog14Date(date);
58          } catch (ParseException ex) {
59              throw new IOException("invalid DT14 " + dt14, ex);
60          }
61      }
62      /**
63       * utility method for updating create14DigitDate from DT14.
64       * @param dt14 DT14 (YYYYmmddHHMMSS)
65       * @throws IOException dt14 is in bad format.
66       */
67      public void setCreate14DigitDateFromDT14(String dt14) throws IOException {
68          create14DigitDate = dt14ToISOZ(dt14);
69      }

133     private static void writeChunked(OutputStream out, byte[] data) throws IOException {
134         int s = 0;
135         while (s < data.length) {
136             int n = data.length - s;
137             if (n > 0x1000) n = 0x1000;
138             out.write(String.format("%x" + CRLF, n).getBytes("UTF-8"));
139             out.write(data, s, n);
140             out.write(CRLF.getBytes("UTF-8"));
141             s += n;
142         }
143         out.write(("0" + CRLF + CRLF).getBytes("UTF-8"));
144     }
145     /**
146      * return content-block bytes for HTTP response.
147      * @param status HTTP status code and status text separated by a space. ex. {@code "200 OK"}.
148      * @param ctype HTTP Content-Type
149      * @param payloadBytes payload bytes
150      * @param chunked if true, use chunked transfer-encoding
151      * @return content-block bytes with HTTP status line, HTTP headers and payload.
152      * @throws IOException
153      */
154     public static byte[] buildHttpResponseBlock(String status, String ctype, byte[] payloadBytes, boolean chunked)
155             throws IOException {
156         ByteArrayOutputStream blockbuf = new ByteArrayOutputStream();
157         Writer bw = new OutputStreamWriter(blockbuf);
158         bw.write("HTTP/1.0 " + status + CRLF);
159         if (chunked) {
160             bw.write("Transfer-Encoding: chunked" + CRLF);
161         } else {
162             bw.write("Content-Length: " + payloadBytes.length + CRLF);
163         }
164         bw.write("Content-Type: " + ctype + CRLF);
165         bw.write(CRLF);
166         bw.flush();
167         if (chunked) {
168             writeChunked(blockbuf, payloadBytes);
169         } else {
170             blockbuf.write(payloadBytes);
171         }
172         bw.close();
173         return blockbuf.toByteArray();
174     }

179     public static byte[] buildHttpRedirectResponseBlock(String location) throws IOException {
180         ByteArrayOutputStream blockbuf = new ByteArrayOutputStream();
181         Writer bw = new OutputStreamWriter(blockbuf);
182         String status = "302 Moved Temporarily";
183         bw.write("HTTP/1.0 " + status + CRLF);
184         bw.write("Content-Length: " + 0 + CRLF);
185         bw.write("Content-Type: text/html" + CRLF);
186         bw.write("Location: " + location + CRLF);
187         bw.write(CRLF);
188         bw.close();
189         return blockbuf.toByteArray();
190     }

192     public static byte[] buildCompressedHttpResponseBlock(String ctype,
193             byte[] payloadBytes, boolean chunked) throws IOException {
194         ByteArrayOutputStream gzippedPayloadBytes = new ByteArrayOutputStream();
195         GZIPOutputStream zout = new GZIPOutputStream(gzippedPayloadBytes);
196         zout.write(payloadBytes);
197         zout.close();
198         payloadBytes = gzippedPayloadBytes.toByteArray();
199         ByteArrayOutputStream blockbuf = new ByteArrayOutputStream();
200         Writer bw = new OutputStreamWriter(blockbuf);
201         bw.write("HTTP/1.0 200 OK" + CRLF);
202         if (chunked) {
203             bw.write("Transfer-Encoding: chunked" + CRLF);
204         } else {
205             bw.write("Content-Length: " + payloadBytes.length + CRLF);
206         }
207         bw.write("Content-Type: " + ctype + CRLF);
208         bw.write("Content-Encoding: gzip" + CRLF);
209         bw.write(CRLF);
210         bw.flush();
211         if (chunked) {
212             writeChunked(blockbuf, payloadBytes);
213         } else {
214             blockbuf.write(payloadBytes);
215         }
216         bw.close();
217         return blockbuf.toByteArray();
218     }

223     /**
224      * generates WARC content for new revisit record.
225      * @param ctype value for Content-Type
226      * @param len value for Content-Length
227      * @param withHeader include HTTP status line and headers.
228      *      passing false generates old-style revisit content block.
229      * @param gzipContent if true, block will have "Content-Encoding: gzip" header.
230      *      (this shall match the compress-ness of previous capture).
231      * @return record content as byte array
232      * @throws IOException
233      */
234     public static byte[] buildRevisitHttpResponseBlock(String ctype, int len,
235             boolean withHeader, boolean gzipContent) throws IOException {
236         ByteArrayOutputStream blockbuf = new ByteArrayOutputStream();
237         Writer bw = new OutputStreamWriter(blockbuf);
238         if (withHeader) {
239             bw.write("HTTP/1.0 200 OK" + CRLF);
240             bw.write("Content-Length: " + len + CRLF);
241             bw.write("Content-Type: " + ctype + CRLF);
242             if (gzipContent)
243                 bw.write("Content-Encoding: gzip" + CRLF);
244             bw.write(CRLF);
245             bw.flush();
246             bw.close();
247         }
248         return blockbuf.toByteArray();
249     }

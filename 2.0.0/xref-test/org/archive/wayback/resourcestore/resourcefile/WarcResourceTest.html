41      /**
42       * plain HTTP response (without any transfer/content-encoding)
43       * @throws Exception
44       */
45      public void testPlainHttpRecord() throws Exception {
46          String payload = "hogehogehogehogehoge";
47          WARCRecordInfo recinfo = TestWARCRecordInfo.createHttpResponse(payload);
48          TestWARCReader ar = new TestWARCReader(recinfo);
49          WARCRecord rec = ar.get(0);
50          WarcResource res = new WarcResource(rec, ar);
51          res.parseHeaders();
52          
53          assertEquals("statusCode", 200, res.getStatusCode());
54          assertEquals("content-type", "text/plain", res.getHeader("Content-Type"));
55          byte[] buf = new byte[payload.getBytes().length + 1];
56          int n = res.read(buf);
57          assertEquals("content length", buf.length - 1, n);
58          
59          res.close();
60      }

143     /**
144      * metadata record with render-able content like site screenshot image.
145      * 
146      * HTTP status is assumed to be 200, and Content-Type of WARC header
147      * becomes Content-Type of replay response.
148      * @throws Exception
149      */
150     public void testMetadataRecord() throws Exception {
151         // 1-dot transparent GIF found everywhere if you wonder :-)
152         final byte[] block = new byte[] {
153                 71, 73, 70, 56, 57, 97, 1, 0, 1, 0, -128, 0, 0, -64, -64, -64,
154                 0, 0, 0, 33, -7, 4, 1, 0, 0, 0, 0, 44, 0, 0, 0, 0,
155                 1, 0, 1, 0, 0, 2, 2, 68, 1, 0, 59, 13, 10, 13, 10
156         };
157         final String ct = "image/gif";
158         WARCRecordInfo recinfo = new TestWARCRecordInfo(block);
159         recinfo.setType(WARCRecordType.metadata);
160         recinfo.setMimetype(ct);
161         TestWARCReader ar = new TestWARCReader(recinfo);
162         WARCRecord rec = ar.get(0);
163         WarcResource res = new WarcResource(rec, ar);
164         // must not fail
165         res.parseHeaders();
166         
167         // should return assumed 200
168         assertEquals("statusCode", 200, res.getStatusCode());
169         // content-type is what's specified in WARC header.
170         assertEquals("content-type", ct, res.getHeader("Content-Type"));
171         // must have Date header, in HTTP Date format.
172         String date = res.getHeader("Date");
173         assertNotNull("has date header", date);
174         new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss z", Locale.ENGLISH).parse(date);
175         
176         // block as content
177         byte[] buf = new byte[block.length + 1];
178         int n = res.read(buf);
179         assertEquals("content length", block.length, n);
180         for (int i = 0; i < block.length; i++) {
181             assertEquals("byte " + i, block[i], buf[i]);
182         }
183         
184         res.close();
185     }

189     /**
190      * new, current revisit record, which has just HTTP response line and
191      * headers part of the capture.

258     /**
259      * resource record, typically used for archiving ftp fetches.
260      * @throws Exception
261      */
262     public void testResourceRecord() throws Exception {
263         final String ct = "text/plain";
264         final byte[] block = "blahblahblah\n".getBytes();
265         WARCRecordInfo recinfo = new TestWARCRecordInfo(block);
266         recinfo.setType(WARCRecordType.resource);
267         recinfo.setUrl("ftp://ftp.example.com/afile.txt");
268         recinfo.setMimetype(ct);
269         ArchiveReader ar = new TestWARCReader(recinfo);
270         WARCRecord rec = (WARCRecord)ar.get(0);
271         WarcResource res = new WarcResource(rec, ar);
272         res.parseHeaders();
273                 
274         int scode = res.getStatusCode();
275         assertEquals("statusCode", 200, scode);
276 
277         Map<String, String> headers = res.getHttpHeaders();
278         assertNotNull("headers", headers);
279         
280         assertEquals("content-type", ct, res.getHeader("Content-Type"));
281 
282         // must have Date header, in HTTP Date format.
283         String date = res.getHeader("Date");
284         assertNotNull("has date header", date);
285         new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss z", Locale.ENGLISH).parse(date);
286         
287         res.close();
288     }

20  package org.archive.wayback.replay.html.transformer;
21  
22  import java.net.URL;
23  
24  import org.archive.wayback.replay.html.transformer.JSStringTransformerTest.RecordingReplayParseContext;
25  
26  import junit.framework.TestCase;
27  
28  /**
29   * @author brad
30   * @author kenji
31   */
32  public class MetaRefreshUrlStringTransformerTest extends TestCase {
33  
34  	URL baseURL;
35  	RecordingReplayParseContext rc;
36  	MetaRefreshUrlStringTransformer st;
37  
38  	@Override
39  	protected void setUp() throws Exception {
40  		baseURL = new URL("http://foo.com/");
41  		rc = new RecordingReplayParseContext(null, baseURL, null);
42  		st = new MetaRefreshUrlStringTransformer();
43  	}
44  
45  	public void testTransformFull() throws Exception {
46  		final String input = "0; URL=https://www.example.com/content";
47  		st.transform(rc, input);
48  		
49  		assertEquals(1, rc.got.size());
50  		assertEquals("https://www.example.com/content", rc.got.get(0));
51  	}
52  
53  	public void testTransformVariations() throws Exception {
54  		final String[][] cases = new String[][] {
55  				{ "0; url=/bar", "/bar" },
56  				{ "10; url =/bar", "/bar" },
57  				{ "2;url=/bar", "/bar" },
58  				// do we want to allow this?
59  				//{ "; url=/bar", "/bar" },
60  				{ "0; URL=/bar", "/bar" },
61  		};
62  		for (String[] c : cases) {
63  			rc = new RecordingReplayParseContext(null, baseURL, null);
64  			st.transform(rc, c[0]);
65  			assertEquals(c[0], 1, rc.got.size());
66  			assertEquals(c[0], c[1], rc.got.get(0));
67  		}
68  	}
69  
70  	public void testRewriteHttpsOnly() throws Exception {
71  		rc.setRewriteHttpsOnly(true);
72  
73  		final String input = "0; URL=https://www.example.com/content";
74  		st.transform(rc, input);
75  
76  		assertEquals(1, rc.got.size());
77  		assertEquals("https://www.example.com/content", rc.got.get(0));
78  	}
79  }

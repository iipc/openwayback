39  	 * fixture CDXServer (unnecessary if CDServer was an interface).
40  	 * <p>
41  	 * Note: {@code testHandleRequest} and {@code testRenderMementoTimemap} uses
42  	 * {@link CDXServer#getCdx(HttpServletRequest, HttpServletResponse, CDXQuery)},
43  	 * which eventually calls {@link #getCdx(CDXQuery, AuthToken, CDXWriter)} here.
44  	 * </p>
45  	 */
46  	public static class TestCDXServer extends CDXServer {
47  		public List<Object[]> capturedArgs = new ArrayList<Object[]>();
48  		public CDXLine[] cdxLines;
49  		
50  		@Override
51  		public void getCdx(CDXQuery query, AuthToken authToken,
52  				CDXWriter responseWriter) throws IOException {
53  			capturedArgs.add(new Object[] { query, authToken, responseWriter });
54  			
55  			responseWriter.begin();
56  			for (CDXLine cdxLine : cdxLines) {
57  				responseWriter.writeLine(cdxLine);
58  			}
59  			responseWriter.end();
60  		}
61  		
62  		public void clearCapturedArgs() {
63  			capturedArgs.clear();
64  		}
65  	}

87  	 * capture search. basic options.
88  	 * @throws Exception
89  	 */
90  	public void testQuery() throws Exception {
91  		WaybackRequest wbr = new WaybackRequest();
92  		wbr.setRequestUrl("http://example.com/");
93  		wbr.setCaptureQueryRequest();
94  		
95  		// urlkey, timestamp, original, mimetype, statuscode, digest, redirect, robotflags,
96  		// length, offset, filename.
97  		FieldSplitFormat fmt = CDXFieldConstants.CDX_ALL_NAMES;
98  		testCDXServer.cdxLines = new CDXLine[] {
99  				new CDXLine(CDXLINE1, fmt)
100 		};
101 		
102 		SearchResults sr = cut.query(wbr);
103 		
104 		assertEquals(1,  sr.getReturnedCount());
105 		
106 		assertEquals(1, testCDXServer.capturedArgs.size());
107 		
108 		Object[] args = testCDXServer.capturedArgs.get(0);
109 		CDXQuery query = (CDXQuery)args[0];
110 		String[] filter = query.getFilter();
111 		assertEquals(1, filter.length);
112 		assertEquals("!statuscode:(500|502|504)", filter[0]);
113 		
114 		AuthToken authToken = (AuthToken)args[1];
115 		assertFalse(authToken.isIgnoreRobots());
116 	}

206 	/**
207 	 * {@link EmbeddedCDXServerIndex#handleRequest(HttpServletRequest, HttpServletResponse)} is
208 	 * a entry point for CDXServer API. It should return all accessible cdx lines, without applying
209 	 * any additional filters not requested by API user.
210 	 * @throws Exception
211 	 */
212 	public void testHandleRequest() throws Exception {
213 		HttpServletRequest request = EasyMock.createNiceMock(HttpServletRequest.class);
214 		EasyMock.expect(request.getParameter("url")).andStubReturn("http://example.com/");
215 		
216 		HttpServletResponse response = EasyMock.createNiceMock(HttpServletResponse.class);
217 		StringWriter sw = new StringWriter();
218 		EasyMock.expect(response.getWriter()).andReturn(new PrintWriter(sw));
219 		
220 		FieldSplitFormat fmt = CDXFieldConstants.CDX_ALL_NAMES;
221 		testCDXServer.cdxLines = new CDXLine[] {
222 				new CDXLine(CDXLINE1, fmt)
223 		};
224 
225 		EasyMock.replay(request, response);
226 		cut.handleRequest(request, response);
227 		
228 		assertEquals(1, testCDXServer.capturedArgs.size());
229 		Object[] args = testCDXServer.capturedArgs.get(0);
230 		
231 		CDXQuery query = (CDXQuery)args[0];
232 		assertEquals("API query should not have filter by default", 0, query.getFilter().length);
233 		
234 		assertEquals(CDXLINE1+"\n", sw.toString());
235 	}

237 	/**
238 	 * {@link EmbeddedCDXServerIndex#renderMementoTimemap(WaybackRequest, HttpServletRequest, HttpServletResponse)}
239 	 * is a CDXServer API entry point for Memento format output.
240 	 * @throws Exception
241 	 */
242 	public void testRenderMementoTimemap() throws Exception {
243 		HttpServletRequest request = EasyMock.createNiceMock(HttpServletRequest.class);
244 		// Used in MementoLinkWriter
245 		EasyMock.expect(request.getRequestURL()).andAnswer(new IAnswer<StringBuffer>() {
246 			@Override
247 			public StringBuffer answer() throws Throwable {
248 				return new StringBuffer("/timemap/memento/http://example.com/");
249 			}
250 		});
251 		HttpServletResponse response = EasyMock.createNiceMock(HttpServletResponse.class);
252 		StringWriter sw = new StringWriter();
253 		EasyMock.expect(response.getWriter()).andReturn(new PrintWriter(sw));
254 		
255 		// needs: 
256 		//   getMementoTimemapFormat() - passed to CDXQuery.output
257 		//   getRequestUrl() - passed to CDXQuery
258 		//   get(MementoConstants.PAGE_STARTS) (optional, passed to CDXQuery.from
259 		//   getAccessPoint() - if getMementoTimemapFormat() == MementoConstants.FORMAT_LINK,
260 		//     CDX is looked up by calling AccessPoint#queryIndex(WaybackRequest)
261 		WaybackRequest wbr = new WaybackRequest();
262 		wbr.setRequestUrl("http://example.com/");
263 		wbr.setMementoTimemapFormat("memento");
264 		
265 		FieldSplitFormat fmt = CDXFieldConstants.CDX_ALL_NAMES;
266 		testCDXServer.cdxLines = new CDXLine[] {
267 				new CDXLine(CDXLINE1, fmt)
268 		};
269 		
270 		EasyMock.replay(request, response);
271 		boolean r = cut.renderMementoTimemap(wbr, request, response);
272 		
273 		assertTrue("renderMementoTimemap returns true", r);
274 
275 		assertEquals(1, testCDXServer.capturedArgs.size());
276 		Object[] args = testCDXServer.capturedArgs.get(0);
277 		
278 		CDXQuery query = (CDXQuery)args[0];
279 		assertEquals("API query should not have filter by default", 0, query.getFilter().length);
280 		
281 		// Here we only check if output *looks like* Memento format. Detailed tests
282 		// shall be done by test case for MementoLinkWriter.
283 		//System.out.println("response=" + sw.toString());
284 		assertTrue(sw.toString().startsWith("<http://example.com/>;"));
285 	}

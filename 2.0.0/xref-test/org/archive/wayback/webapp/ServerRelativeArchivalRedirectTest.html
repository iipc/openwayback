11  /**
12   * test for {@link ServerRelativeArchivalRedirect}.
13   */
14  public class ServerRelativeArchivalRedirectTest extends TestCase {
15  
16  	ServerRelativeArchivalRedirect cut;
17  	HttpServletRequest request;
18  
19  	/*
20  	 * (non-Javadoc)
21  	 *
22  	 * @see junit.framework.TestCase#setUp()
23  	 */
24  	protected void setUp() throws Exception {
25  		super.setUp();
26  		cut = new ServerRelativeArchivalRedirect();
27  		// in current deployment matchHost is not set (=null)
28  		cut.setMatchPort(80);
29  		cut.setUseCollection(true);
30  		cut.setReplayPrefix("/web/");
31  	}
32  
33  	protected void setUpRequest(String requestURI, String referer) {
34  		// to catch common mistake of including protocol & netloc in requestURI
35  		assert requestURI.startsWith("/");
36  
37  		request = EasyMock.createNiceMock(HttpServletRequest.class);
38  		EasyMock.expect(request.getHeader("Referer")).andStubReturn(referer);
39  		EasyMock.expect(request.getServerName()).andStubReturn(
40  			"web.archive.org");
41  		EasyMock.expect(request.getLocalPort()).andStubReturn(80);
42  		EasyMock.expect(request.getRequestURI()).andStubReturn(requestURI);
43  		EasyMock.expect(request.getContextPath()).andStubReturn("");
44  	}
45  
46  	/**
47  	 * test clean case with well-formed target URL in {@code Referer}.
48  	 */
49  	public void testRegular() throws Exception {
50  
51  		final String TARGET_URL = "http://example.com/index.html";
52  		final String REPLAY_BASE = "http://web.archive.org/web/20010203040506/";
53  		final String REFERER = REPLAY_BASE + TARGET_URL;
54  		final String REQUEST_URI = "/js/s_code.js";
55  
56  		final String EXPECTED_REDIRECT_URL = REPLAY_BASE +
57  				"http://example.com/js/s_code.js";
58  
59  		// mocks
60  		setUpRequest(REQUEST_URI, REFERER);
61  
62  		HttpServletResponse response = EasyMock
63  			.createMock(HttpServletResponse.class);
64  		response.addHeader("Vary", "Referer");
65  		EasyMock.expectLastCall().once();
66  		response.sendRedirect(EXPECTED_REDIRECT_URL);
67  		EasyMock.expectLastCall().once();
68  
69  		EasyMock.replay(request, response);
70  
71  		boolean handled = cut.handleRequest(request, response);
72  
73  		EasyMock.verify(response);
74  		assertTrue(handled);
75  	}
76  
77  	/**
78  	 * {@link WaybackRequest#setRequestUrl(String)} fix up malformed target URL
79  	 * like {@code http:/example.com}. So replay page with such target URL can
80  	 * come in as {@code Referer}. {@code ServerRelativeArchivalRedirect} needs
81  	 * to be able to handle it. See ARI-3905.
82  	 */
83  	public void testMalformedTargetURLInReferer() throws Exception {
84  
85  		final String TARGET_URL = "http:/example.com/index.html";
86  		final String REPLAY_BASE = "http://web.archive.org/web/20010203040506/";
87  		final String REFERER = REPLAY_BASE + TARGET_URL;
88  		final String REQUEST_URI = "/js/s_code.js";
89  
90  		final String EXPECTED_REDIRECT_URL = REPLAY_BASE +
91  				"http://example.com/js/s_code.js";
92  
93  		// mocks
94  		setUpRequest(REQUEST_URI, REFERER);
95  
96  		HttpServletResponse response = EasyMock
97  			.createMock(HttpServletResponse.class);
98  		response.addHeader("Vary", "Referer");
99  		EasyMock.expectLastCall().once();
100 		response.sendRedirect(EXPECTED_REDIRECT_URL);
101 		EasyMock.expectLastCall().once();
102 
103 		EasyMock.replay(request, response);
104 
105 		boolean handled = cut.handleRequest(request, response);
106 
107 		EasyMock.verify(response);
108 		assertTrue(handled);
109 	}

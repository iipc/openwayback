34  public class RobotExclusionFilterTest extends TestCase {
35  	
36  	/**
37  	 * 
38  	 */
39  	public void testFoo() {
40  		String re = "^www[0-9]+\\.";
41  		Pattern p = Pattern.compile(re);
42  		String url = "www4.archive.org";
43  		Matcher m = p.matcher(url);
44  		assertTrue(m.find());
45  		
46  	}
47  	
48  	/**
49  	 * 
50  	 */
51  	protected final static String HTTP_PREFIX = "http://";
52  	
53  	public void testSearchResultToRobotUrlStrings() {
54  		RobotExclusionFilter f = new RobotExclusionFilter(null,"",100);
55  		String test1[] = {"www.foo.com","foo.com"};
56  		compareListTo(f.searchResultToRobotUrlStrings("www.foo.com", HTTP_PREFIX),test1);
57  
58  		String test2[] = {"foo.com","www.foo.com"};
59  		compareListTo(f.searchResultToRobotUrlStrings("foo.com", HTTP_PREFIX),test2);
60  
61  		String test3[] = {"fool.foo.com","www.fool.foo.com"};
62  		compareListTo(f.searchResultToRobotUrlStrings("fool.foo.com", HTTP_PREFIX),test3);
63  
64  		String test4[] = {"www.foo.com","foo.com", "www4.foo.com"};
65  		compareListTo(f.searchResultToRobotUrlStrings("www4.foo.com", HTTP_PREFIX),test4);
66  
67  		String test5[] = {"www4w.foo.com"};
68  		compareListTo(f.searchResultToRobotUrlStrings("www4w.foo.com", HTTP_PREFIX),test5);
69  		
70  		String test6[] = {"www.www.foo.com","www.foo.com"};
71  		compareListTo(f.searchResultToRobotUrlStrings("www.www.foo.com", HTTP_PREFIX),test6);
72  	}
73  	
74  	private void compareListTo(List<String> list, String strings[]) {
75  		assertEquals(list.size(), strings.length);
76  		for(int i = 0; i < strings.length; i++) {
77  			String listS = list.get(i);
78  			String arrayS = "http://" + strings[i] + "/robots.txt";
79  			assertEquals(listS, arrayS);
80  		}
81  	}
82  }

20  package org.archive.wayback.archivalurl.requestparser;
21  
22  import org.archive.wayback.archivalurl.ArchivalUrlRequestParser;
23  import org.archive.wayback.core.WaybackRequest;
24  import org.archive.wayback.exception.BetterRequestException;
25  import org.archive.wayback.requestparser.BaseRequestParser;
26  import org.archive.wayback.webapp.AccessPoint;
27  
28  import junit.framework.TestCase;
29  
30  /**
31   *
32   *
33   * @author brad
34   * @version $Date$, $Revision$
35   */
36  
37  public class ReplayRequestParserTest extends TestCase {
38  
39  	/**
40  	 * Test method for {@link org.archive.wayback.archivalurl.requestparser.ReplayRequestParser#parse(java.lang.String)}.
41  	 * @throws BetterRequestException 
42  	 */
43  	public void testParseString() throws Exception {
44  		BaseRequestParser wrapped = new ArchivalUrlRequestParser();
45  		ReplayRequestParser p = new ReplayRequestParser(wrapped);
46  		WaybackRequest r;
47  		AccessPoint ap = null;
48  		r = p.parse("",ap);
49  		assertNull("Should not parse empty string", r);
50  		r = p.parse("20070101000000/foo.com",ap);
51  		assertNotNull("Should parse legit request sans scheme", r);
52  		assertEquals("parsed request Url",r.getRequestUrl(),"http://foo.com");
53  		assertEquals("Parsed timestamp","20070101000000",r.getReplayTimestamp());
54  
55  		r = p.parse("20070101000000/foo.com/",ap);
56  		assertEquals("parsed request Url, maintaining trailing slash",
57  				"http://foo.com/",r.getRequestUrl());
58  
59  		r = p.parse("200701010000/foo.com",ap);
60  		assertEquals("parsed partial date",
61  				"http://foo.com",r.getRequestUrl());
62  		assertEquals("Parsed partial timestamp to earliest",
63  				"20070101000000",r.getReplayTimestamp());
64  
65  		r = p.parse("20070101000000/http://foo.com",ap);
66  		assertEquals("parsed request Url with scheme",
67  				"http://foo.com",r.getRequestUrl());
68  
69  		r = p.parse("20070101000000/http://foo.com/",ap);
70  		assertEquals("parsed request Url with scheme and trailing slash",
71  				"http://foo.com/",r.getRequestUrl());
72  
73  		r = p.parse("20070101000000/ftp://foo.com/",ap);
74  		assertEquals("parsed request Url with ftp scheme",
75  				"ftp://foo.com/",r.getRequestUrl());
76  		
77  		r = p.parse("20070101000000/https://foo.com/",ap);
78  		assertEquals("parsed request Url with https scheme",
79  				"https://foo.com/",r.getRequestUrl());
80  
81  		r = p.parse("20070101000000js_/http://foo.com/",ap);
82  		assertEquals("parsed request Url with js_ flag",
83  				"http://foo.com/",r.getRequestUrl());
84  		assertTrue("parsed js_ flag",r.isJSContext());
85  		assertFalse("css not set",r.isCSSContext());
86  
87  		r = p.parse("20070101000000cs_/http://foo.com/",ap);
88  		assertEquals("parsed request Url with cs_ flag",
89  				"http://foo.com/",r.getRequestUrl());
90  		assertTrue("parsed cs_ flag",r.isCSSContext());
91  		assertFalse("js not set",r.isJSContext());
92  
93  		r = p.parse("20070101000000cs_js_/http://foo.com/",ap);
94  		assertEquals("parsed request Url with cs_ and js_ flags",
95  				"http://foo.com/",r.getRequestUrl());
96  		assertTrue("parsed cs_ flag",r.isCSSContext());
97  		assertTrue("parsed js_ flag",r.isJSContext());
98  
99  		r = p.parse("20070101000000js_cs_/http://foo.com/",ap);
100 		assertEquals("parsed request Url with cs_ and js_ flags, backvards",
101 				"http://foo.com/",r.getRequestUrl());
102 		assertTrue("parsed cs_ flag",r.isCSSContext());
103 		assertTrue("parsed js_ flag",r.isJSContext());
104 
105 		r = p.parse("20070101000000un_/http://foo.com/",ap);
106 		assertEquals("parsed request Url with unknown flag",
107 				"http://foo.com/",r.getRequestUrl());
108 		assertFalse("no cs_ flag",r.isCSSContext());
109 		assertFalse("no js_ flag",r.isJSContext());
110 
111 		r = p.parse("20070101000000un_js_cs_/http://foo.com/",ap);
112 		assertEquals("parsed request Url with falgs and unknown flag",
113 				"http://foo.com/",r.getRequestUrl());
114 		assertTrue("parsed cs_ flag",r.isCSSContext());
115 		assertTrue("parsed js_ flag",r.isJSContext());
116 
117 		r = p.parse("20070101000000js_cs_un_/http://foo.com/",ap);
118 		assertEquals("parsed request Url with falgs and unknown flag at end",
119 				"http://foo.com/",r.getRequestUrl());
120 		assertTrue("parsed cs_ flag",r.isCSSContext());
121 		assertTrue("parsed js_ flag",r.isJSContext());
122 
123 		r = p.parse("20070101000000un_js_cs_un_/http://foo.com/",ap);
124 		assertEquals("parsed request Url with falgs and unknown flags",
125 				"http://foo.com/",r.getRequestUrl());
126 		assertTrue("parsed cs_ flag",r.isCSSContext());
127 		assertTrue("parsed js_ flag",r.isJSContext());
128 
129 	}

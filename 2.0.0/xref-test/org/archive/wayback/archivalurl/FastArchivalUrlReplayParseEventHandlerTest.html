194     /**
195      * URL-rewrite must not unescape HTML entities in URL.
196      * <p>Reported in <a href="https://webarchive.jira.com/browse/ARI-3774">ARI-3774</a>.
197      * Now all attribute values are unescaped before processing, and then escaped back
198      * before writing out.  This has a side-effect: bare "{@code &}" gets rewritten to
199      * "{@code &amp;}"</p>
200      *
201      * @throws Exception
202      */
203 	public void testHTMLEntityInURL() throws Exception {
204 		// note "&amp;amp" - it should appear in translated URL as it does in the original.
205 		final String input = "<html>"
206 				+ "<body>"
207 				+ "<iframe src=\"https://example.com/player/?url=https%3A//api.example.com/"
208 				+ "tracks/135768597%3Ftoken%3Dsss&amp;amp;auto_play=false&bare=1\"></iframe>"
209 				+ "</body>"
210 				+ "</html>";
211 		final String expected = "<html>"
212 				+ "<body>"
213 				+ "<iframe src=\"http://replay.archive.org/2001if_/https://example.com/player/?url=https%3A//api.example.com/"
214 				+ "tracks/135768597%3Ftoken%3Dsss&amp;amp;auto_play=false&amp;bare=1\"></iframe>"
215 				+ "</body>"
216 				+ "</html>";
217 		assertEquals(expected, doEndToEnd(input));
218 	}

855 	public String doEndToEnd(String input) throws Exception {
856 		final String baseUrl = "http://www.example.com/";
857 		final String timestamp = "2001";
858 		final String outputCharset = "UTF-8";
859 		final String charSet = "UTF-8";
860 		
861 		ByteArrayInputStream bais = new ByteArrayInputStream(input.getBytes(charSet));
862 		
863 		ArchivalUrlResultURIConverter uriConverter = new ArchivalUrlResultURIConverter();
864 		uriConverter.setReplayURIPrefix("http://replay.archive.org/");
865 		
866 		ArchivalUrlContextResultURIConverterFactory fact = 
867 			new ArchivalUrlContextResultURIConverterFactory(
868 					(ArchivalUrlResultURIConverter) uriConverter);
869 
870 		// The URL of the page, for resolving in-page relative URLs: 
871     	URL url = new URL(baseUrl);
872 
873 		// To make sure we get the length, we have to buffer it all up...
874 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
875 
876 		// set up the context:
877         ReplayParseContext context = new ReplayParseContext(fact, url,
878                 timestamp);
879 		context.setOutputCharset(outputCharset);
880 		context.setOutputStream(baos);
881 		// jspExec is null for attribute rewrite tests.
882 		context.setJspExec(jspExec);
883 		
884 		// and finally, parse, using the special lexer that knows how to
885 		// handle javascript blocks containing unescaped HTML entities:
886 		Page lexPage = new Page(bais,charSet);
887 		Lexer lexer = new Lexer(lexPage);
888 		Lexer.STRICT_REMARKS = false;
889     	ContextAwareLexer lex = new ContextAwareLexer(lexer, context);
890 
891     	Node node;
892     	while ((node = lex.nextNode()) != null) {
893     	    delegator.handleNode(context, node);
894     	}
895     	delegator.handleParseComplete(context);
896 
897 		// At this point, baos contains the utf-8 encoded bytes of our result:
898 		return new String(baos.toByteArray(),outputCharset);
899 	}

905     /**
906      * test expected behavior of htmlparser.
907      * <p>htmlparser does neither unescape HTML entities found in text, nor
908      * escape special characters in Node.toHtml().  We have a workaround based on this
909      * behavior.  If this expectation breaks, we need to modify our code.</p>
910      * @throws Exception
911      */
912     public void testHtmlParser_attributeValueEscaping() throws Exception {
913     	final String html = "<html>" +
914     			"<body>" +
915     			"<a href=\"http://example.com/api?a=1&amp;b=2&c=3&#34;\">anchor</a>" +
916     			"</body>" +
917     			"</html>";
918     	byte[] bytes = html.getBytes();
919     	ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
920     	Page page = new Page(bais, "UTF-8");
921     	Lexer lexer = new Lexer(page);
922     	Node node;
923     	while ((node = lexer.nextNode()) != null) {
924     		if (node instanceof Tag) {
925     			Tag tag = (Tag)node;
926     			if (tag.getTagName().equalsIgnoreCase("A") && !tag.isEndTag()) {
927     				assertEquals("href", "http://example.com/api?a=1&amp;b=2&c=3&#34;", tag.getAttribute("HREF"));
928 
929     				String htmlout = tag.toHtml();
930     				assertEquals("toHtml output", "<a href=\"http://example.com/api?a=1&amp;b=2&c=3&#34;\">", htmlout);
931     			}
932     		}
933     	}

935     /**
936      * test expected behavior of {@code HTMLParser} - handling of CDATA.
937      * {@code HTMLParser} gives out {@code <![CDATA[ ... ]]>} as single
938      * {@code TagNode}.  Quirk is that {@code tagName} can also include
939      * characters following "<![CDATA[". And apparently there's no way
940      * to update the content.
941      * @throws Exception
942      */
943     public void testHtmlParser_CDATA() throws Exception {
944     	final String html = "<![CDATA[aaaa\nbbbb]]>";
945 
946     	byte[] bytes = html.getBytes();
947     	ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
948     	Page page = new Page(bais, "UTF-8");
949     	Lexer lexer = new Lexer(page);
950     	Node node;
951 
952     	node = lexer.nextNode();
953     	// HTMLParser returns CDATA section as TagNode
954     	assertTrue(node instanceof TagNode);
955     	TagNode tag = (TagNode)node;
956     	// whose tagName is "![CDATA[" *plus* non-whitespace chars following
957     	// it.  And if they are alphabets, they get capitalized.
958     	assertTrue(tag.getTagName().startsWith("![CDATA["));
959     	assertEquals("AAAA", tag.getTagName().substring("![CDATA[".length()));
960     	// Fortunately, getText() returns entire CDATA section, including
961     	// leading "!CDATA[" and trailing "]]" (no < and >), unmodified.
962     	String text = tag.getText();
963     	System.out.println("text=\"" + text + "\" (" + text.length() + " chars)");
964     	assertEquals("![CDATA[aaaa\nbbbb]]", text);
965     	// but setText() results in ClassCastException
966     	try {
967     		tag.setText("![CDATA[bbbb]]");
968     		fail("setText() did not throw an Exception");
969     	} catch (ClassCastException ex) {
970     		// expected
971     	}

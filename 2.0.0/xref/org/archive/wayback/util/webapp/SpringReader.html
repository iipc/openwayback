20  package org.archive.wayback.util.webapp;
21  import java.util.Map;
22  import java.util.logging.Logger;
23  
24  import javax.servlet.ServletContext;
25  
26  import org.springframework.context.ApplicationContext;
27  import org.springframework.context.support.FileSystemXmlApplicationContext;
28  
29  /**
30   * Single static method to read a Spring XML configuration, extract 
31   * RequestHandlers, and return a RequestMapper which delegates requests to
32   * those RequestHandlers.
33   * 
34   * @author brad
35   *
36   */
37  public class SpringReader {
38  	private static final Logger LOGGER = Logger.getLogger(
39  			SpringReader.class.getName());
40  	
41  	protected static ApplicationContext currentContext = null;
42  
43  	/**
44  	 * Read the single Spring XML configuration file located at the specified
45  	 * path, performing PropertyPlaceHolder interpolation, extracting all beans
46  	 * which implement the RequestHandler interface, and construct a 
47  	 * RequestMapper for those RequestHandlers, on the specified ServletContext. 
48  	 * @param configPath the path to the Spring XML file containing the 
49  	 * configuration.
50  	 * @param servletContext the ServletContext where the RequestHandlers should
51  	 * be mapped
52  	 * @return a new ReqeustMapper which delegates requests for the 
53  	 * ServletContext
54  	 */
55  	public static RequestMapper readSpringConfig(String configPath,
56  			ServletContext servletContext) {
57  		LOGGER.info("Loading from config file " + configPath);
58  
59  		currentContext = new FileSystemXmlApplicationContext("file:" + configPath);
60  		Map<String,RequestHandler> beans = 
61  				currentContext.getBeansOfType(RequestHandler.class,false,false);
62  
63  		return new RequestMapper(beans.values(), servletContext);
64  	}
65  	
66  	public static ApplicationContext getCurrentContext() {
67  		return currentContext;
68  	}
69  }

18  	public CDXAccessFilter createAccessFilter(AuthToken token)
19  	{
20  		ExclusionFilter adminFilter = null;
21  		if (adminExclusions != null) {
22  			adminFilter = adminExclusions.get();
23  		}
24  		
25  		ExclusionFilter robotsFilter = null;
26  		if (robotsExclusions != null) {
27  			robotsFilter = robotsExclusions.get();
28  		}
29  		
30  		return new AccessCheckFilter(token, adminFilter, robotsFilter, prefixFilter, null);
31  	}
32

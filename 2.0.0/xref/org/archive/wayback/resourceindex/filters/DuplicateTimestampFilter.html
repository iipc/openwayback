1   package org.archive.wayback.resourceindex.filters;
2   
3   import org.apache.commons.lang.math.NumberUtils;
4   import org.archive.wayback.core.CaptureSearchResult;
5   import org.archive.wayback.util.ObjectFilter;
6   
7   public class DuplicateTimestampFilter implements ObjectFilter<CaptureSearchResult> {
8   
9   	final static int WORST_HTTP_CODE = 9999;
10  	
11  	protected String lastTimestamp;
12  	//protected int lastHttpCode = WORST_HTTP_CODE;
13  	protected int bestHttpCode = WORST_HTTP_CODE;
14  	protected int timestampDedupLength;
15  	
16  	public DuplicateTimestampFilter(int timestampDedupLength)
17  	{
18  		this.timestampDedupLength = timestampDedupLength;
19  	}
20  	
21  	@Override
22  	public int filterObject(CaptureSearchResult o) {
23  		
24  		if (timestampDedupLength <= 0) {
25  			return FILTER_INCLUDE;
26  		}
27  		
28  		String timestamp = o.getCaptureTimestamp();
29  		timestamp = timestamp.substring(0, Math.min(timestampDedupLength, timestamp.length()));
30  		int httpCode = NumberUtils.toInt(o.getHttpCode(), WORST_HTTP_CODE);
31  		
32  		boolean isDupe = false;
33  		
34  		if ((lastTimestamp != null) && timestamp.equals(lastTimestamp)) {
35  			if (httpCode < bestHttpCode) {
36  				bestHttpCode = httpCode;
37  			} else {
38  				isDupe = true;
39  			}
40  		} else {
41  			bestHttpCode = httpCode;
42  		}
43  		
44  		lastTimestamp = timestamp;
45  
46  		
47  		return isDupe ? FILTER_EXCLUDE : FILTER_INCLUDE;
48  	}
49  }

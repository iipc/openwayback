20  package org.archive.wayback.resourceindex.filters;
21  
22  import org.archive.wayback.core.CaptureSearchResult;
23  import org.archive.wayback.util.ObjectFilter;
24  import org.archive.wayback.webapp.AccessPoint;
25  
26  /**
27   * WARC file allows 2 forms of deduplication. The first actually downloads
28   * documents and compares their digest with a database of previous values. When
29   * a new capture of a document exactly matches the previous digest, an 
30   * abbreviated record is stored in the WARC file. The second form uses an HTTP
31   * conditional GET request, sending previous values returned for a given URL
32   * (etag, last-modified, etc). In this case, the remote server either sends a
33   * new document (200) which is stored normally, or the server will return a 
34   * 304 (Not Modified) response, which is stored in the WARC file.
35   * 
36   * For the first record type, the wayback indexer will output a placeholder 
37   * record that includes the digest of the last-stored record. For 304 responses,
38   * the indexer outputs a normal looking record, but the record will have a
39   * SHA1 digest which is easily distinguishable as an "empty" document. The SHA1
40   * is always:
41   * 
42   *   3I42H3S6NNFQ2MSVX7XZKYAYSCX5QBYJ
43   *   
44   * This class will observe a stream of SearchResults, storing the values for
45   * the last seen non-empty SHA1 field. Any subsequent SearchResults with an 
46   * empty SHA1 will be annotated, copying the values from the last non-empty 
47   * record. 
48   * 
49   * This is highly experimental.
50   *
51   * @author brad
52   * @version $Date$, $Revision$
53   */
54  public class ConditionalGetAnnotationFilter 
55  implements ObjectFilter<CaptureSearchResult> {
56  
57  	private final static String EMPTY_VALUE = "-";
58  	private final static String EMPTY_SHA1 = "3I42H3S6NNFQ2MSVX7XZKYAYSCX5QBYJ";
59  
60  	private CaptureSearchResult lastSeen = null;
61  
62  	private int annotate(CaptureSearchResult o) {
63  		if(lastSeen == null) {
64  			// TODO: log missing record digest reference
65  			return FILTER_EXCLUDE;
66  		}
67  		o.setFile(lastSeen.getFile());
68  		o.setOffset(lastSeen.getOffset());
69  		o.setDigest(lastSeen.getDigest());
70  		o.setHttpCode(lastSeen.getHttpCode());
71  		o.setMimeType(lastSeen.getMimeType());
72  		o.setRedirectUrl(lastSeen.getRedirectUrl());
73  		o.flagDuplicateHTTP(lastSeen.getCaptureTimestamp());
74  		return FILTER_INCLUDE;
75  	}
76  
77  	private int remember(CaptureSearchResult o) {
78  		lastSeen = o;
79  		return FILTER_INCLUDE;
80  	}
81  
82  	public int filterObject(CaptureSearchResult o) {
83  		if(o.getFile().equals(EMPTY_VALUE)) {
84  			if(o.getDigest().equals(EMPTY_SHA1)) {
85  				return annotate(o);
86  			}
87  			return FILTER_INCLUDE;
88  		}
89  		return remember(o);
90  	}
91  
92  }

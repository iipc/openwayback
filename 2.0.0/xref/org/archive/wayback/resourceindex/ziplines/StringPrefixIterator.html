22  import java.io.IOException;
23  import java.util.Iterator;
24  
25  import org.archive.util.iterator.CloseableIterator;
26  
27  
28  
29  /**
30   * Iterator<String> decorator, which assumes the decorated is in SORTED order.
31   * This iterator will discard all elements in the iterator LESS than prefix 
32   * constructor argument, return all elements STARTING with prefix, and stop
33   * iterating when a record is GREATER than prefix.
34   * 
35   * @author brad
36   *
37   */
38  public class StringPrefixIterator implements CloseableIterator<String> {
39  	protected String prefix = null;
40  	protected Iterator<String> inner = null;
41  	protected String cachedNext = null;
42  	protected boolean done = false;
43  	protected boolean truncated = false;
44  	
45  	public StringPrefixIterator(Iterator<String> inner, String prefix) {
46  		this.prefix = prefix;
47  		this.inner = inner;
48  		if(inner instanceof ZiplinesChunkIterator) {
49  			truncated = ((ZiplinesChunkIterator)inner).isTruncated();
50  		}
51  	}
52  	public long getTotalMatches() {
53  		return 0 ;
54  	}
55  	public boolean isTruncated() {
56  		return truncated;
57  	}
58  	/* (non-Javadoc)
59  	 * @see java.util.Iterator#hasNext()
60  	 */
61  	public boolean hasNext() {
62  		if(done) return false;
63  		if(cachedNext != null) {
64  			return true;
65  		}
66  		while(inner.hasNext()) {
67  			String tmp = inner.next();
68  			if(tmp.startsWith(prefix)) {
69  				cachedNext = tmp;
70  				return true;
71  			} else if(tmp.compareTo(prefix) > 0) {
72  				done = true;
73  				return false;
74  			}
75  		}
76  		return false;
77  	}
78  	/* (non-Javadoc)
79  	 * @see java.util.Iterator#next()
80  	 */
81  	public String next() {
82  		String tmp = cachedNext;
83  		cachedNext = null;
84  		return tmp;
85  	}
86  	/* (non-Javadoc)
87  	 * @see java.util.Iterator#remove()
88  	 */
89  	public void remove() {
90  		// TODO Auto-generated method stub
91  		
92  	}
93  	/* (non-Javadoc)
94  	 * @see java.io.Closeable#close()
95  	 */
96  	public void close() throws IOException {
97  		if(inner instanceof CloseableIterator) {
98  			CloseableIterator<String> toBeClosed = (CloseableIterator<String>) inner;
99  			toBeClosed.close();
100 		}
101 	}
102 }

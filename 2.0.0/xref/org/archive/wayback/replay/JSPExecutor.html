20  package org.archive.wayback.replay;
21  
22  import java.io.IOException;
23  
24  import javax.servlet.ServletException;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  
28  import org.archive.wayback.ResultURIConverter;
29  import org.archive.wayback.core.CaptureSearchResult;
30  import org.archive.wayback.core.CaptureSearchResults;
31  import org.archive.wayback.core.Resource;
32  import org.archive.wayback.core.UIResults;
33  import org.archive.wayback.core.WaybackRequest;
34  import org.archive.wayback.webapp.PerfWritingHttpServletResponse;
35  
36  /**
37   * Class which encapsulates all Replay context information needed to execute
38   * a .jsp file in the "context" of a particular replay request.
39   * 
40   * This class then manages converting a jsp path into the String it produces.
41   *
42   * @author brad
43   * @version $Date$, $Revision$
44   */
45  
46  public class JSPExecutor {
47  
48  	private HttpServletRequest httpRequest = null;
49  	private HttpServletResponse httpResponse = null;
50  	private UIResults uiResults = null;
51  	
52  	private boolean isAjax = false;
53  	
54  	public JSPExecutor(ResultURIConverter uriConverter,
55  			HttpServletRequest httpRequest, HttpServletResponse httpResponse,
56  			WaybackRequest wbRequest, CaptureSearchResults results, 
57  			CaptureSearchResult result, Resource resource) {
58  		
59  		this.isAjax = wbRequest.isAjaxRequest();
60  
61  		this.httpRequest = httpRequest;
62  		this.httpResponse = httpResponse; 
63  		uiResults = 
64  			new UIResults(wbRequest, uriConverter, results, result, resource);
65  	}
66  	
67  	
68  	public String jspToString(String jspPath) 
69  	throws ServletException, IOException {
70  		
71  		// If ajax request, don't do any jsp insertion
72  		if (isAjax) {
73  			return "";
74  		}
75  		
76  		if (httpResponse instanceof PerfWritingHttpServletResponse) {
77  			uiResults.setPerfResponse((PerfWritingHttpServletResponse)httpResponse);
78  		}
79  
80  		StringHttpServletResponseWrapper wrappedResponse = 
81  			new StringHttpServletResponseWrapper(httpResponse);
82  		uiResults.forward(httpRequest, wrappedResponse, jspPath);
83  		return wrappedResponse.getStringResponse();
84  	}
85  	
86  	public UIResults getUiResults()
87  	{
88  		return uiResults;
89  	}
90  
91  }

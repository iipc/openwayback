1   package org.archive.wayback.replay.html.rewrite;
2   
3   import java.util.List;
4   import java.util.StringTokenizer;
5   
6   import org.archive.wayback.replay.html.ReplayParseContext;
7   import org.archive.wayback.replay.html.StringTransformer;
8   
9   public class RewritingStringTransformer implements StringTransformer {
10  
11  	private List<RewriteRule> policyRules;
12  
13  	public List<RewriteRule> getPolicyRules() {
14  		return policyRules;
15  	}
16  
17  	public void setPolicyRules(List<RewriteRule> policyRules) {
18  		this.policyRules = policyRules;
19  	}
20  
21  	public String transform(ReplayParseContext rpContext, String input) {
22  
23  		if (policyRules == null) {
24  			return input;
25  		}
26  		
27  		if (!rpContext.isInScriptText() && !rpContext.isInJS()) {
28  			return input;
29  		}
30  
31  		String allPolicies = rpContext.getOraclePolicy();
32  
33  		if (allPolicies == null) {
34  			return input;
35  		}
36  
37  		StringTokenizer tokens = new StringTokenizer(allPolicies, ",");
38  
39  		while (tokens.hasMoreElements()) {
40  			String policy = tokens.nextToken();
41  
42  			for (RewriteRule rule : policyRules) {
43  				if (policy.startsWith(rule.getBeanName())) {
44  					input = rule.rewrite(rpContext, policy, input);
45  				}
46  			}
47  		}
48  
49  		return input;
50  	}
51  }

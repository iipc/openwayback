39  public class HttpHeaderOperation {
40  	public final static String HTTP_LENGTH_HEADER = "Content-Length";
41  	public final static String HTTP_LENGTH_HEADER_UP = 
42  		HTTP_LENGTH_HEADER.toUpperCase();
43  	public final static String HTTP_TRANSFER_ENC_HEADER = 
44  		"Transfer-Encoding".toUpperCase();
45  	public final static String HTTP_CHUNKED_ENCODING_HEADER = 
46  		"chunked".toUpperCase();
47  	public final static String HTTP_CONTENT_ENCODING = "Content-Encoding";
48  
49  	
50  	/**
51  	 * @param resource
52  	 * @param httpResponse
53  	 * @throws BadContentException
54  	 */
55  	public static void copyHTTPMessageHeader(Resource resource, 
56  			HttpServletResponse httpResponse) throws BadContentException {
57  
58  		// set status code from original resource (which will definitely confuse
59  		// many clients...)
60  		int code = resource.getStatusCode();
61  		// Only return legit status codes -- don't return any minus
62  		// codes, etc.
63  		if (code <= HttpServletResponse.SC_CONTINUE) {
64  			throw new BadContentException("Bad status code " + code);
65  		}
66  		httpResponse.setStatus(code);
67  	}
68  	
69  	/**
70  	 * @param resource
71  	 * @param result
72  	 * @param uriConverter
73  	 * @param filter 
74  	 * @return
75  	 */
76  	public static Map<String,String> processHeaders(Resource resource, 
77  			CaptureSearchResult result, ResultURIConverter uriConverter, 
78  			HttpHeaderProcessor filter) {
79  		HashMap<String,String> output = new HashMap<String,String>();
80  		
81  		// copy all HTTP headers, as-is, sending "" instead of nulls.
82  		Map<String,String> headers = resource.getHttpHeaders();
83  		if (headers != null) {
84  			Iterator<String> itr = headers.keySet().iterator();
85  			while(itr.hasNext()) {
86  				String key = itr.next();
87  				String value = headers.get(key);
88  				value = (value == null) ? "" : value;
89  				filter.filter(output, key, value, uriConverter, result);
90  			}
91  		}
92  		return output;
93  	}
94  
95  	/**
96  	 * @param headers
97  	 * @param response
98  	 */
99  	public static void sendHeaders(Map<String,String> headers, 
100 			HttpServletResponse response) {
101 		Iterator<String> itr = headers.keySet().iterator();
102 		while(itr.hasNext()) {
103 			String key = itr.next();
104 			String value = headers.get(key);
105 			value = (value == null) ? "" : value;
106 			response.setHeader(key,value);
107 		}
108 	}
109 	
110 	public static String getContentLength(Map<String,String> headers) {
111 		return getHeaderValue(headers,HTTP_LENGTH_HEADER);
112 	}
113 	public static boolean isChunkEncoded(Map<String,String> headers) {
114 		String enc = getHeaderValue(headers,HTTP_TRANSFER_ENC_HEADER);
115 		if(enc != null) {
116 			return enc.toUpperCase().contains(HTTP_CHUNKED_ENCODING_HEADER);
117 		}
118 		return false;
119 	}
120 	public static String getHeaderValue(Map<String,String> headers, String k) {
121 		String value = null;
122 		Iterator<String> itr = headers.keySet().iterator();
123 		String keyUp = k.toUpperCase();
124 		while(itr.hasNext()) {
125 			String key = itr.next();
126 			if(key != null) {
127 				if(key.toUpperCase().contains(keyUp)) {
128 					value = headers.get(key);
129 					break;
130 				}
131 			}
132 		}
133 		return value;
134 	}
135 	
136 	public static boolean removeHeader(Map<String,String> headers, String k) {
137 		Iterator<String> itr = headers.keySet().iterator();
138 		String keyUp = k.toUpperCase();
139 		while(itr.hasNext()) {
140 			String key = itr.next();
141 			if(key != null) {
142 				if (key.toUpperCase().contains(keyUp)) {
143 					itr.remove();
144 					return true;
145 				}
146 			}
147 		}
148 		return false;
149 	}

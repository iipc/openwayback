40  /**
41   * {@link ReplayRenderer} that rewrites URLs found in CSS resource and inserts
42   * {@code jspInserts} at the top of the document.
43   * <p>This ReplayRenderer searches for URLs in CSS document, and rewrites
44   * them with {@link ResultURIConverter} set to {@link TextDocument}.</p>
45   * <p>In fact, this class simply calls {@link TextDocument#resolveCSSUrls()}
46   * for URL rewrites.  Note that ResultURIConverter argument to {@code updatePage}
47   * method is unused.</p>
48   * <p>This class may be used in both Archival-URL and Proxy mode, despite its
49   * name, by choosing appropriate {@code ResultURIConverter}.</p>
50   * <p>There's separate classes for rewriting CSS text embedded
51   * in HTML.  They use their own code for looking up URLs in CSS.</p>
52   * @see TextDocument#resolveCSSUrls()
53   * @see ResultURIConverter
54   * @see org.archive.wayback.replay.html.transformer.BlockCSSStringTransformer
55   * @see org.archive.wayback.replay.html.transformer.InlineCSSStringTransformer
56   * @author brad
57   *
58   */
59  public class ArchivalUrlCSSReplayRenderer extends TextReplayRenderer {
60  
61  	/**
62  	 * @param httpHeaderProcessor which should process HTTP headers
63  	 */
64  	public ArchivalUrlCSSReplayRenderer(HttpHeaderProcessor httpHeaderProcessor) {
65  		super(httpHeaderProcessor);
66  	}
67  
68  	/* (non-Javadoc)
69  	 * @see org.archive.wayback.replay.HTMLReplayRenderer#updatePage(org.archive.wayback.replay.HTMLPage, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse, org.archive.wayback.core.WaybackRequest, org.archive.wayback.core.CaptureSearchResult, org.archive.wayback.core.Resource, org.archive.wayback.ResultURIConverter, org.archive.wayback.core.CaptureSearchResults)
70  	 */
71  	@Override
72  	protected void updatePage(TextDocument page, HttpServletRequest httpRequest,
73  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
74  			CaptureSearchResult result, Resource resource,
75  			ResultURIConverter uriConverter, CaptureSearchResults results)
76  			throws ServletException, IOException {
77  
78  		page.resolveCSSUrls();
79  		// if any CSS-specific jsp inserts are configured, run and insert...
80  		List<String> jspInserts = getJspInserts();
81  
82  		StringBuilder toInsert = new StringBuilder(300);
83  
84  		if (jspInserts != null) {
85  			Iterator<String> itr = jspInserts.iterator();
86  			while (itr.hasNext()) {
87  				toInsert.append(page.includeJspString(itr.next(), httpRequest,
88  						httpResponse, wbRequest, results, result, resource));
89  			}
90  		}
91  
92  //		page.insertAtStartOfDocument(toInsert.toString());
93  		page.insertAtEndOfDocument( toInsert.toString() );
94  	}
95  }

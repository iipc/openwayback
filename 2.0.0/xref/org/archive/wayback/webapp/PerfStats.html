9   public class PerfStats
10  {	
11  	private static final Logger LOGGER = Logger.getLogger(
12  			PerfStats.class.getName());
13  		
14  	public static class PerfStatEntry
15  	{
16  		String name;
17  		long start;
18  		long total;
19  		int count;
20  		boolean isErr;
21  		
22  		public PerfStatEntry(String name)
23  		{
24  			this.name = name;
25  			clear();
26  		}
27  		
28  		public void clear()
29  		{
30  			start = 0;
31  			total = 0;
32  			count = 0;
33  			isErr = false;
34  		}
35  		
36  		public void startNow()
37  		{
38  			if (start > 0) {
39  				isErr = true;
40  			}
41  			start = System.currentTimeMillis();
42  		}
43  		
44  		public long endNow()
45  		{
46  			long elapsed = 0;
47  			
48  			if (start > 0) {
49  				long end = System.currentTimeMillis();
50  				elapsed = (end - start);
51  				count++;
52  			} else {
53  				isErr = true;
54  			}
55  			start = 0;
56  			total += elapsed;
57  			return elapsed;
58  		}
59  		
60  		public String toString()
61  		{
62  			// Skip perf stats that haven't been set at all
63  			if (count == 0 && total == 0) {
64  				return "";
65  			}
66  			
67  			if (start > 0) {
68  				isErr = true;
69  			}
70  			StringBuilder builder = new StringBuilder(name);
71  			builder.append(": ");
72  			builder.append(total);
73  //			builder.append(" ");
74  //			builder.append(count);
75  			if (isErr) {
76  				builder.append(" ERR");
77  			}
78  			return builder.toString();
79  		}
80  		
81  		public long getTotal()
82  		{
83  			return total;
84  		}
85  	}
86  	
87  	static ThreadLocal<Map<String, PerfStatEntry>> perfStats = new ThreadLocal<Map<String, PerfStatEntry>>()
88  	{
89  		@Override
90  		protected Map<String, PerfStatEntry> initialValue() {
91  			return new TreeMap<String, PerfStatEntry>();
92  		}
93  	};
94  	
95  	static ThreadLocal<PerfStatEntry> lastEntry = new ThreadLocal<PerfStatEntry>()
96  	{
97  		@Override
98  		protected PerfStatEntry initialValue() {
99  			return null;
100 		}		
101 	};
102 	
103 	public static PerfStatEntry get(String statName)
104 	{
105 		PerfStatEntry entry = lastEntry.get();
106 		
107 		if ((entry != null) && entry.name.equals(statName)) {
108 			return entry;
109 		}
110 			
111 		entry = perfStats.get().get(statName);
112 		
113 		if (entry == null) {
114 			entry = new PerfStatEntry(statName);
115 			perfStats.get().put(statName, entry);
116 		}
117 		
118 		lastEntry.set(entry);
119 		return entry;
120 	}
121 	
122 	public static void clearAll()
123 	{
124 		lastEntry.set(null);
125 		
126 		for (PerfStatEntry entry : perfStats.get().values()) {
127 			entry.clear();
128 		}
129 	}

144 	}

146 	public static long timeEnd(Enum<?> stat, boolean log)
147 	{
148 		return timeEnd(stat.toString(), log);
149 	}
150 	
151 	public static long timeEnd(String statName)
152 	{
153 		return timeEnd(statName, true);
154 	}
155 	
156 	public static long timeEnd(String statName, boolean log)
157 	{
158 		long elapsed = get(statName).endNow();
159 		
160 		if (log && LOGGER.isLoggable(Level.INFO)) {
161 			StringBuilder builder = new StringBuilder();		
162 			builder.append("WB-PERF\t");
163 			builder.append(statName);
164 			builder.append("\t");
165 			builder.append(elapsed);
166 			LOGGER.info(builder.toString());
167 		}
168 		
169 		return elapsed;
170 	}
171 	
172 	public static String getAllStats()
173 	{
174 		StringBuilder sb = new StringBuilder("[");
175 		boolean first = true;
176 		
177 		for (PerfStatEntry entry : perfStats.get().values()) {
178 			if (entry.count == 0) {
179 				continue;
180 			}
181 			
182 			if (first) {
183 				first = false;
184 			} else {
185 				sb.append(", ");
186 			}
187 			sb.append(entry.toString());
188 		}
189 		
190 		sb.append("]");
191 		return sb.toString();
192 	}

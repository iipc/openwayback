18  public class FilenamePrefixFilter implements CDXFilter {
19  	
20  	private static final Logger LOGGER =
21          Logger.getLogger(FilenamePrefixFilter.class.getName());
22  	
23  	protected String paramFile;
24  	protected int paramIndex = 1;
25  	protected boolean isExclusion = true;
26  	protected char delim = '\t';
27  	
28  	protected String containsMatch;
29  	
30  	protected Set<String> paramSet = null;
31  	protected Pattern patterns[] = null;
32  	
33  	protected List<String> prefixList = null;
34  	
35  	// This method should be set as the init-method in the spring config
36  	// init-method="loadParamFile" when using this Filter
37  	public void loadParamFile() throws IOException
38  	{
39  		SeekableLineReaderFactory fact = null;
40  		SeekableLineReaderIterator iter = null;
41  		
42  		try {	
43  			fact = GeneralURIStreamFactory.createSeekableStreamFactory(paramFile, false);
44  			iter = new SeekableLineReaderIterator(fact.get());
45  			
46  			paramSet = new HashSet<String>();
47  	
48  			while (iter.hasNext()) {
49  				String param = iter.next();
50  				param = param.trim();
51  				
52  				if (param.isEmpty() || param.startsWith("#")) {
53  					continue;
54  				}
55  				
56  				// Use only the first word, ignore the rest
57  				int wordEnd = param.indexOf(delim);
58  				
59  				if (wordEnd > 0) {
60  					param = param.substring(0, wordEnd);
61  				}
62  				
63  				paramSet.add(param);
64  			}
65  		} finally {
66  			if (iter != null) {
67  				iter.close();
68  			}
69  			
70  			if (fact != null) {
71  				fact.close();
72  			}
73  		}
74  	}
75  	
76  	public boolean include(CDXLine line) {
77  		final String file = line.getFilename();
78  		boolean matched = false;
79  		
80  		if (containsMatch != null) {
81  			if (!file.contains(containsMatch)) {
82  				return (isExclusion ? true : false);
83  			}
84  		}
85  		
86  		if (prefixList != null) {	
87  			for (String prefix : prefixList) {
88  				if (file.startsWith(prefix)) {
89  					return (isExclusion ? false : true);
90  				}
91  			}
92  		}
93  		
94  		if (patterns != null) {
95  			for (Pattern pattern : patterns) {
96  				Matcher matcher = pattern.matcher(file);
97  				if (matcher.find()) {
98  					String param = matcher.group(paramIndex);
99  					if (paramSet.contains(param)) {
100 						if (LOGGER.isLoggable(Level.FINE)) {
101 							LOGGER.fine("Excluding (w)arc: " + file);
102 						}
103 						matched = true;
104 						break;
105 					}
106 				}
107 			}
108 		}
109 		
110 		if (isExclusion) {
111 			return (matched ? false : true);	
112 		} else {
113 			return (matched ? true : false);
114 		}
115 	}

167 	public void setPatterns(List<String> patternStrings) {
168 		int size = patternStrings.size();
169 		patterns = new Pattern[size];
170 		for(int i = 0; i < size; i++) {
171 			patterns[i] = Pattern.compile(patternStrings.get(i));
172 		}

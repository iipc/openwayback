11  	class OrigLineDupeTrack extends DupeTrack
12  	{
13  		CDXLine line;
14  	}
15  	
16      protected DupeTrack createDupeTrack()
17      {
18      	return new OrigLineDupeTrack();
19      }
20      
21      protected void handleLine(DupeTrack counter, CDXLine line, boolean isDupe) {
22  	    OrigLineDupeTrack origLineDupeTrack = (OrigLineDupeTrack)counter;
23      	
24  	    CDXLine origLine = null;
25  	    
26      	boolean currIsRevisit = isRevisit(line);
27      	
28      	if ((origLineDupeTrack.line == null) && !currIsRevisit) {
29  	    	origLineDupeTrack.line = line;
30  	    } else {	    	
31  	    	origLine = origLineDupeTrack.line;
32  	    }
33          
34          if ((origLine != null) && currIsRevisit) {
35          	this.fillRevisit(line, origLine);
36          } else {
37          	this.fillBlankOrig(line);
38          }
39      }
